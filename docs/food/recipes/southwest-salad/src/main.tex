\Section{Overview}
Crunchy, creamy, and fresh: chopped romaine with black beans, sweet corn, bell pepper, red onion, and cherry tomatoes, finished with crispy tortilla strips and a bold homemade chipotle ranch dressing.

\Section{At a Glance}
\begin{tabularx}{\textwidth}{@{} l L @{}}
\toprule
\textbf{Yield} & 5 servings \\
\textbf{Prep Time} & 20 minutes \\
\textbf{Total Time} & 20 minutes \\
\textbf{Method} & No-cook assembly (dressing blended) \\
\bottomrule
\end{tabularx}

\Section{Ingredients}

\Subsection{Salad}
\begin{itemize}
  \item 3 hearts romaine, sliced
  \item 10 oz cherry tomatoes, halved
  \item 1 bell pepper, chopped into large chunks
  \item $\tfrac{1}{2}$ red onion, thinly sliced
  \item 2 ears corn, kernels cut from the cob \emph{(or about 2 cups thawed frozen)}
  \item 15 oz can black beans, drained and rinsed
\end{itemize}

\Subsection{For Serving / Toppings}
\begin{itemize}
  \item Avocado slices
  \item Tortilla strips
\end{itemize}

\Subsection{Chipotle Ranch Dressing}
\begin{itemize}
  \item 8 oz sour cream
  \item $\tfrac{3}{4}$ cup mayonnaise
  \item 3 chipotle peppers in adobo sauce
  \item 1\,$\tfrac{1}{2}$ teaspoons garlic powder
  \item 1 teaspoon onion powder
  \item 1 teaspoon dried dill
  \item $\tfrac{1}{4}$ teaspoon salt
  \item 1\,$\tfrac{1}{2}$ tablespoons lime juice
\end{itemize}

\Section{Equipment}
Large salad bowl; cutting board; knife; blender or food processor (for dressing).

\Section{Instructions}
\begin{enumerate}
  \item \textbf{Prep vegetables.} Chop romaine. Halve cherry tomatoes. Chop bell pepper. Thinly slice red onion. Cut corn kernels from the cobs.
  \item \textbf{Beans.} Drain and rinse black beans.
  \item \textbf{Blend dressing.} In a blender or food processor, combine sour cream, mayonnaise, chipotle peppers, garlic powder, onion powder, dill, salt, and lime juice. Blend until smooth and creamy. Taste and adjust salt or lime as desired.%
  \footnote{For a milder dressing, start with 1--2 chipotle peppers and add more to taste.}
  \item \textbf{Assemble.} In a large bowl, layer romaine, black beans, corn, tomatoes, bell pepper, and red onion. Drizzle with chipotle ranch to taste and toss gently.
  \item \textbf{Serve.} Top with tortilla strips and avocado slices. Add extra dressing at the table.
\end{enumerate}

\Section{Make-Ahead \& Storage}
\begin{itemize}
  \item \textbf{Meal prep:} Store components separately (romaine, beans, corn, onion, pepper) in airtight containers up to several days. Cut tomatoes and avocado just before serving.
  \item \textbf{Dressing:} Refrigerate in a jar with a tight lid up to 1 week; shake before using.
\end{itemize}

\Section{Joy's Tips}
\begin{itemize}
  \item Prep the beans, corn, onion, lettuce, and bell pepper ahead and refrigerate in individual bags or containers.
  \item Dress to taste at serving to keep the salad crisp across multiple meals.
\end{itemize}

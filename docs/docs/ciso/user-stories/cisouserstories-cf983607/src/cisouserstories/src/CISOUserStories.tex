  epic={Domain/Chapter or Capability},

  priority={Must},
  sp={3},
  persona={primary persona},

  risks={risks},

}
  \begin{AcceptanceCriteria}
    \Scenario{Happy path}{\Given ...\\ \When ...\\ \Then ...}
    \Scenario{Negative / edge}{\Given ...\\ \When ...\\ \Then ...}
  \end{AcceptanceCriteria}
  \begin{Tasks}
    \item[\cbox] First concrete task (commands/paths/files where useful).
    \item[\cbox] Second concrete task.
    \item[\cbox] Third concrete task.
    \item[\cbox] Validation: job summary/dashboard shows metric(s) A/B/C.
  \end{Tasks}

\section*{Writing Effective User Stories (Quick Guide)}
\begin{InnerBox}
\textbf{INVEST} — Independent, Negotiable, Valuable, Estimable, Small, Testable. \quad
\textbf{3 Cs} — Card, Conversation, Confirmation.
\textbf{Skeletons}
\begin{itemize}
  \item As a [persona], I want to [action] so that [benefit].
  \item When [situation], as [persona], I want [motivation] so I can [outcome].
\end{itemize}
\textbf{Acceptance Criteria Tips}
\begin{itemize}
  \item Prefer observable outcomes; one behavior per scenario.
  \item Cover happy path, negatives, edges; specify data bounds/messages.
  \item Tie to dashboards/metrics where useful.
\end{itemize}
\end{InnerBox}

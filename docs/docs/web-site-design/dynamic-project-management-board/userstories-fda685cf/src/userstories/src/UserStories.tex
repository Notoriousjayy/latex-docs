% --- Story Card macro (9-arg signature) ---
% 1: ID   2: Title   3: Epic/Feature   4: Business Value
% 5: Priority   6: Estimate(SP)   7: Persona   8: Dependencies   9: Assumptions/Risks
\newcommand{\StoryCard}[9]{%
  \newpage
  \begin{tcolorbox}[
    enhanced, breakable,
    colback=gray!2, colframe=gray!50, arc=2pt, boxrule=0.4pt,
    left=8pt,right=8pt,top=8pt,bottom=8pt,
    fonttitle=\bfseries\large,
    title={\textbf{#1}\ \textemdash\ #2},
    colbacktitle=gray!6, coltitle=black,
    borderline west={2pt}{0pt}{MidnightBlue}
  ]
  \small
  \begin{tabularx}{\textwidth}{@{}L{3.2cm}Y@{}}
    Epic / Feature          & #3 \\
    Business Value          & #4 \\
    Priority / Estimate     & \badge{Priority: #5}\ \badge{SP: #6} \\
    Persona                 & #7 \\
    Dependencies            & #8 \\
    Assumptions / Risks     & #9 \\
  \end{tabularx}

  \medskip
  \textbf{Story}\quad
  \emph{As a #7, I want to #2 so that #4.}

  \medskip
  \textbf{Non-Functional}\quad
  \badge{Performance}\ \badge{Security}\ \badge{Reliability}\ \badge{Accessibility}\ \badge{Privacy}\ \badge{i18n}

  \medskip
  \textbf{Acceptance Criteria (BDD)}
  \begin{description}[leftmargin=2.4cm, labelwidth=2.3cm, style=nextline, itemsep=2pt, topsep=2pt]
    \item[\textbf{Scenario}] Happy path
    \item[\textbf{Given}] story context, data, and permissions are in place
    \item[\textbf{When}] the functional objectives below are implemented and exercised
    \item[\textbf{Then}] the stated outcomes are achieved and published with evidence
  \end{description}

  \vspace{0.2\baselineskip}
  {\footnotesize\color{gray!60}\DoR\ \textbullet\ \DoD}
  \end{tcolorbox}
}

% ===== Vision & Strategy =====
\StoryCard{DPMB-001A}{Define Product Vision \& Success KPIs}{Vision \& Strategy}
{create shared direction with measurable outcomes (e.g., cycle time down 20\%, WIP adherence $\geq$ 90\%)}
{Must}{5}
{Product leadership}
{Stakeholder map; analytics baseline}
{KPI sprawl; ensure owners and cadence}
\begin{TasksBox}
  \item Facilitate a cross-functional vision workshop (90 min) and draft a one-page vision.
  \item Select 3--5 KPIs (throughput, cycle time percentiles, WIP adherence, predictability).
  \item Baseline current values; set targets and owners with monthly review cadence.
  \item Publish \texttt{/docs/dpmb/vision.md} and \texttt{/analytics/kpi-baseline.md}.
\end{TasksBox}
\begin{ACBox}
  \item One-page vision approved by Product, Eng, and Delivery leads.
  \item KPI document lists owner, target, and review cadence for each metric.
  \item Baseline snapshot attached and linked from the KPI page.
  \item Calendar invites created for monthly KPI reviews.
\end{ACBox}

\StoryCard{DPMB-001B}{Governance: Decision Log \& Review Cadence}{Vision \& Strategy}
{increase transparency and predictable alignment on product decisions}
{Must}{3}
{Product leadership}
{Vision outcomes}
{Slow decisions; unclear ownership}
\begin{TasksBox}
  \item Create a versioned decision-log template and repo; define decision categories.
  \item Establish monthly KPI reviews and quarterly roadmap check-ins.
  \item Automate decision digests to Slack/email; require Decision IDs in PR template.
  \item Store log at \texttt{/docs/dpmb/decision-log.md}.
\end{TasksBox}
\begin{ACBox}
  \item Decision log repo exists with template and at least 5 exemplar entries.
  \item CI check validates Decision ID presence for PRs touching product goals.
  \item Monthly review cadence on shared calendar with notes linked to the log.
  \item Weekly digest generated to Slack/email with last 7 days of decisions.
\end{ACBox}

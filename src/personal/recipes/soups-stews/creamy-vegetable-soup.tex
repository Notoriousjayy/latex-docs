\Section{Overview}
A cozy, flavor-packed soup with a rich, creamy base and plenty of veggies. Pair with salad, dinner rolls, or serve in a crusty bread bowl for peak comfort.

\Section{At a Glance}
\begin{tabularx}{\textwidth}{@{} l L @{}}

\textbf{Prep Time} & 30 minutes \\

\bottomrule
\end{tabularx}

\Section{Ingredients}

\Subsection{Soup Base}
\begin{itemize}
  \item 1 tablespoon olive oil
  \item 5 tablespoons unsalted butter, divided
  \item 3 cups mirepoix \emph{(or 1 cup diced carrots, 1 cup diced celery, 1 cup diced yellow onion)}
  \item 2 teaspoons minced garlic
  \item 1\,$\tfrac{1}{2}$ pounds baby potatoes (gold or red), diced (about 4 cups)
  \item 1\,$\tfrac{1}{2}$ teaspoons Italian seasoning
  \item $\tfrac{3}{4}$ teaspoon salt
  \item $\tfrac{3}{4}$ teaspoon pepper
  \item 3 cups chicken stock or broth\footnote{Use vegetable broth for a vegetarian version.}
  \item 2 cups frozen broccoli florets, thawed and finely chopped
  \item 1 cup frozen corn
\end{itemize}

\Subsection{Roux \& Finish}
\begin{itemize}
  \item 6 tablespoons all-purpose flour
  \item 3 cups whole milk
  \item $\tfrac{1}{2}$ cup heavy cream
  \item 2 cups shredded sharp Cheddar cheese (freshly grated melts best)
\end{itemize}

\Subsection{For Serving (Optional)}
Hearty buttered bread; fresh herbs (parsley or thyme).

\Section{Equipment}
Large pot; medium saucepan.

\Section{Instructions}
\begin{enumerate}
  \item \textbf{Sauté vegetables.} In a large pot over medium heat, melt 1 tablespoon butter with the olive oil. Add mirepoix (or carrot, celery, onion) and sauté 5--7 minutes until softened. Stir in garlic; cook 30 seconds.
  \item \textbf{Simmer potatoes.} Add potatoes, Italian seasoning, salt, pepper, and stock. Bring to a boil over medium-high heat, then reduce to medium. Cover and cook 10--15 minutes, until vegetables are crisp-tender.
  \item \textbf{Add broccoli \& corn.} Stir in thawed, finely chopped broccoli and the frozen corn. Cook 2--3 minutes until tender and heated through.
  \item \textbf{Make the roux.} In a medium saucepan over medium heat, melt the remaining 4 tablespoons butter. Whisk in flour; cook, whisking, for 1 minute. Slowly pour in the milk, whisking until smooth. Cook, stirring constantly, until thickened and gently boiling. Stir in the heavy cream; remove from heat.
  \item \textbf{Combine \& finish.} Pour the milk mixture into the soup and stir well. Lower the heat; add Cheddar a handful at a time, stirring until melted. Taste and adjust seasoning.
  \item \textbf{Serve.} Ladle into bowls. Garnish with parsley or thyme and serve with toasted, buttered bread.
\end{enumerate}

\Section{Shortcuts}
\begin{itemize}
  \item Use pre-chopped mirepoix (fresh or frozen).
  \item Use jarred minced garlic or a garlic press for quick prep.
\end{itemize}

\Section{What to Serve}
No-knead bread; simple garden or winter salad; grilled cheese; baked sweet potatoes; cornbread.

\Section{Storage}
\begin{itemize}
  \item \textbf{Refrigerate:} Cool completely; store airtight up to 4 days.
  \item \textbf{Freeze:} Up to 3 months (leave headspace for expansion).
  \item \textbf{Reheat:} Thaw overnight in the fridge; warm gently on the stovetop over low heat, stirring. Thin with a splash of broth as needed.
\end{itemize}

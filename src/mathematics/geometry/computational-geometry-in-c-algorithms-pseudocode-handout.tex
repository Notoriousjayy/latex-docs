\subsection{0.2 Numerical discipline}\label{numerical-discipline}

Most algorithms depend on orientation and sidedness predicates. The
practical C programs in this style commonly use integer arithmetic when
possible because geometric predicates are vulnerable to roundoff.

\begin{lstlisting}[language=C]
const double EPS = 1e-12;

int sign(double x) {
    if (x > EPS) return +1;
    if (x < -EPS) return -1;
    return 0;
}

bool equalPoint2(Point2 a, Point2 b) {
    return fabs(a.x - b.x) <= EPS && fabs(a.y - b.y) <= EPS;
}
\end{lstlisting}

\subsection{0.3 2D orientation and area
predicates}\label{d-orientation-and-area-predicates}

\begin{lstlisting}[language=C]
double area2(Point2 a, Point2 b, Point2 c) {
    return (b.x - a.x) * (c.y - a.y)
         - (c.x - a.x) * (b.y - a.y);
}

int areaSign(Point2 a, Point2 b, Point2 c) {
    return sign(area2(a, b, c));
}

bool left(Point2 a, Point2 b, Point2 c) {
    return area2(a, b, c) > EPS;
}

bool leftOn(Point2 a, Point2 b, Point2 c) {
    return area2(a, b, c) >= -EPS;
}

bool collinear(Point2 a, Point2 b, Point2 c) {
    return fabs(area2(a, b, c)) <= EPS;
}
\end{lstlisting}

\subsection{0.4 3D orientation and volume
predicates}\label{d-orientation-and-volume-predicates}

\begin{lstlisting}[language=C]
double dot3(Point3 a, Point3 b) {
    return a.x*b.x + a.y*b.y + a.z*b.z;
}

Point3 cross3(Point3 a, Point3 b) {
    return {
        a.y*b.z - a.z*b.y,
        a.z*b.x - a.x*b.z,
        a.x*b.y - a.y*b.x
    };
}

Point3 sub3(Point3 a, Point3 b) {
    return {a.x-b.x, a.y-b.y, a.z-b.z};
}

double volume6(Point3 a, Point3 b, Point3 c, Point3 d) {
    return dot3(cross3(sub3(b,a), sub3(c,a)), sub3(d,a));
}

int volumeSign(Point3 a, Point3 b, Point3 c, Point3 d) {
    return sign(volume6(a,b,c,d));
}
\end{lstlisting}

\section{1. Polygon algorithms}\label{polygon-algorithms}

\subsection{1.1 Signed area of a simple
polygon}\label{signed-area-of-a-simple-polygon}

\textbf{Purpose.} Compute twice the signed area by summing edge cross
products. The sign indicates orientation.

\begin{lstlisting}[language=C]
double polygonArea2(Polygon P) {
    double sum = 0;

    for (int i = 0; i < P.n; ++i) {
        Point2 a = P.v[i];
        Point2 b = P.v[(i + 1) % P.n];
        sum += a.x * b.y - b.x * a.y;
    }

    return sum;              // actual signed area is sum / 2
}

bool isCCW(Polygon P) {
    return polygonArea2(P) > 0;
}

void ensureCCW(Polygon* P) {
    if (!isCCW(*P)) reverse(P->v, P->n);
}
\end{lstlisting}

\textbf{Complexity.} \passthrough{\lstinline!O(n)!} time and
\passthrough{\lstinline!O(1)!} extra space.

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection{1.2 Point between endpoints on a line
segment}\label{point-between-endpoints-on-a-line-segment}

\begin{lstlisting}[language=C]
bool between(Point2 a, Point2 b, Point2 c) {
    if (!collinear(a, b, c)) return false;

    if (fabs(a.x - b.x) > fabs(a.y - b.y)) {
        return min(a.x,b.x) - EPS <= c.x && c.x <= max(a.x,b.x) + EPS;
    } else {
        return min(a.y,b.y) - EPS <= c.y && c.y <= max(a.y,b.y) + EPS;
    }
}
\end{lstlisting}

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection{1.3 Proper segment
intersection}\label{proper-segment-intersection}

\textbf{Purpose.} Determine whether two open segments cross at an
interior point, excluding endpoint contact and collinear overlap.

\begin{lstlisting}[language=C]
bool intersectProper(Point2 a, Point2 b, Point2 c, Point2 d) {
    if (collinear(a,b,c) || collinear(a,b,d) ||
        collinear(c,d,a) || collinear(c,d,b)) {
        return false;
    }

    bool ab_straddles_cd = left(a,b,c) ^ left(a,b,d);
    bool cd_straddles_ab = left(c,d,a) ^ left(c,d,b);

    return ab_straddles_cd && cd_straddles_ab;
}
\end{lstlisting}

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection{1.4 General segment
intersection}\label{general-segment-intersection}

\begin{lstlisting}[language=C]
bool segmentsIntersect(Point2 a, Point2 b, Point2 c, Point2 d) {
    if (intersectProper(a,b,c,d)) return true;

    if (between(a,b,c)) return true;
    if (between(a,b,d)) return true;
    if (between(c,d,a)) return true;
    if (between(c,d,b)) return true;

    return false;
}
\end{lstlisting}

\textbf{Complexity.} \passthrough{\lstinline!O(1)!}.

\section{2. Polygon triangulation by ear
clipping}\label{polygon-triangulation-by-ear-clipping}

\subsection{2.1 Diagonal-in-cone test}\label{diagonal-in-cone-test}

For a polygon vertex \passthrough{\lstinline!i!}, a candidate diagonal
\passthrough{\lstinline!(i,j)!} must lie inside the local cone at
\passthrough{\lstinline!i!}. Reflex and convex vertices use different
tests.

\begin{lstlisting}[language=C]
bool inCone(Polygon P, int i, int j) {
    int n = P.n;
    Point2 prev = P.v[(i - 1 + n) % n];
    Point2 curr = P.v[i];
    Point2 next = P.v[(i + 1) % n];
    Point2 q    = P.v[j];

    if (leftOn(prev, curr, next)) {
        // Convex vertex: diagonal must be strictly inside the wedge.
        return left(curr, q, prev) && left(q, curr, next);
    } else {
        // Reflex vertex: diagonal is valid unless it lies in the exterior wedge.
        return !(leftOn(curr, q, next) && leftOn(q, curr, prev));
    }
}
\end{lstlisting}

\subsection{2.2 Diagonal-intersection
test}\label{diagonal-intersection-test}

A candidate diagonal must not properly intersect any polygon edge except
the two incident edges.

\begin{lstlisting}[language=C]
bool diagonalie(Polygon P, int i, int j) {
    int n = P.n;
    Point2 a = P.v[i];
    Point2 b = P.v[j];

    for (int k = 0; k < n; ++k) {
        int k1 = (k + 1) % n;

        bool sharesEndpoint =
            k == i || k1 == i || k == j || k1 == j;

        if (sharesEndpoint) continue;

        if (segmentsIntersect(a, b, P.v[k], P.v[k1])) {
            return false;
        }
    }

    return true;
}
\end{lstlisting}

\subsection{2.3 Full valid-diagonal
predicate}\label{full-valid-diagonal-predicate}

\begin{lstlisting}[language=C]
bool isDiagonal(Polygon P, int i, int j) {
    return inCone(P, i, j)
        && inCone(P, j, i)
        && diagonalie(P, i, j);
}
\end{lstlisting}

\subsection{2.4 Initialize ears}\label{initialize-ears}

A vertex \passthrough{\lstinline!i!} is an ear if the diagonal joining
its predecessor and successor is valid.

\begin{lstlisting}[language=C]
void initializeEars(Polygon P, bool ear[0..n-1]) {
    for (int i = 0; i < P.n; ++i) {
        int prev = (i - 1 + P.n) % P.n;
        int next = (i + 1) % P.n;
        ear[i] = isDiagonal(P, prev, next);
    }
}
\end{lstlisting}

\subsection{2.5 Ear-clipping
triangulation}\label{ear-clipping-triangulation}

\begin{lstlisting}[language=C]
list<Triangle2> triangulateByEars(Polygon P) {
    ensureCCW(&P);

    CircularList<int> alive = circularList(0..P.n-1);
    bool ear[0..P.n-1];
    initializeEars(P, ear);

    list<Triangle2> triangles;

    while (alive.size() > 3) {
        int i = findAnyIndex(alive, lambda idx: ear[idx]);

        int prev = alive.prev(i);
        int next = alive.next(i);

        triangles.append({P.v[prev], P.v[i], P.v[next]});

        alive.remove(i);

        int pp = alive.prev(prev);
        int nn = alive.next(next);

        ear[prev] = isDiagonal(P, pp, next);
        ear[next] = isDiagonal(P, prev, nn);
    }

    int a = alive.first();
    int b = alive.next(a);
    int c = alive.next(b);
    triangles.append({P.v[a], P.v[b], P.v[c]});

    return triangles;
}
\end{lstlisting}

\textbf{Invariant.} Each clipped ear is a triangle wholly inside the
current polygon, and deleting it leaves a smaller simple polygon.

\textbf{Complexity.} With the direct diagonal test above,
\passthrough{\lstinline!O(n^3)!} worst case if ears are repeatedly
searched naively; with ear bookkeeping and direct edge scans, typically
stated as \passthrough{\lstinline!O(n^2)!} for the simple
implementation.

\section{3. Partitioning and monotone polygon
triangulation}\label{partitioning-and-monotone-polygon-triangulation}

\subsection{3.1 Detect polygon monotonicity with respect to
x-axis}\label{detect-polygon-monotonicity-with-respect-to-x-axis}

\begin{lstlisting}[language=C]
bool isXMonotone(Polygon P) {
    int left = indexOfMinX(P);
    int right = indexOfMaxX(P);

    // Walk both boundary chains from left to right.
    int i = left;
    while (i != right) {
        int next = (i + 1) % P.n;
        if (P.v[next].x + EPS < P.v[i].x) return false;
        i = next;
    }

    i = left;
    while (i != right) {
        int prev = (i - 1 + P.n) % P.n;
        if (P.v[prev].x + EPS < P.v[i].x) return false;
        i = prev;
    }

    return true;
}
\end{lstlisting}

\subsection{3.2 Triangulate an x-monotone
polygon}\label{triangulate-an-x-monotone-polygon}

\begin{lstlisting}[language=C]
list<Triangle2> triangulateMonotone(Polygon P) {
    VertexRef order[0..P.n-1] = sortVerticesByIncreasingX(P);
    Stack<VertexRef> S;

    S.push(order[0]);
    S.push(order[1]);

    list<Triangle2> triangles;

    for (int k = 2; k < P.n - 1; ++k) {
        VertexRef v = order[k];

        if (oppositeChains(v, S.top())) {
            VertexRef last = S.pop();

            while (!S.empty()) {
                VertexRef u = S.pop();
                triangles.append({v.p, last.p, u.p});
                last = u;
            }

            S.push(order[k-1]);
            S.push(v);
        } else {
            VertexRef last = S.pop();

            while (!S.empty() && diagonalInsideMonotone(P, v, S.top())) {
                VertexRef u = S.pop();
                triangles.append({v.p, last.p, u.p});
                last = u;
            }

            S.push(last);
            S.push(v);
        }
    }

    VertexRef lastVertex = order[P.n - 1];
    S.pop();

    while (S.size() > 1) {
        VertexRef u = S.pop();
        triangles.append({lastVertex.p, u.p, S.top().p});
    }

    return triangles;
}
\end{lstlisting}

\textbf{Complexity.} \passthrough{\lstinline!O(n log n)!} if sorting is
included; \passthrough{\lstinline!O(n)!} after vertices are already
ordered along the monotone chains.

\subsection{3.3 Trapezoidization by vertical
decomposition}\label{trapezoidization-by-vertical-decomposition}

\begin{lstlisting}[language=C]
list<Trapezoid> trapezoidize(Polygon P) {
    Event events = sortVerticesByX(P);
    SweepStatus status;
    list<Trapezoid> cells;

    for (Event e in events) {
        Vertex v = e.vertex;

        Segment edgeAbove = status.edgeImmediatelyAbove(v);
        Segment edgeBelow = status.edgeImmediatelyBelow(v);

        if (edgeAbove != null) {
            addVerticalWall(cells, v, edgeAbove);
        }
        if (edgeBelow != null) {
            addVerticalWall(cells, v, edgeBelow);
        }

        updateSweepStatusForIncidentEdges(status, v);
    }

    return cells;
}
\end{lstlisting}

\textbf{Use.} Trapezoids form a planar subdivision that can be refined
into monotone pieces.

\section{4. Convex hulls in two
dimensions}\label{convex-hulls-in-two-dimensions}

\subsection{4.1 Naive extreme-edge hull}\label{naive-extreme-edge-hull}

An oriented edge \passthrough{\lstinline!(i,j)!} is on the hull if all
other points lie on one side of it.

\begin{lstlisting}[language=C]
list<Edge2> convexHullNaiveEdges(Point2 P[0..n-1]) {
    list<Edge2> hullEdges;

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            if (i == j) continue;

            bool allLeftOrOn = true;

            for (int k = 0; k < n; ++k) {
                if (k == i || k == j) continue;
                if (!leftOn(P[i], P[j], P[k])) {
                    allLeftOrOn = false;
                    break;
                }
            }

            if (allLeftOrOn) {
                hullEdges.append({P[i], P[j]});
            }
        }
    }

    return orderHullEdges(hullEdges);
}
\end{lstlisting}

\textbf{Complexity.} \passthrough{\lstinline!O(n^3)!}.

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection{4.2 Gift wrapping / Jarvis
march}\label{gift-wrapping-jarvis-march}

\begin{lstlisting}[language=C]
list<Point2> convexHullGiftWrap(Point2 P[0..n-1]) {
    int start = indexOfLowestThenLeftmost(P);
    int current = start;

    list<Point2> H;

    do {
        H.append(P[current]);

        int next = anyIndexExcept(current);

        for (int r = 0; r < n; ++r) {
            if (r == current) continue;

            double s = area2(P[current], P[next], P[r]);

            if (s < -EPS ||
                (fabs(s) <= EPS &&
                 distance2(P[current], P[r]) > distance2(P[current], P[next]))) {
                next = r;
            }
        }

        current = next;
    } while (current != start);

    return H;
}
\end{lstlisting}

\textbf{Complexity.} \passthrough{\lstinline!O(nh)!}, where
\passthrough{\lstinline!h!} is the number of hull vertices.

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection{4.3 Graham scan}\label{graham-scan}

\begin{lstlisting}[language=C]
list<Point2> convexHullGraham(Point2 P[0..n-1]) {
    int pivot = indexOfLowestThenLeftmost(P);
    swap(P[0], P[pivot]);

    sort P[1..n-1] by polar angle around P[0],
         breaking ties by increasing distance;

    Stack<Point2> S;
    S.push(P[0]);
    S.push(P[1]);

    for (int i = 2; i < n; ++i) {
        while (S.size() >= 2) {
            Point2 top = S.pop();
            Point2 nextToTop = S.top();

            if (left(nextToTop, top, P[i])) {
                S.push(top);
                break;
            }
        }

        S.push(P[i]);
    }

    return S.asList();
}
\end{lstlisting}

\textbf{Complexity.} \passthrough{\lstinline!O(n log n)!} due to
sorting; \passthrough{\lstinline!O(n)!} scan after sorting.

\subsection{4.4 QuickHull}\label{quickhull}

\begin{lstlisting}[language=C]
list<Point2> convexHullQuickHull(Point2 P[0..n-1]) {
    Point2 a = leftmostPoint(P);
    Point2 b = rightmostPoint(P);

    list<Point2> H;
    H.append(a);

    quickHullSide(P, a, b, +1, H);
    H.append(b);
    quickHullSide(P, b, a, +1, H);

    return H;
}

void quickHullSide(Point2 P[], Point2 a, Point2 b, int side, list<Point2>* H) {
    int far = -1;
    double maxDist = 0;

    for each point p in P {
        double val = side * area2(a, b, p);
        if (val > maxDist) {
            maxDist = val;
            far = index(p);
        }
    }

    if (far == -1) return;

    Point2 c = P[far];

    quickHullSide(P, a, c, sideOf(a,c,b), H);
    H->append(c);
    quickHullSide(P, c, b, sideOf(c,b,a), H);
}
\end{lstlisting}

\textbf{Complexity.} Average-case often good; worst-case
\passthrough{\lstinline!O(n^2)!}.

\subsection{4.5 Incremental 2D hull}\label{incremental-2d-hull}

\subsection{5.3 Incremental 3D convex
hull}\label{incremental-3d-convex-hull}

\begin{lstlisting}[language=C]
Polyhedron convexHullIncremental3D(Point3 P[0..n-1]) {
    Polyhedron H = initialTetrahedron(P);

    for each point p in P not already used {
        list<Face3*> visibleFaces;

        for each face f in H.faces {
            f->visible = faceVisibleFromPoint(f, p);
            if (f->visible) visibleFaces.append(f);
        }

        if (visibleFaces.empty()) {
            continue;       // point lies inside or on current hull
        }

        Vertex3* v = H.addVertex(p);

        for each edge e in H.edges {
            bool leftVisible  = e->adj[0] != null && e->adj[0]->visible;
            bool rightVisible = e->adj[1] != null && e->adj[1]->visible;

            if (leftVisible ^ rightVisible) {
                // Horizon edge: create one new face joining edge to p.
                Face3* nf = makeFaceOrientedOutward(e->a, e->b, v, H);
                e->newFace = nf;
            }
        }

        delete all visible faces from H;
        delete all edges not incident to any remaining face;
        integrate all newly created horizon faces;
    }

    return H;
}
\end{lstlisting}

\textbf{Invariant.} Visible faces form the portion of the old hull
removed by the new point. The boundary between visible and invisible
faces is the horizon cycle.

\textbf{Complexity.} Simple implementation:
\passthrough{\lstinline!O(n^2)!} typical/worst for many inputs, with
storage proportional to final hull size plus intermediate faces.

\subsection{5.4 Delaunay triangulation through
lifting}\label{delaunay-triangulation-through-lifting}

The standard lifting transform maps \passthrough{\lstinline!(x,y)!} to
\passthrough{\lstinline!(x,y,x^2+y^2)!}. The lower convex hull
projects to the Delaunay triangulation.

\begin{lstlisting}[language=C]
list<Triangle2> delaunayByLifting(Point2 P[0..n-1]) {
    Point3 Q[0..n-1];

    for (int i = 0; i < n; ++i) {
        Q[i] = {P[i].x, P[i].y, P[i].x*P[i].x + P[i].y*P[i].y};
    }

    Polyhedron H = convexHullIncremental3D(Q);

    list<Triangle2> DT;

    for each face f in H.faces {
        if (faceNormal(f).z < 0) {       // lower hull face
            DT.append({
                projectXY(f->v[0]->p),
                projectXY(f->v[1]->p),
                projectXY(f->v[2]->p)
            });
        }
    }

    return DT;
}
\end{lstlisting}

\textbf{Predicate equivalent.} The 2D in-circle test corresponds to the
orientation of lifted points in 3D.

\section{6. Voronoi and Delaunay

\subsection{6.1 Circumcenter of a
triangle}\label{circumcenter-of-a-triangle}

\begin{lstlisting}[language=C]
Point2 circumcenter(Point2 a, Point2 b, Point2 c) {
    double d = 2 * (
        a.x*(b.y-c.y) +
        b.x*(c.y-a.y) +
        c.x*(a.y-b.y)
    );

    if (fabs(d) <= EPS) error("degenerate triangle");

    double aa = a.x*a.x + a.y*a.y;
    double bb = b.x*b.x + b.y*b.y;
    double cc = c.x*c.x + c.y*c.y;

    double ux = (aa*(b.y-c.y) + bb*(c.y-a.y) + cc*(a.y-b.y)) / d;
    double uy = (aa*(c.x-b.x) + bb*(a.x-c.x) + cc*(b.x-a.x)) / d;

    return {ux, uy};
}
\end{lstlisting}

\subsection{6.2 Voronoi diagram from Delaunay
triangulation}\label{voronoi-diagram-from-delaunay-triangulation}

\begin{lstlisting}[language=C]
VoronoiDiagram voronoiFromDelaunay(list<Triangle2> DT) {
    map<Triangle2, Point2> center;

    for each triangle t in DT {
        center[t] = circumcenter(t.a, t.b, t.c);
    }

    VoronoiDiagram V;

    for each pair of adjacent Delaunay triangles (t1, t2) sharing edge e {
        V.addEdge(center[t1], center[t2], dualOf(e));
    }

    for each boundary Delaunay edge e incident to triangle t {
        Ray r = outwardPerpendicularRay(e, center[t]);
        V.addRay(r);
    }

    return V;
}
\end{lstlisting}

Arrangement arrangementOfLines(Line L[0..n-1]) {

\subsection{7.2 Bentley-Ottmann-style segment intersection
sweep}\label{bentley-ottmann-style-segment-intersection-sweep}

\begin{lstlisting}[language=C]
list<Point2> allSegmentIntersections(Segment2 S[0..n-1]) {
    PriorityQueue<Event> Q;
    SweepStatus T;
    list<Point2> X;

    for each segment s in S {
        Q.push(leftEndpointEvent(s));
        Q.push(rightEndpointEvent(s));
    }

    while (!Q.empty()) {
        Event e = Q.popMinByXThenY();
        setSweepX(e.x);

        if (e.type == LEFT_ENDPOINT) {
            T.insert(e.segment);

            Segment2 above = T.above(e.segment);
            Segment2 below = T.below(e.segment);

            scheduleIntersection(Q, e.segment, above);
            scheduleIntersection(Q, e.segment, below);
        }

        else if (e.type == RIGHT_ENDPOINT) {
            Segment2 above = T.above(e.segment);
            Segment2 below = T.below(e.segment);

            T.remove(e.segment);
            scheduleIntersection(Q, above, below);
        }

        else if (e.type == INTERSECTION) {
            X.append(e.point);

            Segment2 a = e.segmentA;
            Segment2 b = e.segmentB;

            T.swapOrder(a, b);

            scheduleIntersection(Q, a, T.above(a));
            scheduleIntersection(Q, b, T.below(b));
        }
    }

    return X;
}
\end{lstlisting}

\textbf{Complexity.} \passthrough{\lstinline!O((n+k) log n)!}, where
\passthrough{\lstinline!k!} is the number of reported intersections,
under standard general-position assumptions.

\section{8. Point containment}\label{point-containment}

\subsection{8.1 Point in polygon by ray
crossing}\label{point-in-polygon-by-ray-crossing}

\begin{lstlisting}[language=C]
enum Location { OUTSIDE, INSIDE, BOUNDARY };

Location pointInPolygonCrossing(Polygon P, Point2 q) {
    bool inside = false;

    for (int i = 0; i < P.n; ++i) {
        Point2 a = P.v[i];
        Point2 b = P.v[(i + 1) % P.n];

        if (between(a, b, q)) return BOUNDARY;

        bool crosses =
            ((a.y > q.y) != (b.y > q.y));

        if (crosses) {
            double xAtY = a.x + (q.y - a.y) * (b.x - a.x) / (b.y - a.y);

            if (xAtY > q.x + EPS) {
                inside = !inside;
            }
        }
    }

    return inside ? INSIDE : OUTSIDE;
}
\end{lstlisting}

\textbf{Invariant.} Each proper crossing of a horizontal ray toggles
parity.

\textbf{Complexity.} \passthrough{\lstinline!O(n)!}.

\subsection{8.2 Point in convex polygon by binary
search}\label{point-in-convex-polygon-by-binary-search}

\begin{lstlisting}[language=C]
Location pointInConvexPolygon(Polygon P, Point2 q) {
    if (!leftOn(P.v[0], P.v[1], q)) return OUTSIDE;
    if (!leftOn(P.v[P.n-1], P.v[0], q)) return OUTSIDE;

    int lo = 1;
    int hi = P.n - 1;

    while (hi - lo > 1) {
        int mid = (lo + hi) / 2;

        if (leftOn(P.v[0], P.v[mid], q)) {
            lo = mid;
        } else {
            hi = mid;
        }
    }

    double s = area2(P.v[lo], P.v[lo+1], q);

    if (fabs(s) <= EPS && between(P.v[lo], P.v[lo+1], q)) return BOUNDARY;
    return s >= -EPS ? INSIDE : OUTSIDE;
}
\end{lstlisting}

\textbf{Complexity.} \passthrough{\lstinline!O(log n)!} for a convex
polygon in cyclic order.

Location pointInPolyhedron(Polyhedron H, Point3 q) {
    Ray3 r = chooseGenericRayFrom(q);
    int crossings = 0;

    for each triangular face f in H.faces {
        IntersectionResult hit = intersectRayTriangle(r, f);

        if (hit.type == ON_BOUNDARY) return BOUNDARY;

        if (hit.type == PROPER_INTERSECTION) {
            crossings++;
        }

        // If the ray hits an edge or vertex, perturb ray direction or use
        // symbolic tie-breaking to avoid double counting.
    }

    return (crossings % 2 == 1) ? INSIDE : OUTSIDE;
}

\section{9. Segment-triangle and ray-triangle
intersection}\label{segment-triangle-and-ray-triangle-intersection}

\subsection{9.1 Segment-plane
intersection}\label{segment-plane-intersection}

\begin{lstlisting}[language=C]
bool segmentPlaneIntersection(Point3 p, Point3 q,
                              Point3 a, Point3 b, Point3 c,
                              Point3* out) {
    double vp = volume6(a,b,c,p);
    double vq = volume6(a,b,c,q);

    if (sign(vp) == 0 && sign(vq) == 0) {
        return false;       // coplanar: handle separately
    }

    if (sign(vp) == sign(vq)) {
        return false;       // same open side
    }

    double t = vp / (vp - vq);
    out->x = p.x + t * (q.x - p.x);
    out->y = p.y + t * (q.y - p.y);
    out->z = p.z + t * (q.z - p.z);

    return true;
}
\end{lstlisting}

\subsection{9.2 Point in triangle in 3D by consistent
orientation}\label{point-in-triangle-in-3d-by-consistent-orientation}

\begin{lstlisting}[language=C]
bool pointInTriangle3D(Point3 x, Triangle3 tri) {
    Point3 n = cross3(sub3(tri.b, tri.a), sub3(tri.c, tri.a));

    Point3 c0 = cross3(sub3(tri.b, tri.a), sub3(x, tri.a));
    Point3 c1 = cross3(sub3(tri.c, tri.b), sub3(x, tri.b));
    Point3 c2 = cross3(sub3(tri.a, tri.c), sub3(x, tri.c));

    return dot3(n, c0) >= -EPS &&
           dot3(n, c1) >= -EPS &&
           dot3(n, c2) >= -EPS;
}
\end{lstlisting}

\section{10. Polygon intersection}\label{polygon-intersection}

\subsection{10.1 Convex polygon intersection: advancing-calipers
structure}\label{convex-polygon-intersection-advancing-calipers-structure}

\begin{lstlisting}[language=C]
Polygon intersectConvexPolygons(Polygon P, Polygon Q) {
    ensureCCW(&P);
    ensureCCW(&Q);

    int i = indexOfLowestThenLeftmost(P);
    int j = indexOfLowestThenLeftmost(Q);

    int startI = i;
    int startJ = j;

    Polygon R = emptyPolygon();

    int advances = 0;
    bool first = true;

    while (first || (i != startI || j != startJ)) {
        first = false;

        Point2 p0 = P.v[i];
        Point2 p1 = P.v[(i+1) % P.n];
        Point2 q0 = Q.v[j];
        Point2 q1 = Q.v[(j+1) % Q.n];

        if (segmentsIntersect(p0,p1,q0,q1)) {
            Point2 x = lineSegmentIntersectionPoint(p0,p1,q0,q1);
            appendIfNew(&R, x);
        }

        bool pInsideQ = pointInConvexPolygon(Q, p1) != OUTSIDE;
        bool qInsideP = pointInConvexPolygon(P, q1) != OUTSIDE;

        if (pInsideQ) appendIfNew(&R, p1);
        if (qInsideP) appendIfNew(&R, q1);

        double cross = area2({0,0}, sub2(p1,p0), sub2(q1,q0));
        bool qLeftOfP = leftOn(p0, p1, q1);
        bool pLeftOfQ = leftOn(q0, q1, p1);

        if (cross >= 0) {
            if (qLeftOfP) j = (j + 1) % Q.n;
            else          i = (i + 1) % P.n;
        } else {
            if (pLeftOfQ) i = (i + 1) % P.n;
            else          j = (j + 1) % Q.n;
        }

        advances++;
        if (advances > 2*(P.n + Q.n)) break;
    }

    return removeCollinearDuplicateVertices(R);
}
\end{lstlisting}

\textbf{Complexity.} \passthrough{\lstinline!O(n+m)!} with a fully
refined implementation.

\section{11. Minkowski sums and
convolution}\label{minkowski-sums-and-convolution}

\subsection{11.1 Minkowski sum of two convex
polygons}\label{minkowski-sum-of-two-convex-polygons}

For convex polygons in counterclockwise order, the boundary of the
Minkowski sum is obtained by merging edge-angle sequences.

\begin{lstlisting}[language=C]
Polygon minkowskiSumConvex(Polygon P, Polygon Q) {
    ensureCCW(&P);
    ensureCCW(&Q);

    rotateSoLowestLeftmostFirst(&P);
    rotateSoLowestLeftmostFirst(&Q);

    vector<Point2> EP = edgeVectors(P);
    vector<Point2> EQ = edgeVectors(Q);

    int i = 0, j = 0;
    Point2 current = add2(P.v[0], Q.v[0]);

    Polygon R;
    R.append(current);

    while (i < P.n || j < Q.n) {
        double cross;

        if (i == P.n) cross = -1;
        else if (j == Q.n) cross = +1;
        else cross = area2({0,0}, EP[i], EQ[j]);

        if (j == Q.n || cross > EPS) {
            current = add2(current, EP[i]);
            i++;
        } else if (i == P.n || cross < -EPS) {
            current = add2(current, EQ[j]);
            j++;
        } else {
            current = add2(current, add2(EP[i], EQ[j]));
            i++;
            j++;
        }

        R.append(current);
    }

    removeFinalDuplicate(&R);
    return removeCollinearDuplicateVertices(R);
}
\end{lstlisting}

\textbf{Use in motion planning.} The configuration-space obstacle for
translating a convex robot \passthrough{\lstinline!R!} around an
obstacle \passthrough{\lstinline!O!} is commonly
\passthrough{\lstinline!MinkowskiSum(O, -R)!}.

list<Segment2> minkowskiConvolution(Polygon P, Polygon Q) {
    list<Segment2> C;

    for each directed edge eP = (p_i, p_{i+1}) of P {
        for each vertex q_j of Q {
            if (edgeNormalSupportsAtVertex(eP, Q, q_j)) {
                C.append({add2(p_i, q_j), add2(p_{i+1}, q_j)});
            }
        }
    }

    for each directed edge eQ = (q_j, q_{j+1}) of Q {
        for each vertex p_i of P {
            if (edgeNormalSupportsAtVertex(eQ, P, p_i)) {
                C.append({add2(p_i, q_j), add2(p_i, q_{j+1})});
            }
        }
    }

    return C;
}

\section{12. Motion planning

\subsection{12.1 Translating convex polygonal robot among convex

FreeSpace buildTranslationFreeSpace(Polygon robot, list<Polygon> obstacles) {
    Polygon negRobot = negatePolygon(robot);
    list<Polygon> cObstacles;

    for each obstacle O in obstacles {
        cObstacles.append(minkowskiSumConvex(O, negRobot));

\subsection{12.2 Shortest path in visibility

Path shortestPathAmongPolygonalObstacles(Point2 s, Point2 t, list<Polygon> obstacles) {
    Graph G;
    G.addVertex(s);
    G.addVertex(t);

    for each obstacle vertex v {
        G.addVertex(v);
    }

    for each pair of vertices (u,v) in G {
        Segment2 uv = {u.point, v.point};

        if (segmentLiesInFreeSpace(uv, obstacles)) {
            G.addUndirectedEdge(u, v, distance(u.point, v.point));
        }
    }

    return dijkstra(G, s, t);
}

\subsection{12.4 Multilink planar robot arm
reachability}\label{multilink-planar-robot-arm-reachability}

A common reachability test for a serial chain with link lengths
\passthrough{\lstinline!l[0..n-1]!} asks whether an endpoint distance
\passthrough{\lstinline!d!} is achievable.

\begin{lstlisting}[language=C]
bool multilinkArmCanReach(double length[0..n-1], double d) {
    double sum = 0;
    double maxLen = 0;

    for (int i = 0; i < n; ++i) {
        sum += length[i];
        maxLen = max(maxLen, length[i]);
    }

    double minReach = max(0, maxLen - (sum - maxLen));
    double maxReach = sum;

    return minReach - EPS <= d && d <= maxReach + EPS;
}
\end{lstlisting}

\subsection{12.5 Construct one multilink configuration reaching a
target}\label{construct-one-multilink-configuration-reaching-a-target}

\begin{lstlisting}[language=C]
Point2[] constructArmConfiguration(double L[0..n-1], Point2 target) {
    double d = norm(target);

    if (!multilinkArmCanReach(L, d)) {
        error("target unreachable");
    }

    Point2 joint[0..n];
    joint[0] = {0,0};

    // High-level constructive strategy:
    // 1. Arrange all links along target direction.
    // 2. Fold surplus length by rotating selected links.
    // 3. Solve final triangle for the last two effective links.
    // Details depend on whether one uses cyclic coordinate descent,
    // two-link reduction, or interval-based constructive geometry.

    Vector2 dir = normalize(target);

    for (int i = 1; i <= n; ++i) {
        joint[i] = add2(joint[i-1], scale2(dir, L[i-1]));
    }

    while (distance(joint[n], target) > EPS) {
        for (int k = n-1; k >= 0; --k) {
            rotateSubchainAroundJointTowardTarget(joint, k, target);
        }
    }

    return joint;
}
\end{lstlisting}

\subsection{13.1 In-circle predicate}\label{in-circle-predicate}

\begin{lstlisting}[language=C]
double incircleDet(Point2 a, Point2 b, Point2 c, Point2 d) {
    double ax = a.x - d.x, ay = a.y - d.y;
    double bx = b.x - d.x, by = b.y - d.y;
    double cx = c.x - d.x, cy = c.y - d.y;

    double a2 = ax*ax + ay*ay;
    double b2 = bx*bx + by*by;
    double c2 = cx*cx + cy*cy;

    return ax*(by*c2 - b2*cy)
         - ay*(bx*c2 - b2*cx)
         + a2*(bx*cy - by*cx);
}

bool pointInsideCircumcircle(Point2 a, Point2 b, Point2 c, Point2 d) {
    double det = incircleDet(a,b,c,d);

    if (area2(a,b,c) < 0) det = -det;

    return det > EPS;
}
\end{lstlisting}

\subsection{13.2 Edge flip for local Delaunay
restoration}\label{edge-flip-for-local-delaunay-restoration}

\begin{lstlisting}[language=C]
bool legalizeEdge(Triangulation* T, Edge e) {
    Triangle t1 = e.leftTriangle;
    Triangle t2 = e.rightTriangle;

    if (t2 == null) return false;

    Point2 a = vertexOpposite(t1, e);
    Point2 b = vertexOpposite(t2, e);

    if (pointInsideCircumcircle(t1.p0, t1.p1, t1.p2, b)) {
        flipEdge(T, e);
        return true;
    }

    return false;
}
\end{lstlisting}

\subsection{13.3 Incremental Delaunay triangulation by edge
flips}\label{incremental-delaunay-triangulation-by-edge-flips}

\begin{lstlisting}[language=C]
Triangulation delaunayIncremental2D(Point2 P[0..n-1]) {
    Triangulation T = superTriangleContainingAllPoints(P);

    for each point p in randomized(P) {
        Triangle t = locateTriangleContaining(T, p);
        splitTriangleIntoThree(T, t, p);

        Stack<Edge> S = edgesOppositeInsertedPoint(p);

        while (!S.empty()) {
            Edge e = S.pop();

            if (legalizeEdge(&T, e)) {
                pushEdgesOppositeInsertedPointAfterFlip(S, p, e);
            }
        }
    }

    removeTrianglesIncidentToSuperTriangle(T);
    return T;
}
\end{lstlisting}

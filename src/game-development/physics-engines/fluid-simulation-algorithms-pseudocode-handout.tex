\begin{lstlisting}
void stepIncompressibleFluid(
    MACGrid *vel,
    Grid3f *smoke,
    Grid3f *temperature,
    LevelSet *solidPhi,
    Vec3 gravity,
    float frameDt,
    float density)
{
    float t = 0.0f;

    while (t < frameDt) {
        float dt = chooseFluidDt(vel, gravity, frameDt - t);

        MACGrid oldVel = cloneMACGrid(vel);
        Grid3f oldSmoke = cloneGrid(smoke);
        Grid3f oldTemp = cloneGrid(temperature);

        advectMACVelocity(vel, &oldVel, dt, solidPhi);
        advectScalar(smoke, &oldSmoke, &oldVel, dt, solidPhi);
        advectScalar(temperature, &oldTemp, &oldVel, dt, solidPhi);

        addBodyForce(vel, gravity, dt);
        projectVelocity(vel, solidPhi, density, dt);

        t += dt;
        freeMACGrid(&oldVel);
        freeGrid(&oldSmoke);
        freeGrid(&oldTemp);
    }
}
\end{lstlisting}

\begin{lstlisting}
float chooseFluidDt(MACGrid *vel, Vec3 bodyForce, float remainingFrameDt) {
    float maxSpeed = maxMACSpeed(vel);
    float accel = length3(bodyForce);

    // Estimate speed after acceleration. Avoid zero denominator.
    float predictedSpeed = maxSpeed + sqrtf(maxf(0.0f, 5.0f * vel->dx * accel));
    predictedSpeed = maxf(predictedSpeed, 1.0e-8f);

    float cflDt = 5.0f * vel->dx / predictedSpeed;
    return minf(remainingFrameDt, cflDt);
}
\end{lstlisting}

Vec3 sampleMACVelocity(MACGrid *vel, Vec3 x) {
    Vec3 ux = sub3(x, vel->origin);
    float invDx = 1.0f / vel->dx;

    // u lives at x faces: (i*dx, (j+0.5)dx, (k+0.5)dx)
    Vec3 pu = { ux.x * invDx,
                ux.y * invDx - 0.5f,
                ux.z * invDx - 0.5f };

    // v lives at y faces: ((i+0.5)dx, j*dx, (k+0.5)dx)
    Vec3 pv = { ux.x * invDx - 0.5f,
                ux.y * invDx,
                ux.z * invDx - 0.5f };

    // w lives at z faces: ((i+0.5)dx, (j+0.5)dx, k*dx)
    Vec3 pw = { ux.x * invDx - 0.5f,
                ux.y * invDx - 0.5f,
                ux.z * invDx };

    Vec3 out;
    out.x = trilinearSample(&vel->u, pu);
    out.y = trilinearSample(&vel->v, pv);
    out.z = trilinearSample(&vel->w, pw);
    return out;
}
\end{lstlisting}

Vec3 cellCenteredVelocity(MACGrid *vel, int i, int j, int k) {
    Vec3 out;
    out.x = 0.5f * (getF(&vel->u, i, j, k) + getF(&vel->u, i + 1, j, k));
    out.y = 0.5f * (getF(&vel->v, i, j, k) + getF(&vel->v, i, j + 1, k));
    out.z = 0.5f * (getF(&vel->w, i, j, k) + getF(&vel->w, i, j, k + 1));
    return out;
}

Vec3 backtraceRK2(MACGrid *vel, Vec3 x, float dt) {
    Vec3 v0 = sampleMACVelocity(vel, x);
    Vec3 xMid = sub3(x, mul3(v0, 0.5f * dt));
    Vec3 vMid = sampleMACVelocity(vel, xMid);
    return sub3(x, mul3(vMid, dt));
}

\subsection*{6. RK3 Backtrace for Stronger Swirl Preservation}
\addcontentsline{toc}{subsection}{6. RK3 Backtrace for Stronger Swirl Preservation}

\begin{lstlisting}
Vec3 backtraceRK3(MACGrid *vel, Vec3 x, float dt) {
    Vec3 k1 = mul3(sampleMACVelocity(vel, x), -dt);
    Vec3 x1 = add3(x, mul3(k1, 0.5f));

    Vec3 k2 = mul3(sampleMACVelocity(vel, x1), -dt);
    Vec3 x2 = add3(x, mul3(k2, 0.75f));

    Vec3 k3 = mul3(sampleMACVelocity(vel, x2), -dt);

    Vec3 out = x;
    out = add3(out, mul3(k1, 2.0f / 9.0f));
    out = add3(out, mul3(k2, 3.0f / 9.0f));
    out = add3(out, mul3(k3, 4.0f / 9.0f));
    return out;
}
\end{lstlisting}

void advectScalar(Grid3f *dst, Grid3f *src, MACGrid *vel, float dt, LevelSet *solidPhi) {
    for (int k = 0; k < dst->nz; ++k) {
        for (int j = 0; j < dst->ny; ++j) {
            for (int i = 0; i < dst->nx; ++i) {
                Vec3 x = cellCenter(dst, i, j, k);
                Vec3 xp = backtraceRK2(vel, x, dt);

                if (sampleLevelSet(solidPhi, xp) < 0.0f) {
                    xp = closestFluidBoundaryPoint(solidPhi, xp);
                }

                float q = tricubicSample(src, gridCoords(src, xp));
                setF(dst, i, j, k, q);
            }
        }
    }
}
\end{lstlisting}

void advectMACVelocity(MACGrid *dst, MACGrid *src, float dt, LevelSet *solidPhi) {
    forEachUFace(dst, i, j, k) {
        Vec3 x = uFacePosition(dst, i, j, k);
        Vec3 xp = backtraceRK2(src, x, dt);
        if (sampleLevelSet(solidPhi, xp) < 0.0f) xp = closestFluidBoundaryPoint(solidPhi, xp);
        setF(&dst->u, i, j, k, sampleMACVelocity(src, xp).x);
    }

    forEachVFace(dst, i, j, k) {
        Vec3 x = vFacePosition(dst, i, j, k);
        Vec3 xp = backtraceRK2(src, x, dt);
        if (sampleLevelSet(solidPhi, xp) < 0.0f) xp = closestFluidBoundaryPoint(solidPhi, xp);
        setF(&dst->v, i, j, k, sampleMACVelocity(src, xp).y);
    }

    forEachWFace(dst, i, j, k) {
        Vec3 x = wFacePosition(dst, i, j, k);
        Vec3 xp = backtraceRK2(src, x, dt);
        if (sampleLevelSet(solidPhi, xp) < 0.0f) xp = closestFluidBoundaryPoint(solidPhi, xp);
        setF(&dst->w, i, j, k, sampleMACVelocity(src, xp).z);
    }
}

\addcontentsline{toc}{subsection}{9. Cubic Interpolation Weights}

\begin{lstlisting}
void cubicWeights(float s, float w[4]) {
    float s2 = s * s;
    float s3 = s2 * s;

    w[0] = -s / 3.0f + 0.5f * s2 - s3 / 6.0f;
    w[1] =  1.0f - s2 + 0.5f * (s3 - s);
    w[2] =  s + 0.5f * (s2 - s3);
    w[3] =  (s3 - s) / 6.0f;
}

float tricubicSample(Grid3f *g, Vec3 p) {
    int i0 = floorf(p.x) - 1;
    int j0 = floorf(p.y) - 1;
    int k0 = floorf(p.z) - 1;

    float wx[4], wy[4], wz[4];
    cubicWeights(p.x - floorf(p.x), wx);
    cubicWeights(p.y - floorf(p.y), wy);
    cubicWeights(p.z - floorf(p.z), wz);

    float sum = 0.0f;
    for (int dz = 0; dz < 4; ++dz)
        for (int dy = 0; dy < 4; ++dy)
            for (int dx = 0; dx < 4; ++dx)
                sum += wx[dx] * wy[dy] * wz[dz] * getF(g, i0 + dx, j0 + dy, k0 + dz);
    return sum;
}
\end{lstlisting}

float sdfSphere(Vec3 x, Vec3 c, float r) {

float sdfPlane(Vec3 x, Vec3 p, Vec3 n) {

float sdfAxisAlignedBox(Vec3 x, Vec3 lo, Vec3 hi) {
    bool inside = (lo.x < x.x && x.x < hi.x && lo.y < x.y && x.y < hi.y && lo.z < x.z && x.z < hi.z);
    if (inside) {
        float dx = minf(x.x - lo.x, hi.x - x.x);
        float dy = minf(x.y - lo.y, hi.y - x.y);
        float dz = minf(x.z - lo.z, hi.z - x.z);
        return -minf(dx, minf(dy, dz));
    }

    Vec3 p;
    p.x = clampf(x.x, lo.x, hi.x);
    p.y = clampf(x.y, lo.y, hi.y);
    p.z = clampf(x.z, lo.z, hi.z);
    return length3(sub3(x, p));
}

Vec3 closestPointIterative(LevelSet *phi, Vec3 x, int maxOuter, int maxBacktrack, float eps) {
    Vec3 p = x;
    float ph = sampleLevelSet(phi, p);
    Vec3 grad = sampleLevelSetGradient(phi, p);

    for (int iter = 0; iter < maxOuter; ++iter) {
        float alpha = 1.0f;
        for (int bt = 0; bt < maxBacktrack; ++bt) {
            Vec3 q = sub3(p, mul3(grad, alpha * ph));
            float qPhi = sampleLevelSet(phi, q);

            if (absf(qPhi) < absf(ph)) {
                p = q;
                ph = qPhi;
                grad = sampleLevelSetGradient(phi, p);
                if (absf(ph) < eps) return p;
                break;
            }
            alpha *= 0.7f;
        }
    }
    return p;
}
\end{lstlisting}

void computeDistanceToPoints(Grid3f *phi, Grid3i *closestPoint, Vec3 *points, int pointCount) {
    fillGrid(phi, INF_FLOAT);
    fillGridInt(closestPoint, -1);

    for (int e = 0; e < pointCount; ++e) {
        Vec3 p = points[e];
        CellBounds b = cellsAroundPoint(phi, p);
        forCellsInBounds(b, i, j, k) {
            float d = length3(sub3(cellCenter(phi, i, j, k), p));
            if (d < getF(phi, i, j, k)) {
                setF(phi, i, j, k, d);
                setI(closestPoint, i, j, k, e);
            }
        }
    }

    for (int pass = 0; pass < 2; ++pass) {
        forEachSweepOrder(order) {
            sweepPropagatePointDistance(phi, closestPoint, points, order);
        }
    }
}
\end{lstlisting}

\begin{lstlisting}
void sweepPropagatePointDistance(Grid3f *phi, Grid3i *closestPoint, Vec3 *points, SweepOrder order) {
    forEachCellInOrder(phi, order, i, j, k) {
        forEachSixNeighbor(i, j, k, ni, nj, nk) {
            int e = getI(closestPoint, ni, nj, nk);
            if (e < 0) continue;
            float d = length3(sub3(cellCenter(phi, i, j, k), points[e]));
            if (d < getF(phi, i, j, k)) {
                setF(phi, i, j, k, d);
                setI(closestPoint, i, j, k, e);

        }
    }

\subsection*{16. Point-Triangle Distance}
\addcontentsline{toc}{subsection}{16. Point-Triangle Distance}

\begin{lstlisting}
float pointTriangleDistance(Vec3 x, Vec3 a, Vec3 b, Vec3 c) {
    Vec3 ab = sub3(b, a);
    Vec3 ac = sub3(c, a);
    Vec3 ax = sub3(x, a);

    float d00 = dot3(ab, ab);
    float d01 = dot3(ab, ac);
    float d11 = dot3(ac, ac);
    float d20 = dot3(ax, ab);
    float d21 = dot3(ax, ac);
    float denom = d00 * d11 - d01 * d01;

    float beta = (d11 * d20 - d01 * d21) / denom;
    float gamma = (d00 * d21 - d01 * d20) / denom;
    float alpha = 1.0f - beta - gamma;

    if (alpha >= 0.0f && beta >= 0.0f && gamma >= 0.0f) {
        Vec3 p = add3(add3(mul3(a, alpha), mul3(b, beta)), mul3(c, gamma));
        return length3(sub3(x, p));
    }

    float dab = pointSegmentDistance(x, a, b);
    float dbc = pointSegmentDistance(x, b, c);
    float dca = pointSegmentDistance(x, c, a);
    return minf(dab, minf(dbc, dca));
}
\end{lstlisting}

void redistanceLevelSet(LevelSet *dst, LevelSet *src) {
    forEachCell(&src->phi, i, j, k) {
        float f = getF(&src->phi, i, j, k);
        float best = INF_FLOAT;

        forEachSixNeighbor(i, j, k, ni, nj, nk) {
            float fn = getF(&src->phi, ni, nj, nk);
            if (f * fn <= 0.0f) {
                float theta = absf(f) / (absf(f) + absf(fn));
                best = minf(best, theta * src->phi.dx);
            }
        }

        if (best < INF_FLOAT) setF(&dst->phi, i, j, k, signf(f) * best);
        else setF(&dst->phi, i, j, k, signf(f) * INF_FLOAT);
    }

    for (int pass = 0; pass < 2; ++pass) {
        forEachSweepOrder(order) sweepEikonal(dst, order);
    }
}

\begin{lstlisting}
float eikonalUpdate(float a, float b, float c, float dx) {
    float v[3] = { absf(a), absf(b), absf(c) };
    sortAscending(v, 3);

    float d = v[0] + dx;
    if (d > v[1]) {
        float disc = maxf(0.0f, 2.0f * dx * dx - (v[1] - v[0]) * (v[1] - v[0]));
        d = 0.5f * (v[0] + v[1] + sqrtf(disc));
    }
    if (d > v[2]) {
        float s = v[0] + v[1] + v[2];
        float q = v[0] * v[0] + v[1] * v[1] + v[2] * v[2] - dx * dx;
        d = (s + sqrtf(maxf(0.0f, s * s - 3.0f * q))) / 3.0f;
    }
    return d;
}

\begin{lstlisting}
void extrapolateGridBFS(Grid3f *value, Grid3i *knownMask) {
    Queue q = emptyQueue();
    Grid3i distance = makeIntGrid(value->nx, value->ny, value->nz, INT_MAX);

    forEachCell(value, i, j, k) {
        if (getI(knownMask, i, j, k)) setI(&distance, i, j, k, 0);
    }

    forEachCell(value, i, j, k) {
        if (getI(&distance, i, j, k) == 0) continue;
        if (hasNeighborWithDistance(&distance, i, j, k, 0)) {
            setI(&distance, i, j, k, 1);
            queuePush(&q, cellIndex(i, j, k));
        }
    }

    while (!queueEmpty(&q)) {
        Cell c = queuePop(&q);
        float sum = 0.0f;
        int count = 0;

        forEachSixNeighbor(c.i, c.j, c.k, ni, nj, nk) {
            if (getI(&distance, ni, nj, nk) < getI(&distance, c.i, c.j, c.k)) {
                sum += getF(value, ni, nj, nk);
                count++;
            }
        }
        if (count > 0) setF(value, c.i, c.j, c.k, sum / (float)count);

        forEachSixNeighbor(c.i, c.j, c.k, ni, nj, nk) {
            if (getI(&distance, ni, nj, nk) == INT_MAX) {
                setI(&distance, ni, nj, nk, getI(&distance, c.i, c.j, c.k) + 1);
                queuePush(&q, cellIndex(ni, nj, nk));
            }
        }
    }
}

void buildNegativeDivergence(Grid3f *rhs, MACGrid *vel, Grid3Label *label) {
    float invDx = 1.0f / vel->dx;

    forEachCell(rhs, i, j, k) {
        if (getLabel(label, i, j, k) != CELL_FLUID) {
            setF(rhs, i, j, k, 0.0f);
            continue;
        }

        float div = (getF(&vel->u, i + 1, j, k) - getF(&vel->u, i, j, k)
                   + getF(&vel->v, i, j + 1, k) - getF(&vel->v, i, j, k)
                   + getF(&vel->w, i, j, k + 1) - getF(&vel->w, i, j, k)) * invDx;
        setF(rhs, i, j, k, -div);
    }
}

void addSolidBoundaryTerms(Grid3f *rhs, MACGrid *vel, MACGrid *solidVel, Grid3Label *label) {
    float invDx = 1.0f / vel->dx;

    forEachCell(rhs, i, j, k) {
        if (getLabel(label, i, j, k) != CELL_FLUID) continue;

        if (getLabel(label, i - 1, j, k) == CELL_SOLID)
            addF(rhs, i, j, k, -invDx * (getF(&vel->u, i, j, k) - getF(&solidVel->u, i, j, k)));
        if (getLabel(label, i + 1, j, k) == CELL_SOLID)
            addF(rhs, i, j, k,  invDx * (getF(&vel->u, i + 1, j, k) - getF(&solidVel->u, i + 1, j, k)));

        if (getLabel(label, i, j - 1, k) == CELL_SOLID)
            addF(rhs, i, j, k, -invDx * (getF(&vel->v, i, j, k) - getF(&solidVel->v, i, j, k)));
        if (getLabel(label, i, j + 1, k) == CELL_SOLID)
            addF(rhs, i, j, k,  invDx * (getF(&vel->v, i, j + 1, k) - getF(&solidVel->v, i, j + 1, k)));

        if (getLabel(label, i, j, k - 1) == CELL_SOLID)
            addF(rhs, i, j, k, -invDx * (getF(&vel->w, i, j, k) - getF(&solidVel->w, i, j, k)));
        if (getLabel(label, i, j, k + 1) == CELL_SOLID)
            addF(rhs, i, j, k,  invDx * (getF(&vel->w, i, j, k + 1) - getF(&solidVel->w, i, j, k + 1)));
    }
}

\begin{lstlisting}
void buildPressureMatrix(Grid3f *diag, Grid3f *plusX, Grid3f *plusY, Grid3f *plusZ,
                         Grid3Label *label, float dt, float density, float dx) {
    float scale = dt / (density * dx * dx);
    clearGrid(diag); clearGrid(plusX); clearGrid(plusY); clearGrid(plusZ);

    forEachCell(diag, i, j, k) {
        if (getLabel(label, i, j, k) != CELL_FLUID) continue;

        forEachAxisNeighbor(axis, ni, nj, nk) {
            CellLabel n = getLabel(label, ni, nj, nk);
            if (n == CELL_FLUID) {
                addF(diag, i, j, k, scale);
                if (axis == POS_X) setF(plusX, i, j, k, -scale);
                if (axis == POS_Y) setF(plusY, i, j, k, -scale);
                if (axis == POS_Z) setF(plusZ, i, j, k, -scale);
            } else if (n == CELL_EMPTY) {
                addF(diag, i, j, k, scale);     // Dirichlet p=0 neighbor
            } else {
                // Solid Neumann condition: no diagonal increment from the solid side.
            }
        }
    }
}

\subsection*{28. Preconditioned Conjugate Gradient}
\addcontentsline{toc}{subsection}{28. Preconditioned Conjugate Gradient}

\begin{lstlisting}
int solvePCG(Vector *p, Vector *b, LinearOperator A, Preconditioner M, double tol, int maxIters) {
    Vector r = cloneVector(b);              // p starts at zero
    Vector z = applyPreconditioner(M, &r);
    Vector s = cloneVector(&z);
    double sigma = dotVector(&z, &r);

    if (maxAbsVector(&r) <= tol) return 0;

    for (int iter = 1; iter <= maxIters; ++iter) {
        Vector As = applyA(A, &s);
        double alpha = sigma / dotVector(&As, &s);

        axpy(p,  alpha, &s);                // p += alpha*s
        axpy(&r, -alpha, &As);              // r -= alpha*A*s

        if (maxAbsVector(&r) <= tol) return iter;

        destroyVector(&z);
        z = applyPreconditioner(M, &r);
        double sigmaNew = dotVector(&z, &r);
        double beta = sigmaNew / sigma;

        scaleVector(&s, beta);
        axpy(&s, 1.0, &z);                  // s = z + beta*s
        sigma = sigmaNew;
        destroyVector(&As);
    }
    return -1;
}
\end{lstlisting}

\subsection*{29. Modified Incomplete Cholesky MIC(0) Factor Setup}
\addcontentsline{toc}{subsection}{29. Modified Incomplete Cholesky MIC(0) Factor Setup}

\begin{lstlisting}
void buildMIC0Preconditioner(Grid3f *precon, Grid3f *diag, Grid3f *plusX, Grid3f *plusY, Grid3f *plusZ,
                             Grid3Label *label) {
    const double tau = 0.97;
    const double sigma = 0.25;

    forEachCell(precon, i, j, k) {
        if (getLabel(label, i, j, k) != CELL_FLUID) {
            setF(precon, i, j, k, 0.0f);
            continue;
        }

        double e = getF(diag, i, j, k);

        double axm = getF(plusX, i - 1, j, k);
        double aym = getF(plusY, i, j - 1, k);
        double azm = getF(plusZ, i, j, k - 1);
        double pxm = getF(precon, i - 1, j, k);
        double pym = getF(precon, i, j - 1, k);
        double pzm = getF(precon, i, j, k - 1);

        e -= axm * axm * pxm * pxm;
        e -= aym * aym * pym * pym;
        e -= azm * azm * pzm * pzm;

        e -= tau * axm * (getF(plusY, i - 1, j, k) + getF(plusZ, i - 1, j, k)) * pxm * pxm;
        e -= tau * aym * (getF(plusX, i, j - 1, k) + getF(plusZ, i, j - 1, k)) * pym * pym;
        e -= tau * azm * (getF(plusX, i, j, k - 1) + getF(plusY, i, j, k - 1)) * pzm * pzm;

        if (e < sigma * getF(diag, i, j, k)) e = getF(diag, i, j, k);
        setF(precon, i, j, k, (float)(1.0 / sqrt(e)));
    }
}

\subsection*{30. Apply MIC(0) Preconditioner}

\begin{lstlisting}
void applyMIC0(Grid3f *z, Grid3f *r, Grid3f *precon, Grid3f *plusX, Grid3f *plusY, Grid3f *plusZ,
               Grid3Label *label) {
    Grid3f q = makeGridLike(r);

    forEachCellForward(r, i, j, k) {
        if (getLabel(label, i, j, k) != CELL_FLUID) continue;

        double t = getF(r, i, j, k);
        t -= getF(plusX, i - 1, j, k) * getF(precon, i - 1, j, k) * getF(&q, i - 1, j, k);
        t -= getF(plusY, i, j - 1, k) * getF(precon, i, j - 1, k) * getF(&q, i, j - 1, k);
        t -= getF(plusZ, i, j, k - 1) * getF(precon, i, j, k - 1) * getF(&q, i, j, k - 1);
        setF(&q, i, j, k, (float)(t * getF(precon, i, j, k)));
    }

    forEachCellBackward(r, i, j, k) {
        if (getLabel(label, i, j, k) != CELL_FLUID) continue;

        double t = getF(&q, i, j, k);
        t -= getF(plusX, i, j, k) * getF(precon, i, j, k) * getF(z, i + 1, j, k);
        t -= getF(plusY, i, j, k) * getF(precon, i, j, k) * getF(z, i, j + 1, k);
        t -= getF(plusZ, i, j, k) * getF(precon, i, j, k) * getF(z, i, j, k + 1);
        setF(z, i, j, k, (float)(t * getF(precon, i, j, k)));
    }

    freeGrid(&q);
}

void enforceCompatibility(Grid3f *rhs, Grid3Label *label) {
    int *component = floodFillFluidComponents(label);
    int compCount = maxComponent(component) + 1;

    double *sum = calloc(compCount, sizeof(double));
    int *count = calloc(compCount, sizeof(int));

    forEachCell(rhs, i, j, k) {
        int c = component[index3(i, j, k, rhs->nx, rhs->ny)];
        if (c >= 0) {
            sum[c] += getF(rhs, i, j, k);
            count[c]++;
        }
    }

    forEachCell(rhs, i, j, k) {
        int c = component[index3(i, j, k, rhs->nx, rhs->ny)];
        if (c >= 0 && count[c] > 0) {
            addF(rhs, i, j, k, (float)(-sum[c] / count[c]));
        }
    }
}

        CellLabel b = getLabel(label, i, j, k);

        else if (a == CELL_FLUID || b == CELL_FLUID) {

            addF(&vel->u, i, j, k, -scale * (pR - pL));
        }
    }

    forEachVFace(vel, i, j, k) { /* same idea with y-neighbor pressures */ }
    forEachWFace(vel, i, j, k) { /* same idea with z-neighbor pressures */ }
}

\begin{lstlisting}
void projectVelocity(MACGrid *vel, LevelSet *solidPhi, float density, float dt) {
    Grid3Label label = classifyDomainForProjection(vel, solidPhi);
    MACGrid solidVel = buildSolidVelocityField(vel, solidPhi);
    Grid3f rhs = makeCellGrid(vel);
    Grid3f pressure = makeCellGrid(vel);
    Grid3f diag = makeCellGrid(vel), plusX = makeCellGrid(vel), plusY = makeCellGrid(vel), plusZ = makeCellGrid(vel);
    Grid3f precon = makeCellGrid(vel);

    buildNegativeDivergence(&rhs, vel, &label);
    addSolidBoundaryTerms(&rhs, vel, &solidVel, &label);
    enforceCompatibility(&rhs, &label);
    buildPressureMatrix(&diag, &plusX, &plusY, &plusZ, &label, dt, density, vel->dx);
    buildMIC0Preconditioner(&precon, &diag, &plusX, &plusY, &plusZ, &label);
    solveStructuredPCG(&pressure, &rhs, &diag, &plusX, &plusY, &plusZ, &precon, &label);
    applyPressure(vel, &pressure, &label, &solidVel, dt, density);

    extrapolateMACVelocityIntoInvalidRegions(vel, &label);
}

\subsection*{34. Ghost-Fluid Free-Surface Pressure Update}
\addcontentsline{toc}{subsection}{34. Ghost-Fluid Free-Surface Pressure Update}

\begin{lstlisting}
float ghostFluidPressureScale(float phiFluid, float phiAir, float dx) {
    float theta = phiFluid / (phiFluid - phiAir);
    theta = clampf(theta, 0.01f, 1.0f);
    return 1.0f / (theta * dx);
}

void applyGhostFluidFreeSurfaceU(MACGrid *vel, Grid3f *p, LevelSet *liquidPhi,
                                 int i, int j, int k, float dt, float rho) {
    float phiL = getF(&liquidPhi->phi, i - 1, j, k);
    float phiR = getF(&liquidPhi->phi, i, j, k);

    if (phiL <= 0.0f && phiR > 0.0f) {
        float s = ghostFluidPressureScale(phiL, phiR, vel->dx);
        addF(&vel->u, i, j, k, -(dt / rho) * (0.0f - getF(p, i - 1, j, k)) * s);
    } else if (phiR <= 0.0f && phiL > 0.0f) {
        float s = ghostFluidPressureScale(phiR, phiL, vel->dx);
        addF(&vel->u, i, j, k, -(dt / rho) * (getF(p, i, j, k) - 0.0f) * s);
    }
}
\end{lstlisting}

    }

\begin{lstlisting}
void transferParticlesToGrid(MACGrid *gridVel, Grid3f *weightU, Grid3f *weightV, Grid3f *weightW,
                             ParticleSet *particles) {
    clearMACGrid(gridVel);
    clearGrid(weightU); clearGrid(weightV); clearGrid(weightW);

    for (int p = 0; p < particles->count; ++p) {
        Particle *pt = &particles->p[p];

        splatToFaceGrid(&gridVel->u, weightU, pt->x, pt->v.x);
        splatToFaceGrid(&gridVel->v, weightV, pt->x, pt->v.y);
        splatToFaceGrid(&gridVel->w, weightW, pt->x, pt->v.z);
    }

    normalizeByWeights(&gridVel->u, weightU);
    normalizeByWeights(&gridVel->v, weightV);
    normalizeByWeights(&gridVel->w, weightW);
}
\end{lstlisting}

void splatToFaceGrid(Grid3f *faceValues, Grid3f *weights, Vec3 x, float value) {
    Stencil s = interpolationStencil(faceValues, x);
    for (int n = 0; n < s.count; ++n) {
        addF(faceValues, s.i[n], s.j[n], s.k[n], s.w[n] * value);
        addF(weights,    s.i[n], s.j[n], s.k[n], s.w[n]);
    }
}

void reseedParticles(ParticleSet *particles, LevelSet *fluidPhi, int minPerCell, int maxPerCell) {
    Grid3i counts = countParticlesPerCell(particles, &fluidPhi->phi);

    forEachCell(&fluidPhi->phi, i, j, k) {
        if (getF(&fluidPhi->phi, i, j, k) >= 0.0f) continue;

        int n = getI(&counts, i, j, k);
        while (n < minPerCell) {
            Vec3 x = randomPointInCell(&fluidPhi->phi, i, j, k);
            if (sampleLevelSet(fluidPhi, x) < 0.0f) {
                Particle q = makeParticleAt(x);
                q.v = interpolateCurrentVelocityAt(x);
                addParticle(particles, q);
                n++;
            }
        }
        if (n > maxPerCell) deleteRandomParticlesInCell(particles, i, j, k, n - maxPerCell);
    }
}

void stepThinFlame(LevelSet *fuelPhi, Grid3f *temperature, Grid3f *smoke, MACGrid *vel,
                   float flameSpeed, float heatRelease, float sootYield, float dt) {
    LevelSet oldFuel = cloneLevelSet(fuelPhi);

    // Move premixed flame front by advection plus normal propagation.
    advectLevelSet(fuelPhi, &oldFuel, vel, dt);
    forEachCell(&fuelPhi->phi, i, j, k) {
        Vec3 n = normalize3(sampleLevelSetGradient(fuelPhi, cellCenter(&fuelPhi->phi, i, j, k)));
        float phi = getF(&fuelPhi->phi, i, j, k);
        setF(&fuelPhi->phi, i, j, k, phi - flameSpeed * dt);
    }
    redistanceLevelSet(fuelPhi, fuelPhi);

    // Burn cells crossed by the front.
    forEachCell(&fuelPhi->phi, i, j, k) {
        float before = getF(&oldFuel.phi, i, j, k);
        float after = getF(&fuelPhi->phi, i, j, k);
        if (before < 0.0f && after >= 0.0f) {
            addF(temperature, i, j, k, heatRelease);
            addF(smoke, i, j, k, sootYield);
        }
    }
}

        }
    }

\subsection*{55. Build Implicit Viscosity System}
\addcontentsline{toc}{subsection}{55. Build Implicit Viscosity System}

\begin{lstlisting}
void buildViscositySystem(CSRMatrix *A, Vector *b, MACGrid *vel, LevelSet *fluidPhi,
                          LevelSet *solidPhi, Grid3f *mu, float dt, float rho) {
    // Unknowns are active u, v, w face velocities. The matrix is SPD when constructed
    // from the discrete viscous energy terms.
    initSparseSystemForActiveFaces(A, b, vel, fluidPhi, solidPhi);

    forEachActiveUFace(vel, fluidPhi, i, j, k) {
        int row = unknownIndexU(i, j, k);
        addIdentityTerm(A, row, rho / dt);
        addViscousDerivativeTermsForU(A, row, i, j, k, mu, solidPhi);
        b->a[row] = (rho / dt) * getF(&vel->u, i, j, k);
    }

    forEachActiveVFace(vel, fluidPhi, i, j, k) {
        int row = unknownIndexV(i, j, k);
        addIdentityTerm(A, row, rho / dt);
        addViscousDerivativeTermsForV(A, row, i, j, k, mu, solidPhi);
        b->a[row] = (rho / dt) * getF(&vel->v, i, j, k);
    }

    forEachActiveWFace(vel, fluidPhi, i, j, k) {
        int row = unknownIndexW(i, j, k);
        addIdentityTerm(A, row, rho / dt);
        addViscousDerivativeTermsForW(A, row, i, j, k, mu, solidPhi);
        b->a[row] = (rho / dt) * getF(&vel->w, i, j, k);
    }
}
\end{lstlisting}

\subsection*{58. Compute Cell-Centered Vorticity}

\begin{lstlisting}
void computeVorticity(Grid3Vec3 *omega, MACGrid *vel) {
    forEachCellVec3(omega, i, j, k) {
        float dw_dy = (sampleWAtCell(vel, i, j + 1, k) - sampleWAtCell(vel, i, j - 1, k)) / (2.0f * vel->dx);
        float dv_dz = (sampleVAtCell(vel, i, j, k + 1) - sampleVAtCell(vel, i, j, k - 1)) / (2.0f * vel->dx);
        float du_dz = (sampleUAtCell(vel, i, j, k + 1) - sampleUAtCell(vel, i, j, k - 1)) / (2.0f * vel->dx);
        float dw_dx = (sampleWAtCell(vel, i + 1, j, k) - sampleWAtCell(vel, i - 1, j, k)) / (2.0f * vel->dx);
        float dv_dx = (sampleVAtCell(vel, i + 1, j, k) - sampleVAtCell(vel, i - 1, j, k)) / (2.0f * vel->dx);
        float du_dy = (sampleUAtCell(vel, i, j + 1, k) - sampleUAtCell(vel, i, j - 1, k)) / (2.0f * vel->dx);

        setVec3(omega, i, j, k, (Vec3){ dw_dy - dv_dz, du_dz - dw_dx, dv_dx - du_dy });
    }
}
\end{lstlisting}

\Needspace{6\baselineskip}
\subsection*{59. Vorticity Confinement Force}
\addcontentsline{toc}{subsection}{59. Vorticity Confinement Force}

\begin{lstlisting}
void addVorticityConfinement(MACGrid *vel, float epsilon, float dt) {
    Grid3Vec3 omega = makeVec3CellGrid(vel);
    Grid3f mag = makeCellGrid(vel);
    computeVorticity(&omega, vel);

    forEachCell(&mag, i, j, k) setF(&mag, i, j, k, length3(getVec3(&omega, i, j, k)));

    Grid3Vec3 force = makeVec3CellGrid(vel);
    forEachCell(&mag, i, j, k) {
        Vec3 grad;
        grad.x = (getF(&mag, i + 1, j, k) - getF(&mag, i - 1, j, k)) / (2.0f * vel->dx);
        grad.y = (getF(&mag, i, j + 1, k) - getF(&mag, i, j - 1, k)) / (2.0f * vel->dx);
        grad.z = (getF(&mag, i, j, k + 1) - getF(&mag, i, j, k - 1)) / (2.0f * vel->dx);
        Vec3 N = normalize3(add3(grad, (Vec3){1.0e-20f, 1.0e-20f, 1.0e-20f}));
        Vec3 f = mul3(cross3(N, getVec3(&omega, i, j, k)), epsilon * vel->dx);
        setVec3(&force, i, j, k, f);
    }

    addCellForceToMAC(vel, &force, dt);
}

void stepShallowWater(Grid2f *height, Grid2f *bottom, Grid2Vec2 *vel, float dt, float dx) {
    Grid2f oldH = cloneGrid2(height);
    Grid2Vec2 oldU = cloneGrid2Vec2(vel);

    forEachCell2(height, i, j) {
        float h = get2(height, i, j);
        Vec2 u = get2v(vel, i, j);

        float fluxX = derivativeXOfHU(&oldH, &oldU, i, j, dx);
        float fluxZ = derivativeZOfHU(&oldH, &oldU, i, j, dx);
        set2(height, i, j, h - dt * (fluxX + fluxZ));
    }

    forEachCell2(height, i, j) {
        Vec2 u = get2v(&oldU, i, j);
        float etaL = get2(&oldH, i - 1, j) + get2(bottom, i - 1, j);
        float etaR = get2(&oldH, i + 1, j) + get2(bottom, i + 1, j);
        float etaD = get2(&oldH, i, j - 1) + get2(bottom, i, j - 1);
        float etaU = get2(&oldH, i, j + 1) + get2(bottom, i, j + 1);
        u.x += -dt * GRAVITY * (etaR - etaL) / (2.0f * dx);
        u.y += -dt * GRAVITY * (etaU - etaD) / (2.0f * dx);
        set2v(vel, i, j, u);
    }
}

void evaluateOceanSurface(ComplexGrid *H0, Grid2f *height, Grid2Vec2 *horizontalDisp,
                          float t, float L, float chop) {
    ComplexGrid Ht = makeComplexGridLike(H0);
    ComplexGrid Dx = makeComplexGridLike(H0);
    ComplexGrid Dz = makeComplexGridLike(H0);

    forEachFrequency(H0, i, j) {
        Vec2 k = frequencyVector(i, j, L);
        float omega = sqrtf(GRAVITY * length2(k));
        Complex h = addComplex(complexMul(getComplex(H0, i, j), expI( omega * t)),
                               complexMul(conjComplex(getComplex(H0, -i, -j)), expI(-omega * t)));
        setComplex(&Ht, i, j, h);

        if (length2(k) > 0.0f) {
            Complex factor = complexMul((Complex){0.0f, 1.0f}, h);
            setComplex(&Dx, i, j, complexScale(factor, chop * k.x / length2(k)));
            setComplex(&Dz, i, j, complexScale(factor, chop * k.y / length2(k)));
        }
    }

    inverseFFT2(&Ht, height);
    inverseFFT2Vec(&Dx, &Dz, horizontalDisp);
}

void accumulatePressureForceTorque(RigidBody *body, Grid3f *pressure, FaceFractionGrid *solidFractions, Vec3 center) {
    Vec3 F = {0,0,0};
    Vec3 T = {0,0,0};

    forEachCell(pressure, i, j, k) {
        Vec3 netAreaNormal = cellSolidAreaNormal(solidFractions, i, j, k);

        T = add3(T, cross3(sub3(x, center), force));
    }

    body->force = add3(body->force, F);
    body->torque = add3(body->torque, T);
}

void immersedBoundaryPressureStep(MACGrid *vel, LevelSet *solidPhi, RigidBody *solid,
                                  float fluidDensity, float solidDensity, float dt) {
    MACGrid mixedVel = cloneMACGrid(vel);
    Grid3f mixedDensityU = makeGridLike(&vel->u);

    Grid3f mixedDensityW = makeGridLike(&vel->w);

        float solidFrac = 1.0f - fluidFrac;

        float uFluid = faceVelocity(vel, face);

        setFaceDensity(faceDensityGrid(face), face, fluidFrac * fluidDensity + solidFrac * solidDensity);
    }

    projectVelocityVariableDensity(&mixedVel, solidPhi, &mixedDensityU, &mixedDensityV, &mixedDensityW, dt);
    recoverSolidVelocityFromMixedGrid(solid, &mixedVel, solidPhi);
    copyFluidRegionVelocity(vel, &mixedVel, solidPhi);
}

void sparseRowAdd(SparseRow *row, int col, double value) {

        if (row->e[n].col == col) {
            row->e[n].value += value;

        }
    }

    row->e[row->count++] = (SparseRowEntry){ col, value };

CSRMatrix buildCSR(SparseRow *rows, int n) {
    CSRMatrix A;
    A.n = n;
    A.rowStart = malloc((n + 1) * sizeof(int));

    int nnz = 0;
    for (int r = 0; r < n; ++r) {
        sortRowByColumn(&rows[r]);
        A.rowStart[r] = nnz;
        nnz += rows[r].count;
    }
    A.rowStart[n] = nnz;

    A.col = malloc(nnz * sizeof(int));
    A.value = malloc(nnz * sizeof(double));

    for (int r = 0; r < n; ++r) {
        int start = A.rowStart[r];
        for (int e = 0; e < rows[r].count; ++e) {
            A.col[start + e] = rows[r].e[e].col;
            A.value[start + e] = rows[r].e[e].value;
        }
    }
    return A;
}
\end{lstlisting}

void csrMultiply(Vector *y, CSRMatrix *A, Vector *x) {

        for (int e = A->rowStart[r]; e < A->rowStart[r + 1]; ++e) {

\begin{lstlisting}
void buildStrongCouplingSystem(CSRMatrix *A, Vector *rhs, MACGrid *vel, LevelSet *fluidPhi,
                               RigidBody *body, FaceFractionGrid *solidFractions,
                               float dt, float rho) {
    // Base fluid pressure matrix.
    buildGeneralPressureMatrix(A, rhs, vel, fluidPhi, dt, rho);

    // J maps grid pressures to rigid-body force/torque.
    Matrix J = buildPressureToRigidForceTorque(solidFractions, body->centerOfMass);
    Matrix Minv = rigidBodyInverseMassMatrix(body);

    // Coupled Schur complement contribution: A += dt * J^T M^{-1} J.
    CSRMatrix coupling = multiplyJTMinvJ(J, Minv, dt);
    csrAddInPlace(A, &coupling);

    // RHS also includes current rigid-body velocity contribution at the boundary.
    addRigidBoundaryVelocityTerms(rhs, vel, body, solidFractions);
}

void solveStrongCoupledFluidRigidStep(MACGrid *vel, RigidBody *body, LevelSet *fluidPhi,
                                      FaceFractionGrid *fractions, float dt, float rho) {
    CSRMatrix A;
    Vector rhs, p;
    buildStrongCouplingSystem(&A, &rhs, vel, fluidPhi, body, fractions, dt, rho);
    p = zeroVector(A.n);
    solvePCG(&p, &rhs, makeCSROperator(&A), makeICPreconditioner(&A), 1.0e-7, 1000);

    applyPressureToFluid(vel, &p, dt, rho);
    Vec6 impulse = multiplyJByPressure(fractions, &p);
    applyRigidPressureImpulse(body, impulse, dt);
}
\end{lstlisting}

bool fluidSolverInvariantsHold(MACGrid *vel, Grid3Label *label, LevelSet *solidPhi) {
    // 1. Advection velocity must already satisfy projection constraints.
    if (maxFluidDivergence(vel, label) > DIVERGENCE_TOLERANCE) return false;

    // 2. Solid faces must not have inward relative normal velocity.
    if (maxSolidPenetrationVelocity(vel, solidPhi) > PENETRATION_TOLERANCE) return false;

    // 3. Pressure matrix should be symmetric positive semidefinite/definite.
    if (!pressureMatrixSymmetryCheck()) return false;

    // 4. All quantities advected with tricubic interpolation must be clamped if physically nonnegative.
    if (minSmokeOrFuel() < -NONNEGATIVE_TOLERANCE) return false;

    // 5. Level sets used as geometry should be redistanced when operations corrupt |grad(phi)| approximately 1.
    if (maxLevelSetGradientError(solidPhi) > LEVELSET_GRADIENT_TOLERANCE) return false;

    return true;
}

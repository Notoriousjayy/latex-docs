void integrateOrientation(RigidBody *b, float dt) {
    Quat omega = {0.0f, b->angularVelocity.x, b->angularVelocity.y, b->angularVelocity.z};
    Quat dq = quatMul(omega, b->orientation);

    b->orientation.w += 0.5f * dq.w * dt;
    b->orientation.x += 0.5f * dq.x * dt;
    b->orientation.y += 0.5f * dq.y * dt;
    b->orientation.z += 0.5f * dq.z * dt;
    b->orientation = quatNormalize(b->orientation);
}

void integrateRigidBody(RigidBody *b, Vec3 gravity, float dt) {
    if (b->sleeping) return;

    updateRigidBodyDerivedData(b);

    Vec3 linearAcceleration = gravity;
    if (b->inverseMass > 0.0f) {
        linearAcceleration = add3(linearAcceleration, scale3(b->force, b->inverseMass));
    }

    Vec3 angularAcceleration = mat3Vec(b->worldInvInertia, b->torque);

    b->position = add3(b->position, scale3(b->linearVelocity, dt));
    integrateOrientation(b, dt);

    b->linearVelocity = add3(b->linearVelocity, scale3(linearAcceleration, dt));
    b->angularVelocity = add3(b->angularVelocity, scale3(angularAcceleration, dt));

    clearForces(b);
    updateRigidBodyDerivedData(b);
}

\subsection*{6. Surface-Integral Mass Properties for a Closed Triangle Polyhedron}
\addcontentsline{toc}{subsection}{6. Surface-Integral Mass Properties for a Closed Triangle Polyhedron}

This is a C-like reconstruction of the book's closed-polyhedron mass-property procedure: accumulate polynomial volume integrals over oriented triangles, then convert them to mass, center of mass, and inertia.

\begin{lstlisting}
typedef struct MassProperties {
    float mass;
    Vec3 centerOfMass;
    Mat3 inertiaAboutCenter;
} MassProperties;

typedef struct PolyIntegrals {
    float I[10]; // 1, x, y, z, x^2, y^2, z^2, xy, yz, zx
} PolyIntegrals;

static void subexpressions(float w0, float w1, float w2,
                           float *f1, float *f2, float *f3,
                           float *g0, float *g1, float *g2) {
    float t0 = w0 + w1;
    *f1 = t0 + w2;

    float t1 = w0 * w0;
    float t2 = t1 + w1 * t0;
    *f2 = t2 + w2 * (*f1);

    *f3 = w0 * t1 + w1 * t2 + w2 * (*f2);
    *g0 = (*f2) + w0 * ((*f1) + w0);
    *g1 = (*f2) + w1 * ((*f1) + w1);
    *g2 = (*f2) + w2 * ((*f1) + w2);
}

MassProperties computePolyhedronMass(Vec3 *v, int *tri, int triangleCount, float density) {
    const float C1 = 1.0f / 6.0f;
    const float C2 = 1.0f / 24.0f;
    const float C3 = 1.0f / 60.0f;
    const float C4 = 1.0f / 120.0f;

    PolyIntegrals P = {0};

    for (int t = 0; t < triangleCount; ++t) {
        Vec3 p0 = v[tri[3*t + 0]];
        Vec3 p1 = v[tri[3*t + 1]];
        Vec3 p2 = v[tri[3*t + 2]];

        Vec3 e1 = sub3(p1, p0);
        Vec3 e2 = sub3(p2, p0);
        Vec3 n  = cross3(e1, e2);       // oriented area vector, not normalized

        float f1x, f2x, f3x, g0x, g1x, g2x;
        float f1y, f2y, f3y, g0y, g1y, g2y;
        float f1z, f2z, f3z, g0z, g1z, g2z;

        subexpressions(p0.x, p1.x, p2.x, &f1x, &f2x, &f3x, &g0x, &g1x, &g2x);
        subexpressions(p0.y, p1.y, p2.y, &f1y, &f2y, &f3y, &g0y, &g1y, &g2y);
        subexpressions(p0.z, p1.z, p2.z, &f1z, &f2z, &f3z, &g0z, &g1z, &g2z);

        P.I[0] += n.x * f1x;
        P.I[1] += n.x * f2x;
        P.I[2] += n.y * f2y;
        P.I[3] += n.z * f2z;
        P.I[4] += n.x * f3x;
        P.I[5] += n.y * f3y;
        P.I[6] += n.z * f3z;
        P.I[7] += n.x * (p0.y*g0x + p1.y*g1x + p2.y*g2x);
        P.I[8] += n.y * (p0.z*g0y + p1.z*g1y + p2.z*g2y);
        P.I[9] += n.z * (p0.x*g0z + p1.x*g1z + p2.x*g2z);
    }

    P.I[0] *= C1;
    P.I[1] *= C2; P.I[2] *= C2; P.I[3] *= C2;
    P.I[4] *= C3; P.I[5] *= C3; P.I[6] *= C3;
    P.I[7] *= C4; P.I[8] *= C4; P.I[9] *= C4;

    MassProperties out;
    out.mass = density * P.I[0];
    out.centerOfMass.x = P.I[1] / P.I[0];
    out.centerOfMass.y = P.I[2] / P.I[0];
    out.centerOfMass.z = P.I[3] / P.I[0];

    Mat3 I = {0};
    I.m[0][0] = density * (P.I[5] + P.I[6]);
    I.m[1][1] = density * (P.I[4] + P.I[6]);
    I.m[2][2] = density * (P.I[4] + P.I[5]);
    I.m[0][1] = I.m[1][0] = -density * P.I[7];
    I.m[1][2] = I.m[2][1] = -density * P.I[8];
    I.m[2][0] = I.m[0][2] = -density * P.I[9];

    out.inertiaAboutCenter = shiftInertiaToCenter(I, out.mass, out.centerOfMass);
    return out;
}
\end{lstlisting}

\subsection*{7. Parallel-Axis Shift of Inertia Tensor}
\addcontentsline{toc}{subsection}{7. Parallel-Axis Shift of Inertia Tensor}

\begin{lstlisting}
Mat3 shiftInertiaToCenter(Mat3 inertiaAboutOrigin, float mass, Vec3 c) {
    float cx = c.x, cy = c.y, cz = c.z;
    Mat3 S = {0};

    S.m[0][0] = mass * (cy*cy + cz*cz);
    S.m[1][1] = mass * (cx*cx + cz*cz);
    S.m[2][2] = mass * (cx*cx + cy*cy);
    S.m[0][1] = S.m[1][0] = -mass * cx * cy;
    S.m[0][2] = S.m[2][0] = -mass * cx * cz;
    S.m[1][2] = S.m[2][1] = -mass * cy * cz;

    return mat3Sub(inertiaAboutOrigin, S);
}
\end{lstlisting}

Vec3 springForce(Particle *a, Particle *b, float restLength, float stiffness, float damping) {
    Vec3 delta = sub3(b->position, a->position);
    float length = length3(delta);
    if (length == 0.0f) return clearVec3();

    Vec3 dir = scale3(delta, 1.0f / length);
    Vec3 relVel = sub3(b->velocity, a->velocity);

    float stretch = length - restLength;
    float speedAlongSpring = dot3(relVel, dir);
    float magnitude = stiffness * stretch + damping * speedAlongSpring;

    return scale3(dir, magnitude);
}
\end{lstlisting}

void simulateMassSpringSystem(Particle *p, int particleCount,
                              Spring *s, int springCount,
                              Vec3 gravity, float dt) {
    for (int i = 0; i < particleCount; ++i) {
        p[i].force = p[i].fixed ? clearVec3() : scale3(gravity, 1.0f / p[i].inverseMass);
    }

    for (int k = 0; k < springCount; ++k) {
        Particle *a = &p[s[k].a];
        Particle *b = &p[s[k].b];
        Vec3 f = springForce(a, b, s[k].restLength, s[k].stiffness, s[k].damping);
        if (!a->fixed) a->force = add3(a->force, f);
        if (!b->fixed) b->force = sub3(b->force, f);
    }

    for (int i = 0; i < particleCount; ++i) {
        integrateParticleExplicitEuler(&p[i], dt);
    }
}

    if (bend) {
        addOffset(offsets, &count, 2,0,0); addOffset(offsets, &count, 0,2,0); addOffset(offsets, &count, 0,0,2);
    }

    for (int z = 0; z < nz; ++z)
    for (int y = 0; y < ny; ++y)
    for (int x = 0; x < nx; ++x) {
        int a = gridIndex3(x, y, z, nx, ny);

            if (!inside3(x2, y2, z2, nx, ny, nz)) continue;

            float rest = spacing * sqrtf((float)(offsets[k][0]*offsets[k][0] + offsets[k][1]*offsets[k][1] + offsets[k][2]*offsets[k][2]));

    }

\subsection*{11. Recursive B-Spline Basis Function}

\begin{lstlisting}
float bsplineBasisRecursive(int i, int degree, float u, float *knot) {
    if (degree == 0) {
        return (knot[i] <= u && u < knot[i + 1]) ? 1.0f : 0.0f;
    }

    float left = 0.0f;
    float denom0 = knot[i + degree] - knot[i];
    if (denom0 != 0.0f) {
        left = ((u - knot[i]) / denom0) * bsplineBasisRecursive(i, degree - 1, u, knot);
    }

    float right = 0.0f;
    float denom1 = knot[i + degree + 1] - knot[i + 1];
    if (denom1 != 0.0f) {
        right = ((knot[i + degree + 1] - u) / denom1) * bsplineBasisRecursive(i + 1, degree - 1, u, knot);
    }

    return left + right;
}
\end{lstlisting}

\subsection*{12. Dynamic-Programming B-Spline Basis Evaluation}
\addcontentsline{toc}{subsection}{12. Dynamic-Programming B-Spline Basis Evaluation}

\begin{lstlisting}
int bsplineBasisDP(float u, int n, int degree, float *knot, float *outBasis) {
    float basis[MAX_DEGREE + 1][MAX_KNOTS] = {0};
    int key = -1;

    for (int i = 0; i <= n + degree; ++i) {
        if (knot[i] <= u && u < knot[i + 1]) {
            basis[0][i] = 1.0f;
            key = i;
        }
    }

    for (int j = 1; j <= degree; ++j) {
        for (int i = 0; i <= n + degree - j; ++i) {
            float left = 0.0f, right = 0.0f;
            float d0 = knot[i + j] - knot[i];
            float d1 = knot[i + j + 1] - knot[i + 1];
            if (d0 != 0.0f) left = ((u - knot[i]) / d0) * basis[j - 1][i];
            if (d1 != 0.0f) right = ((knot[i + j + 1] - u) / d1) * basis[j - 1][i + 1];
            basis[j][i] = left + right;
        }
    }

    for (int i = 0; i <= n; ++i) outBasis[i] = basis[degree][i];
    return key;
}
\end{lstlisting}

void setScalarBoundary(Grid2 *x, BoundaryMode mode) {
    int n = x->width;
    int m = x->height;

    for (int i = 1; i < n - 1; ++i) {
        x->value[i][0]     = (mode == BOUNDARY_Y) ? -x->value[i][1]     : x->value[i][1];
        x->value[i][m-1]   = (mode == BOUNDARY_Y) ? -x->value[i][m-2]   : x->value[i][m-2];
    }
    for (int j = 1; j < m - 1; ++j) {
        x->value[0][j]     = (mode == BOUNDARY_X) ? -x->value[1][j]     : x->value[1][j];
        x->value[n-1][j]   = (mode == BOUNDARY_X) ? -x->value[n-2][j]   : x->value[n-2][j];
    }

    x->value[0][0]       = 0.5f * (x->value[1][0]     + x->value[0][1]);
    x->value[0][m-1]     = 0.5f * (x->value[1][m-1]   + x->value[0][m-2]);
    x->value[n-1][0]     = 0.5f * (x->value[n-2][0]   + x->value[n-1][1]);
    x->value[n-1][m-1]   = 0.5f * (x->value[n-2][m-1] + x->value[n-1][m-2]);
}

void diffuseScalar(Grid2 *x, Grid2 *x0, float diffusion, float dt, int iterations) {
    int n = x->width;
    int m = x->height;
    float a = dt * diffusion * (n - 2) * (m - 2);

    for (int iter = 0; iter < iterations; ++iter) {
        for (int j = 1; j < m - 1; ++j)
        for (int i = 1; i < n - 1; ++i) {
            x->value[i][j] = (x0->value[i][j] + a * (
                x->value[i-1][j] + x->value[i+1][j] +
                x->value[i][j-1] + x->value[i][j+1])) / (1.0f + 4.0f * a);
        }
        setScalarBoundary(x, BOUNDARY_NONE);
    }
}

void advectScalar(Grid2 *dst, Grid2 *src, Grid2 *u, Grid2 *v, float dt) {
    int n = dst->width;
    int m = dst->height;
    float dtX = dt * (n - 2);
    float dtY = dt * (m - 2);

    for (int j = 1; j < m - 1; ++j)
    for (int i = 1; i < n - 1; ++i) {
        float x = i - dtX * u->value[i][j];
        float y = j - dtY * v->value[i][j];

        x = clampf(x, 0.5f, n - 1.5f);
        y = clampf(y, 0.5f, m - 1.5f);

        int i0 = (int)floorf(x), i1 = i0 + 1;
        int j0 = (int)floorf(y), j1 = j0 + 1;
        float sx = x - i0;
        float sy = y - j0;

        dst->value[i][j] = bilerp(src->value[i0][j0], src->value[i1][j0],
                                  src->value[i0][j1], src->value[i1][j1], sx, sy);
    }

    setScalarBoundary(dst, BOUNDARY_NONE);
}

\subsection*{25. Velocity Projection for Incompressibility}

\begin{lstlisting}
void projectVelocity(Grid2 *u, Grid2 *v, Grid2 *pressure, Grid2 *divergence, int iterations) {
    int n = u->width;
    int m = u->height;

    for (int j = 1; j < m - 1; ++j)
    for (int i = 1; i < n - 1; ++i) {
        divergence->value[i][j] = -0.5f * (
            u->value[i+1][j] - u->value[i-1][j] +
            v->value[i][j+1] - v->value[i][j-1]) / (float)(n - 2);
        pressure->value[i][j] = 0.0f;
    }

    setScalarBoundary(divergence, BOUNDARY_NONE);
    setScalarBoundary(pressure, BOUNDARY_NONE);

    for (int iter = 0; iter < iterations; ++iter) {
        for (int j = 1; j < m - 1; ++j)
        for (int i = 1; i < n - 1; ++i) {
            pressure->value[i][j] = (divergence->value[i][j] +
                pressure->value[i-1][j] + pressure->value[i+1][j] +
                pressure->value[i][j-1] + pressure->value[i][j+1]) / 4.0f;
        }
        setScalarBoundary(pressure, BOUNDARY_NONE);
    }

    for (int j = 1; j < m - 1; ++j)
    for (int i = 1; i < n - 1; ++i) {
        u->value[i][j] -= 0.5f * (pressure->value[i+1][j] - pressure->value[i-1][j]) * (n - 2);
        v->value[i][j] -= 0.5f * (pressure->value[i][j+1] - pressure->value[i][j-1]) * (m - 2);
    }

    setScalarBoundary(u, BOUNDARY_X);
    setScalarBoundary(v, BOUNDARY_Y);
}

void simulateFluid2D(Fluid2D *f, float viscosity, float diffusion, float dt, int iterations) {
    addScalarSource(&f->u, &f->uSource, dt);
    addScalarSource(&f->v, &f->vSource, dt);

    swapGrid(&f->uPrev, &f->u);
    diffuseScalar(&f->u, &f->uPrev, viscosity, dt, iterations);

    swapGrid(&f->vPrev, &f->v);
    diffuseScalar(&f->v, &f->vPrev, viscosity, dt, iterations);

    projectVelocity(&f->u, &f->v, &f->pressure, &f->divergence, iterations);

    swapGrid(&f->uPrev, &f->u);
    swapGrid(&f->vPrev, &f->v);
    advectScalar(&f->u, &f->uPrev, &f->uPrev, &f->vPrev, dt);
    advectScalar(&f->v, &f->vPrev, &f->uPrev, &f->vPrev, dt);

    projectVelocity(&f->u, &f->v, &f->pressure, &f->divergence, iterations);

    addScalarSource(&f->density, &f->densitySource, dt);
    swapGrid(&f->densityPrev, &f->density);
    diffuseScalar(&f->density, &f->densityPrev, diffusion, dt, iterations);
    swapGrid(&f->densityPrev, &f->density);
    advectScalar(&f->density, &f->densityPrev, &f->u, &f->v, dt);
}

void addVorticityConfinement2D(Fluid2D *f, float epsilon, float dt) {
    Grid2 omega = allocateLike(&f->density);
    Grid2 mag = allocateLike(&f->density);

    for (int j = 1; j < f->height - 1; ++j)
    for (int i = 1; i < f->width - 1; ++i) {
        omega.value[i][j] = 0.5f * ((f->v.value[i+1][j] - f->v.value[i-1][j]) -
                                    (f->u.value[i][j+1] - f->u.value[i][j-1]));
        mag.value[i][j] = absf(omega.value[i][j]);
    }

    for (int j = 2; j < f->height - 2; ++j)
    for (int i = 2; i < f->width - 2; ++i) {
        Vec2 grad = {0.5f * (mag.value[i+1][j] - mag.value[i-1][j]),
                     0.5f * (mag.value[i][j+1] - mag.value[i][j-1])};
        float len = sqrtf(grad.x*grad.x + grad.y*grad.y);
        if (len == 0.0f) continue;
        grad.x /= len;
        grad.y /= len;

        float vort = omega.value[i][j];
        f->u.value[i][j] += dt * epsilon * grad.y * vort;
        f->v.value[i][j] -= dt * epsilon * grad.x * vort;
    }
}

        Interval ib = projectPoints(b->worldVertex, b->vertexCount, axis);

    }

    return true;

\subsection*{37. Moving Interval Contact Time Along One Axis}
\addcontentsline{toc}{subsection}{37. Moving Interval Contact Time Along One Axis}

\begin{lstlisting}
bool movingIntervalsOverlapTime(Interval a, float va,
                                Interval b, float vb,
                                float tMax,
                                float *enter, float *leave) {
    float v = vb - va;

    if (v == 0.0f) {
        if (!intervalsOverlap(a, b)) return false;
        *enter = 0.0f;
        *leave = tMax;
        return true;
    }

    float t0 = (a.min - b.max) / v;
    float t1 = (a.max - b.min) / v;
    if (t0 > t1) swapFloat(&t0, &t1);

    *enter = maxf(*enter, t0);
    *leave = minf(*leave, t1);
    return *enter <= *leave && *leave >= 0.0f && *enter <= tMax;
}

    if (length3(dir) == 0.0f) dir = (Vec3){1,0,0};

        lastClosest = closest;
        dir = closest;
        if (length3(dir) < GJK_EPSILON) return 0.0f;
    }

    return length3(lastClosest);
}

\subsection*{44. Single-Contact Normal Impulse}
\addcontentsline{toc}{subsection}{44. Single-Contact Normal Impulse}

\begin{lstlisting}
void resolveContactImpulse(Contact *c) {
    RigidBody *A = c->a;
    RigidBody *B = c->b;

    Vec3 ra = sub3(c->point, A->position);
    Vec3 rb = sub3(c->point, B->position);

    Vec3 va = add3(A->linearVelocity, cross3(A->angularVelocity, ra));
    Vec3 vb = add3(B->linearVelocity, cross3(B->angularVelocity, rb));
    Vec3 rel = sub3(vb, va);

    float vn = dot3(rel, c->normal);
    if (vn > 0.0f) return;

    Vec3 raCrossN = cross3(ra, c->normal);
    Vec3 rbCrossN = cross3(rb, c->normal);

    float denom = A->inverseMass + B->inverseMass;
    denom += dot3(cross3(mat3Vec(A->worldInvInertia, raCrossN), ra), c->normal);
    denom += dot3(cross3(mat3Vec(B->worldInvInertia, rbCrossN), rb), c->normal);

    float e = minf(A->restitution, B->restitution);
    float j = -(1.0f + e) * vn / denom;
    if (j < 0.0f) j = 0.0f;

    Vec3 impulse = scale3(c->normal, j);
    applyImpulse(A, scale3(impulse, -1.0f), ra);
    applyImpulse(B, impulse, rb);
}
\end{lstlisting}

\subsection*{45. Tangential Friction Impulse}
\addcontentsline{toc}{subsection}{45. Tangential Friction Impulse}

\begin{lstlisting}
void resolveFrictionImpulse(Contact *c, float normalImpulse) {
    RigidBody *A = c->a;
    RigidBody *B = c->b;
    Vec3 ra = sub3(c->point, A->position);
    Vec3 rb = sub3(c->point, B->position);

    Vec3 va = add3(A->linearVelocity, cross3(A->angularVelocity, ra));
    Vec3 vb = add3(B->linearVelocity, cross3(B->angularVelocity, rb));
    Vec3 rel = sub3(vb, va);

    Vec3 tangent = sub3(rel, scale3(c->normal, dot3(rel, c->normal)));
    if (length3(tangent) < 1e-6f) return;
    tangent = normalize3(tangent);

    Vec3 raCrossT = cross3(ra, tangent);
    Vec3 rbCrossT = cross3(rb, tangent);
    float denom = A->inverseMass + B->inverseMass;
    denom += dot3(cross3(mat3Vec(A->worldInvInertia, raCrossT), ra), tangent);
    denom += dot3(cross3(mat3Vec(B->worldInvInertia, rbCrossT), rb), tangent);

    float jt = -dot3(rel, tangent) / denom;
    float mu = sqrtf(A->friction * B->friction);
    jt = clampf(jt, -mu * normalImpulse, mu * normalImpulse);

    Vec3 impulse = scale3(tangent, jt);
    applyImpulse(A, scale3(impulse, -1.0f), ra);
    applyImpulse(B, impulse, rb);
}
\end{lstlisting}

            solveFrictionForContact(c);
        }
    }

typedef struct ConstraintRow {
    RigidBody *a;
    RigidBody *b;
    Vec3 jaLinear;
    Vec3 jaAngular;
    Vec3 jbLinear;
    Vec3 jbAngular;
    float bias;
    float lowerLimit;
    float upperLimit;
    float accumulatedImpulse;
} ConstraintRow;

float effectiveMass(ConstraintRow *r) {
    float k = 0.0f;
    k += r->a->inverseMass * dot3(r->jaLinear, r->jaLinear);
    k += dot3(r->jaAngular, mat3Vec(r->a->worldInvInertia, r->jaAngular));
    if (r->b) {
        k += r->b->inverseMass * dot3(r->jbLinear, r->jbLinear);
        k += dot3(r->jbAngular, mat3Vec(r->b->worldInvInertia, r->jbAngular));
    }
    return (k > 0.0f) ? 1.0f / k : 0.0f;
}

        }
    }

\addcontentsline{toc}{subsection}{51. Gaussian Elimination with Partial Pivoting}

\begin{lstlisting}
bool gaussianElimination(Matrix *A, Vector *b, Vector *x) {
    int n = A->rows;

    for (int k = 0; k < n; ++k) {
        int pivot = k;
        float best = absf(A->data[k][k]);
        for (int i = k + 1; i < n; ++i) {
            float candidate = absf(A->data[i][k]);
            if (candidate > best) { best = candidate; pivot = i; }
        }
        if (best == 0.0f) return false;

        swapRows(A, k, pivot);
        swapVectorEntries(b, k, pivot);

        for (int i = k + 1; i < n; ++i) {
            float factor = A->data[i][k] / A->data[k][k];
            A->data[i][k] = 0.0f;
            for (int j = k + 1; j < n; ++j) {
                A->data[i][j] -= factor * A->data[k][j];
            }
            b->data[i] -= factor * b->data[k];
        }
    }

    for (int i = n - 1; i >= 0; --i) {
        float sum = b->data[i];
        for (int j = i + 1; j < n; ++j) sum -= A->data[i][j] * x->data[j];
        x->data[i] = sum / A->data[i][i];
    }
    return true;
}
\end{lstlisting}

bool luDecompose(Matrix *A, Matrix *L, Matrix *U) {
    int n = A->rows;
    matrixZero(L); matrixZero(U);

    for (int i = 0; i < n; ++i) L->data[i][i] = 1.0f;

    for (int j = 0; j < n; ++j) {
        for (int i = 0; i <= j; ++i) {
            float sum = 0.0f;
            for (int k = 0; k < i; ++k) sum += L->data[i][k] * U->data[k][j];
            U->data[i][j] = A->data[i][j] - sum;
        }
        if (U->data[j][j] == 0.0f) return false;
        for (int i = j + 1; i < n; ++i) {
            float sum = 0.0f;
            for (int k = 0; k < j; ++k) sum += L->data[i][k] * U->data[k][j];
            L->data[i][j] = (A->data[i][j] - sum) / U->data[j][j];
        }
    }
    return true;
}

Vec3 barycentricTriangle(Vec3 p, Vec3 a, Vec3 b, Vec3 c) {
    Vec3 v0 = sub3(b, a);
    Vec3 v1 = sub3(c, a);
    Vec3 v2 = sub3(p, a);

    float d00 = dot3(v0, v0);
    float d01 = dot3(v0, v1);
    float d11 = dot3(v1, v1);
    float d20 = dot3(v2, v0);
    float d21 = dot3(v2, v1);
    float denom = d00 * d11 - d01 * d01;

    float v = (d11 * d20 - d01 * d21) / denom;
    float w = (d00 * d21 - d01 * d20) / denom;
    float u = 1.0f - v - w;
    return (Vec3){u, v, w};
}

Quat slerp(Quat q0, Quat q1, float t) {
    float cosTheta = quatDot(q0, q1);
    if (cosTheta < 0.0f) {
        q1 = quatNeg(q1);
        cosTheta = -cosTheta;
    }

    if (cosTheta > 0.9995f) {
        return quatNormalize(quatAdd(quatScale(q0, 1.0f - t), quatScale(q1, t)));
    }

    float theta = acosf(cosTheta);
    float sinTheta = sinf(theta);
    float a = sinf((1.0f - t) * theta) / sinTheta;
    float b = sinf(t * theta) / sinTheta;
    return quatAdd(quatScale(q0, a), quatScale(q1, b));
}

    State derivativeAtPredicted = f(predicted, t + dt);
    State corrected = adamsMoultonCorrect(hist, derivativeAtPredicted, dt);
    historyPush(hist, corrected, derivativeAtPredicted);
}

float richardsonExtrapolate(float estimate[MAX_LEVELS][MAX_LEVELS], int level, int order) {
    for (int k = 1; k <= level; ++k) {
        float factor = powf(2.0f, order * k);
        estimate[level][k] = (factor * estimate[level][k - 1] - estimate[level - 1][k - 1]) / (factor - 1.0f);
    }
    return estimate[level][level];
}
\end{lstlisting}

\subsection*{66. Modified Midpoint Method}
\addcontentsline{toc}{subsection}{66. Modified Midpoint Method}

\begin{lstlisting}
State modifiedMidpoint(State x0, float t, float H, int n, DerivativeFunction f) {
    float h = H / (float)n;
    State z0 = x0;
    State z1 = stateAdd(z0, stateScale(f(z0, t), h));

    for (int m = 1; m < n; ++m) {
        State dz = f(z1, t + m * h);
        State z2 = stateAdd(z0, stateScale(dz, 2.0f * h));
        z0 = z1;
        z1 = z2;
    }

    State endDerivative = f(z1, t + H);
    return stateScale(stateAdd(stateAdd(z0, z1), stateScale(endDerivative, h)), 0.5f);
}
\end{lstlisting}

\subsection*{67. Bulirsch-Stoer Step Skeleton}
\addcontentsline{toc}{subsection}{67. Bulirsch-Stoer Step Skeleton}

\begin{lstlisting}
State bulirschStoerStep(State x, float t, float H, DerivativeFunction f, float tolerance) {
    State table[MAX_LEVELS][MAX_LEVELS];
    int sequence[MAX_LEVELS] = {2, 4, 6, 8, 10, 12, 14, 16};

    for (int k = 0; k < MAX_LEVELS; ++k) {
        table[k][0] = modifiedMidpoint(x, t, H, sequence[k], f);

        for (int j = 1; j <= k; ++j) {
            table[k][j] = rationalOrPolynomialExtrapolate(table, k, j, sequence);
        }

        if (k > 0 && stateError(table[k][k], table[k-1][k-1]) < tolerance) {
            return table[k][k];
        }
    }
    return table[MAX_LEVELS-1][MAX_LEVELS-1];
}
\end{lstlisting}

bool solveLinearProgram2D(HalfPlane *constraint, int count, Vec2 objective, Vec2 *best) {
    bool found = false;
    float bestValue = -FLT_MAX;

    for (int i = 0; i < count; ++i)
    for (int j = i + 1; j < count; ++j) {
        Vec2 x;
        if (!intersectBoundaryLines(constraint[i], constraint[j], &x)) continue;
        if (!satisfiesAllHalfPlanes(x, constraint, count)) continue;

        float value = dot2(objective, x);
        if (!found || value > bestValue) {
            found = true;
            bestValue = value;
            *best = x;
        }
    }

    return found;
}

bool lemkeLCP(Matrix *M, Vector *q, Vector *z) {
    if (allNonnegative(q)) {
        vectorZero(z);
        return true;
    }

    Dictionary D = buildInitialLemkeDictionary(M, q);
    int leaving = rowWithMostNegativeConstant(&D);
    int entering = ARTIFICIAL_VARIABLE;
    pivotDictionary(&D, leaving, entering);

    entering = complementOf(D.basic[leaving]);

    for (int iter = 0; iter < MAX_LEMKE_ITERS; ++iter) {
        leaving = minimumRatioLeavingRow(&D, entering);
        if (leaving < 0) return false;

        int leavingVar = D.basic[leaving];
        pivotDictionary(&D, leaving, entering);

        if (leavingVar == ARTIFICIAL_VARIABLE) {
            extractLcpSolution(&D, z);
            return true;
        }

        entering = complementOf(leavingVar);
    }

    return false;
}
\end{lstlisting}

\subsection*{71. Projected Gauss-Seidel for LCP}
\addcontentsline{toc}{subsection}{71. Projected Gauss-Seidel for LCP}

\begin{lstlisting}
void projectedGaussSeidelLCP(Matrix *M, Vector *q, Vector *z, int iterations) {
    int n = q->count;
    vectorZero(z);

    for (int iter = 0; iter < iterations; ++iter) {
        for (int i = 0; i < n; ++i) {
            float sum = q->data[i];
            for (int j = 0; j < n; ++j) {
                if (j != i) sum += M->data[i][j] * z->data[j];
            }
            float candidate = -sum / M->data[i][i];
            z->data[i] = maxf(0.0f, candidate);
        }
    }
}
\end{lstlisting}

\begin{lstlisting}
bool solveConvexQP(QPProblem *qp, Vector *x) {
    ActiveSet active = initialFeasibleActiveSet(qp, x);

    for (int iter = 0; iter < MAX_QP_ITERS; ++iter) {
        KKTSystem kkt = buildKKT(qp, &active, *x);
        Vector step, lambda;
        if (!solveKKT(&kkt, &step, &lambda)) return false;

        if (norm(step) < QP_EPSILON) {
            int bad = mostNegativeMultiplier(lambda, &active);
            if (bad < 0) return true;
            activeSetRemove(&active, bad);
            continue;
        }

        float alpha = largestFeasibleStep(qp, *x, step, &active);
        *x = vectorAdd(*x, vectorScale(step, alpha));
        addNewlyActiveConstraints(qp, *x, &active);
    }

    return false;
}
\end{lstlisting}

\subsection*{Practical C/C++ Porting Notes}
\addcontentsline{toc}{subsection}{Practical C/C++ Porting Notes}

\begin{enumerate}
\item Treat every matrix, quaternion, and vector routine as a tested primitive before implementing the physics algorithms.
\item Use robust tolerances for geometric predicates; exact comparison to zero is unsafe in collision detection.
\item Keep collision broad phase, narrow phase, and response separated. This makes profiling and debugging tractable.
\item Prefer stable fixed time steps for gameplay physics. Use interpolation for rendering rather than variable-step dynamics.
\item Validate mass properties using simple primitives before using arbitrary triangle meshes.
\item Build diagnostic visualizers for AABBs, contact normals, impulses, constraint rows, and centers of mass.
\item Keep a reference implementation of RK4 or velocity Verlet for comparing custom integrators.
\item For real-time engines, use iterative solvers for large contact sets and reserve exact LCP-style solvers for small or diagnostic cases.
\end{enumerate}

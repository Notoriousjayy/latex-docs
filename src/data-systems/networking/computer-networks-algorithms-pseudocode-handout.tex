\subsection{3.2 Byte Stuffing with FLAG and
ESC}\label{byte-stuffing-with-flag-and-esc}

\begin{lstlisting}[language=C]
List<Byte> byte_stuff(List<Byte> payload, Byte FLAG, Byte ESC) {
    List<Byte> out;
    out.push(FLAG);

    for (Byte b : payload) {
        if (b == FLAG || b == ESC) {
            out.push(ESC);
        }
        out.push(b);
    }

    out.push(FLAG);
    return out;
}

List<Byte> byte_unstuff(List<Byte> frame, Byte FLAG, Byte ESC) {
    List<Byte> out;
    bool escaped = false;

    for (Byte b : frame) {
        if (b == FLAG && !escaped) continue;
        if (escaped) {
            out.push(b);
            escaped = false;
        } else if (b == ESC) {
            escaped = true;
        } else {
            out.push(b);
        }
    }

    return out;
}
\end{lstlisting}

\subsection{3.4 Hamming Single-Error
Correction}\label{hamming-single-error-correction}

\begin{lstlisting}[language=C]
int required_hamming_check_bits(int data_bits) {
    int r = 0;
    while ((1 << r) < data_bits + r + 1) {
        r++;
    }
    return r;
}

BitVector hamming_encode(BitVector data) {
    int m = data.length;
    int r = required_hamming_check_bits(m);
    BitVector code(m + r + 1);   // one-based positions

    int j = 0;
    for (int pos = 1; pos <= m + r; pos++) {
        if (is_power_of_two(pos)) {
            code[pos] = 0;
        } else {
            code[pos] = data[j++];
        }
    }

    for (int p = 1; p <= m + r; p <<= 1) {
        int parity = 0;
        for (int pos = 1; pos <= m + r; pos++) {
            if (pos & p) parity ^= code[pos];
        }
        code[p] = parity;
    }

    return code;
}

int hamming_syndrome(BitVector received) {
    int syndrome = 0;
    int n = received.length - 1;

    for (int p = 1; p <= n; p <<= 1) {
        int parity = 0;
        for (int pos = 1; pos <= n; pos++) {
            if (pos & p) parity ^= received[pos];
        }
        if (parity != 0) syndrome += p;
    }

    return syndrome;
}

BitVector hamming_correct(BitVector received) {
    int bad = hamming_syndrome(received);
    if (bad != 0) received[bad] ^= 1;
    return received;
}
\end{lstlisting}

\subsection{3.5 CRC Generation and
Verification}\label{crc-generation-and-verification}

\begin{lstlisting}[language=C]
BitVector crc_remainder(BitVector message, BitVector generator) {
    int r = generator.length - 1;
    BitVector work = message.concat(zeros(r));

    for (int i = 0; i <= message.length - 1; i++) {
        if (work[i] == 1) {
            for (int j = 0; j < generator.length; j++) {
                work[i + j] ^= generator[j];
            }
        }
    }

    return work.suffix(r);
}

BitVector crc_encode(BitVector message, BitVector generator) {
    return message.concat(crc_remainder(message, generator));
}

bool crc_check(BitVector received, BitVector generator) {
    BitVector work = received.copy();
    int last_start = received.length - generator.length;

    for (int i = 0; i <= last_start; i++) {
        if (work[i] == 1) {
            for (int j = 0; j < generator.length; j++) {
                work[i + j] ^= generator[j];
            }
        }
    }

    return work.suffix(generator.length - 1).all_zero();
}
\end{lstlisting}

\subsection{4.3 Protocol 3: Positive ACK with
Retransmission}\label{protocol-3-positive-ack-with-retransmission}

\begin{lstlisting}[language=C]
const SeqNr MAX_SEQ_PAR = 1;

void sender_par() {
    SeqNr next = 0;
    Packet buffer = from_network_layer();

    while (true) {
        Frame f;
        f.kind = DATA;
        f.seq = next;
        f.info = buffer;

        to_physical_layer(f);
        start_timer(next);

        Event e = wait_for_event();
        if (e == FRAME_ARRIVAL) {
            Frame ack = from_physical_layer();
            if (ack.ack == next) {
                stop_timer(next);
                buffer = from_network_layer();
                next = inc(next, MAX_SEQ_PAR);
            }
        }
        // CHECKSUM_ERROR and TIMEOUT both cause retransmission.
    }
}

void receiver_par() {
    SeqNr expected = 0;

    while (true) {
        Event e = wait_for_event();
        if (e == FRAME_ARRIVAL) {
            Frame r = from_physical_layer();
            if (r.seq == expected) {
                to_network_layer(r.info);
                expected = inc(expected, MAX_SEQ_PAR);
            }

            Frame ack;
            ack.kind = ACK;
            ack.ack = 1 - expected;     // last correctly received frame
            to_physical_layer(ack);
        }
    }
}
\end{lstlisting}

\subsection{4.4 Protocol 4: One-Bit Sliding
Window}\label{protocol-4-one-bit-sliding-window}

\begin{lstlisting}[language=C]
void protocol_one_bit_sliding_window(bool initiator) {
    SeqNr next_to_send = 0;
    SeqNr expected = 0;
    Packet buffer = from_network_layer();

    if (initiator) {
        Frame f = { .kind = DATA, .seq = next_to_send,
                    .ack = 1 - expected, .info = buffer };
        to_physical_layer(f);
        start_timer(f.seq);
    }

    while (true) {
        Event e = wait_for_event();

        if (e == FRAME_ARRIVAL) {
            Frame r = from_physical_layer();

            if (r.seq == expected) {
                to_network_layer(r.info);
                expected = inc(expected, 1);
            }

            if (r.ack == next_to_send) {
                stop_timer(r.ack);
                buffer = from_network_layer();
                next_to_send = inc(next_to_send, 1);
            }
        }

        // Retransmit on timeout or after receiving a frame.
        Frame s;
        s.kind = DATA;
        s.seq = next_to_send;
        s.ack = 1 - expected;
        s.info = buffer;
        to_physical_layer(s);
        start_timer(s.seq);
    }
}
\end{lstlisting}

\subsection{4.5 Circular Range Test for Sliding

\begin{lstlisting}[language=C]
bool between(SeqNr a, SeqNr b, SeqNr c) {
    return ((a <= b) && (b < c)) ||
           ((c < a) && (a <= b)) ||
           ((b < c) && (c < a));
}
\end{lstlisting}

\subsection{4.6 Protocol 5: Go-Back-N}\label{protocol-5-go-back-n}

\begin{lstlisting}[language=C]
const SeqNr MAX_SEQ_GBN = 7;

void send_gbn_data(SeqNr frame_nr,
                   SeqNr frame_expected,
                   Packet buffer[MAX_SEQ_GBN + 1]) {
    Frame s;
    s.kind = DATA;
    s.info = buffer[frame_nr];
    s.seq = frame_nr;
    s.ack = (frame_expected + MAX_SEQ_GBN) % (MAX_SEQ_GBN + 1);
    to_physical_layer(s);
    start_timer(frame_nr);
}

void protocol_go_back_n() {
    SeqNr ack_expected = 0;
    SeqNr next_to_send = 0;
    SeqNr frame_expected = 0;
    SeqNr buffered = 0;
    Packet buffer[MAX_SEQ_GBN + 1];

    enable_network_layer();

    while (true) {
        Event e = wait_for_event();

        switch (e) {
        case NETWORK_LAYER_READY:
            buffer[next_to_send] = from_network_layer();
            buffered++;
            send_gbn_data(next_to_send, frame_expected, buffer);
            next_to_send = inc(next_to_send, MAX_SEQ_GBN);
            break;

        case FRAME_ARRIVAL: {
            Frame r = from_physical_layer();

            if (r.seq == frame_expected) {
                to_network_layer(r.info);
                frame_expected = inc(frame_expected, MAX_SEQ_GBN);
            }

            while (between(ack_expected, r.ack, next_to_send)) {
                buffered--;
                stop_timer(ack_expected);
                ack_expected = inc(ack_expected, MAX_SEQ_GBN);
            }
            break;
        }

        case CHECKSUM_ERROR:
            // Receiver discards bad/out-of-order frames.
            break;

        case TIMEOUT:
            next_to_send = ack_expected;
            for (SeqNr i = 1; i <= buffered; i++) {
                send_gbn_data(next_to_send, frame_expected, buffer);
                next_to_send = inc(next_to_send, MAX_SEQ_GBN);
            }
            break;
        }

        if (buffered < MAX_SEQ_GBN) enable_network_layer();
        else disable_network_layer();
    }
}
\end{lstlisting}

struct TimerNode {

    int ticks_to_go;

TimerNode *timer_head = NULL;

\subsection{4.8 Protocol 6: Selective
Repeat}\label{protocol-6-selective-repeat}

\begin{lstlisting}[language=C]
const SeqNr MAX_SEQ_SR = 7;
const int NR_BUFS = (MAX_SEQ_SR + 1) / 2;

bool no_nak = true;
SeqNr oldest_frame = MAX_SEQ_SR + 1;

void send_sr_frame(FrameKind kind,
                   SeqNr frame_nr,
                   SeqNr frame_expected,
                   Packet out_buf[NR_BUFS]) {
    Frame s;
    s.kind = kind;
    s.seq = frame_nr;
    s.ack = (frame_expected + MAX_SEQ_SR) % (MAX_SEQ_SR + 1);

    if (kind == DATA) {
        s.info = out_buf[frame_nr % NR_BUFS];
    }

    if (kind == NAK) no_nak = false;

    to_physical_layer(s);

    if (kind == DATA) start_timer(frame_nr % NR_BUFS);
    stop_ack_timer();
}

void protocol_selective_repeat() {
    SeqNr ack_expected = 0;
    SeqNr next_to_send = 0;
    SeqNr frame_expected = 0;
    SeqNr too_far = NR_BUFS;
    SeqNr buffered = 0;

    Packet out_buf[NR_BUFS];
    Packet in_buf[NR_BUFS];
    bool arrived[NR_BUFS] = { false };

    enable_network_layer();

    while (true) {
        Event e = wait_for_event();

        switch (e) {
        case NETWORK_LAYER_READY:
            buffered++;
            out_buf[next_to_send % NR_BUFS] = from_network_layer();
            send_sr_frame(DATA, next_to_send, frame_expected, out_buf);
            next_to_send = inc(next_to_send, MAX_SEQ_SR);
            break;

        case FRAME_ARRIVAL: {
            Frame r = from_physical_layer();

            if (r.kind == DATA) {
                if (r.seq != frame_expected && no_nak) {
                    send_sr_frame(NAK, 0, frame_expected, out_buf);
                } else {
                    start_ack_timer();
                }

                if (between(frame_expected, r.seq, too_far) &&
                    !arrived[r.seq % NR_BUFS]) {
                    arrived[r.seq % NR_BUFS] = true;
                    in_buf[r.seq % NR_BUFS] = r.info;

                    while (arrived[frame_expected % NR_BUFS]) {
                        to_network_layer(in_buf[frame_expected % NR_BUFS]);
                        no_nak = true;
                        arrived[frame_expected % NR_BUFS] = false;
                        frame_expected = inc(frame_expected, MAX_SEQ_SR);
                        too_far = inc(too_far, MAX_SEQ_SR);
                        start_ack_timer();
                    }
                }
            }

            if (r.kind == NAK &&
                between(ack_expected,
                        (r.ack + 1) % (MAX_SEQ_SR + 1),
                        next_to_send)) {
                send_sr_frame(DATA,
                              (r.ack + 1) % (MAX_SEQ_SR + 1),
                              frame_expected,
                              out_buf);
            }

            while (between(ack_expected, r.ack, next_to_send)) {
                buffered--;
                stop_timer(ack_expected % NR_BUFS);
                ack_expected = inc(ack_expected, MAX_SEQ_SR);
            }
            break;
        }

        case CHECKSUM_ERROR:
            if (no_nak) send_sr_frame(NAK, 0, frame_expected, out_buf);
            break;

        case TIMEOUT:
            send_sr_frame(DATA, oldest_frame, frame_expected, out_buf);
            break;

        case ACK_TIMEOUT:
            send_sr_frame(ACK, 0, frame_expected, out_buf);
            break;
        }

\begin{lstlisting}[language=C]
const int INF = 1000000000;

List<Node> shortest_path(Graph g, Node source, Node target) {
    for (Node v : g.nodes) {
        dist[v] = INF;
        pred[v] = NIL;
        permanent[v] = false;
    }

    dist[target] = 0;
    permanent[target] = true;
    Node k = target;

    while (k != source) {
        for (Node i : g.nodes) {
            if (g.edge(k, i).exists && !permanent[i]) {
                int candidate = dist[k] + g.edge(k, i).cost;
                if (candidate < dist[i]) {
                    dist[i] = candidate;
                    pred[i] = k;
                }
            }
        }

        k = tentative_node_with_smallest_dist();
        permanent[k] = true;
    }

    List<Node> path;
    for (Node v = source; v != NIL; v = pred[v]) {
        path.push(v);
    }
    return path;
}
\end{lstlisting}

void update_rto(double measured_rtt) {
    if (first_sample) {
        srtt = measured_rtt;
        rttvar = measured_rtt / 2;
    } else {
        rttvar = (1 - beta) * rttvar + beta * abs(srtt - measured_rtt);
        srtt = (1 - alpha) * srtt + alpha * measured_rtt;
    }

    rto = srtt + 4 * rttvar;
}

    if (cwnd < ssthresh) {
        cwnd += MSS;             // slow start: roughly double each RTT
    } else {
        cwnd += MSS * MSS / cwnd; // congestion avoidance: roughly +1 MSS/RTT

}

    ssthresh = max(cwnd / 2, 2 * MSS);
    cwnd = MSS;

    if (dupacks == 3) {
        ssthresh = max(cwnd / 2, 2 * MSS);
        retransmit(a.next_expected_seq);
        cwnd = ssthresh + 3 * MSS;
    } else if (dupacks > 3) {
        cwnd += MSS;
        send_if_window_allows();
    }
}

    cwnd = ssthresh;

Block64 des_encrypt(Block64 plaintext, Key56 key) {
    Block64 x = initial_permutation(plaintext);
    Block32 L = left_half(x);
    Block32 R = right_half(x);
    RoundKey K[16] = des_key_schedule(key);

    for (int i = 1; i <= 16; i++) {
        Block32 oldL = L;
        L = R;
        R = oldL XOR des_f(R, K[i]);

\subsection{10.6 Rijndael / AES Outline}\label{rijndael-aes-outline}

\begin{lstlisting}[language=C]
const int LENGTH = 16;
const int NROWS = 4;
const int NCOLS = 4;
const int ROUNDS = 10;

type Byte = unsigned 8-bit integer;

void aes128_encrypt(Byte plaintext[LENGTH],
                    Byte ciphertext[LENGTH],
                    Byte key[LENGTH]) {
    Byte state[NROWS][NCOLS];
    Byte round_key[ROUNDS + 1][NROWS][NCOLS];

    expand_key(key, round_key);
    copy_plaintext_to_state_columnwise(state, plaintext);

    xor_round_key(state, round_key[0]);

    for (int round = 1; round <= ROUNDS; round++) {
        substitute_bytes(state);
        shift_rows_left_by_row_index(state);

        if (round < ROUNDS) {
            mix_columns_gf_2_8(state);
        }

        xor_round_key(state, round_key[round]);
    }

    copy_state_to_ciphertext(ciphertext, state);
}

\subsection{10.7 Cipher Block Chaining}\label{cipher-block-chaining}

List<Block> cbc_encrypt(List<Block> plaintext, Key k, Block iv) {
    List<Block> c;
    Block prev = iv;

    for (Block p : plaintext) {
        Block x = p XOR prev;
        Block y = encrypt_block(k, x);
        c.push(y);
        prev = y;
    }

    return c;
}

List<Block> cbc_decrypt(List<Block> ciphertext, Key k, Block iv) {
    List<Block> p;
    Block prev = iv;

    for (Block c : ciphertext) {
        p.push(decrypt_block(k, c) XOR prev);
        prev = c;
    }

    return p;
}
\end{lstlisting}

\section{12. Implementation Notes}\label{implementation-notes}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  The data-link protocol pseudocode preserves the book's event-driven
  simulator model, but normalizes names into conventional C style.
\item
  Real production protocols require error handling, memory safety,
  concurrency control, validation of adversarial inputs, and precise
  timer semantics. The book examples are pedagogical.
\item
  The socket file server is sequential and intentionally minimal; a
  production server would add path validation, bounded reads,
  concurrency, structured error handling, and least-privilege execution.
\item
  The cryptographic pseudocode is educational. Real cryptographic
  implementations should use reviewed libraries, constant-time
  operations where relevant, safe randomness, authenticated modes, and
  careful key lifecycle handling.
\item
  Routing and congestion-control algorithms are expressed as canonical
  procedures. Real routers implement optimized data structures,
  route-policy engines, hardware forwarding tables, and distributed
  convergence safeguards.
\end{enumerate}

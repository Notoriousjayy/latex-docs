\section{Reading model}\label{reading-model}

TeX82 is organized as a WEB program. The source is not arranged in
compiler order; it is arranged pedagogically. Many routines are built
from named code fragments such as
\passthrough{\lstinline!Get ready to start line breaking!},
\passthrough{\lstinline!Find optimal breakpoints!}, or
\passthrough{\lstinline!Move to next line of file!}. In the pseudocode
below:

\begin{itemize}
\tightlist
\item
  Global TeX variables remain global, matching TeX82's style.
\item
  Node-memory fields such as \passthrough{\lstinline!link(p)!},
  \passthrough{\lstinline!type(p)!},
  \passthrough{\lstinline!subtype(p)!},
  \passthrough{\lstinline!width(p)!},
  \passthrough{\lstinline!glue\_ptr(p)!}, etc. are treated as accessors
  over TeX's packed memory arrays.
\item
  Routines named \passthrough{\lstinline!scan\_*!} read tokens and leave
  results in \passthrough{\lstinline!cur\_val!},
  \passthrough{\lstinline!cur\_val\_level!},
  \passthrough{\lstinline!cur\_order!}, or other global scanner outputs.
\item
  Routines named \passthrough{\lstinline!new\_*!} allocate TeX nodes and
  initialize their fields.
\item
  The pseudocode uses structured control flow where the original uses
  labels and \passthrough{\lstinline!goto!}.
\item
  Error recovery is summarized unless the recovery branch materially
  affects TeX state.
\end{itemize}

\subsection{3.3 Input line acquisition}\label{input-line-acquisition}

\begin{lstlisting}[language=C]
bool input_ln(AlphaFile *f, bool bypass_eoln) {
    int last_nonblank;

    if (bypass_eoln && !eof(f)) get(f);

    last = first;
    if (eof(f)) return false;

    last_nonblank = first;
    while (!eoln(f)) {
        if (last >= max_buf_stack) {
            max_buf_stack = last + 1;
            if (max_buf_stack == buf_size) overflow("buffer size", buf_size);
        }
        buffer[last] = xord[current_char(f)];
        get(f);
        last++;

        if (buffer[last - 1] != ' ') last_nonblank = last;
    }

    last = last_nonblank;       // trim trailing blanks
    return true;
}
\end{lstlisting}

\subsection{3.4 Terminal startup}\label{terminal-startup}

bool init_terminal(void) {
    open_terminal_input();

    for (;;) {
        wake_up_terminal();
        prompt("**");

        if (!input_ln(&term_in, true)) {
            print("! End of file on the terminal");
            return false;
        }

        loc = first;
        while (loc < last && buffer[loc] == ' ') loc++;
        if (loc < last) return true;

        print("Please type the name of your input file.");
    }
}

\subsection{3.5 String-pool creation}\label{string-pool-creation}

\begin{lstlisting}[language=C]
StringNumber make_string(void) {
    if (str_ptr == max_strings) overflow("number of strings", max_strings - init_str_ptr);
    str_ptr++;
    str_start[str_ptr] = pool_ptr;
    return str_ptr - 1;
}

void append_char(int c) {
    str_pool[pool_ptr++] = pack_ascii(c);
}

void flush_string(void) {
    str_ptr--;
    pool_ptr = str_start[str_ptr];
}

bool str_eq_buf(StringNumber s, int k) {
    for (int j = str_start[s]; j < str_start[s + 1]; j++, k++) {
        if (unpack_ascii(str_pool[j]) != buffer[k]) return false;
    }
    return true;
}

bool str_eq_str(StringNumber s, StringNumber t) {
    if (length(s) != length(t)) return false;
    for (int j = str_start[s], k = str_start[t]; j < str_start[s + 1]; j++, k++) {
        if (str_pool[j] != str_pool[k]) return false;
    }
    return true;
}
\end{lstlisting}

\subsection{3.7 Scaled arithmetic}\label{scaled-arithmetic}

\begin{lstlisting}[language=C]
Scaled round_decimals(int digits[], int n) {
    int a = 0;
    while (n > 0) {
        n--;
        a = (a + digits[n] * TWO) / 10;  // TWO is TeX's fixed-point base step
    }
    return (a + 1) / 2;
}

int mult_and_add(int n, int x, int y, int max_answer) {
    if (n < 0) { x = -x; n = -n; }
    if (n == 0) return y;
    if (abs(x) <= (max_answer - abs(y)) / n) return n * x + y;

    arith_error = true;
    return 0;
}

int x_over_n(int x, int n, int *remainder) {
    bool negative = false;
    if (n == 0) { arith_error = true; *remainder = x; return 0; }
    if (n < 0) { x = -x; n = -n; }
    if (x < 0) { negative = true; x = -x; }

    int q = x / n;
    *remainder = x % n;
    if (negative) { q = -q; *remainder = -*remainder; }
    return q;
}

int xn_over_d(int x, int n, int d, int *remainder) {
    // Computes floor(x*n/d) with overflow checks and remainder propagation.
    // TeX splits x into high and low parts to avoid machine overflow.
    if (d <= 0) { arith_error = true; return 0; }
    bool negative = (x < 0);
    if (negative) x = -x;

    int high = x / 32768;
    int low  = x % 32768;
    int t = (low * n);
    int u = (high * n) + (t / 32768);
    int v = ((u % d) * 32768) + (t % 32768);
    int q = (u / d) * 32768 + (v / d);
    *remainder = v % d;

    if (negative) { q = -q; *remainder = -*remainder; }
    return q;
}

int badness(Scaled t, Scaled s) {
    if (t == 0) return 0;
    if (s <= 0) return INF_BAD;
    double r = (double)t / (double)s;
    int b = (int)(100.0 * r * r * r + 0.5);
    return min(b, INF_BAD);
}
\end{lstlisting}

Pointer get_avail(void) {
    Pointer p;
    if (avail != NULL) {
        p = avail;
        avail = link(avail);
    } else {
        if (mem_end < mem_max) p = ++mem_end;
        else if (hi_mem_min > lo_mem_max + 1) p = --hi_mem_min;
        else overflow("main memory size", mem_max + 1 - mem_min);
    }
    link(p) = NULL;
    return p;
}

void free_avail(Pointer p) {
    link(p) = avail;
    avail = p;
}

\subsection{3.9 Variable-size memory
allocation}\label{variable-size-memory-allocation}

\begin{lstlisting}[language=C]
Pointer get_node(int size) {
    Pointer p, q, r;

    restart:
    p = rover;
    do {
        q = p + node_size(p);        // address just after this free block

        while (is_empty_free_block(q)) {
            // coalesce adjacent free blocks
            node_size(p) += node_size(q);
            unlink_free_block(q);
            q = p + node_size(p);
        }

        if (node_size(p) >= size) {
            if (node_size(p) == size) {
                unlink_free_block(p);
                rover = next_free_after(p);
                return p;
            }
            node_size(p) -= size;
            return p + node_size(p); // allocate from high end of block
        }
        p = rlink(p);
    } while (p != rover);

    if (lo_mem_max + 2 < hi_mem_min) {
        grow_low_memory_area();
        goto restart;
    }

    overflow("main memory size", mem_max + 1 - mem_min);
}

void free_node(Pointer p, int size) {
    node_size(p) = size;
    link_into_free_list(p, rover);
}
\end{lstlisting}

\subsection{3.10 Recursive node-list
destruction}\label{recursive-node-list-destruction}

\begin{lstlisting}[language=C]
void flush_node_list(Pointer p) {
    while (p != NULL) {
        Pointer q = link(p);

        if (is_char_node(p)) {
            free_avail(p);
        } else {
            switch (type(p)) {
                case HLIST_NODE:
                case VLIST_NODE:
                case UNSET_NODE:
                    flush_node_list(list_ptr(p));
                    free_node(p, box_node_size(type(p)));
                    break;
                case RULE_NODE:
                    free_node(p, rule_node_size);
                    break;
                case INS_NODE:
                    flush_node_list(ins_ptr(p));
                    delete_glue_ref(split_top_ptr(p));
                    free_node(p, ins_node_size);
                    break;
                case WHATSIT_NODE:
                    free_whatsit_substructure(p);
                    break;
                case GLUE_NODE:
                    delete_glue_ref(glue_ptr(p));
                    flush_node_list(leader_ptr(p));
                    free_node(p, small_node_size);
                    break;
                case DISC_NODE:
                    flush_node_list(pre_break(p));
                    flush_node_list(post_break(p));
                    free_node(p, small_node_size);
                    break;
                case MATH_NODE:
                case KERN_NODE:
                case PENALTY_NODE:
                    free_node(p, small_node_size);
                    break;
                default:
                    free_node(p, node_size_for_type(type(p)));
            }
        }
        p = q;
    }
}
\end{lstlisting}

Pointer id_lookup(int j, int l) {

    int h = hash_first_character(buffer[j]);
    for (int k = j + 1; k < j + l; k++) h = update_hash(h, buffer[k]);

    equiv(q) = NULL;
    eq_level(q) = LEVEL_ZERO;

\subsection{\texorpdfstring{3.15 Token acquisition:
\texttt{get\_next}}{3.15 Token acquisition: get\_next}}\label{token-acquisition-get_next}

\begin{lstlisting}[language=C]
void get_next(void) {
    cur_cs = 0;

    for (;;) {
        if (state == TOKEN_LIST) {
            if (loc == NULL) {
                end_token_list();
                continue;
            }

            Token t = info(loc);
            loc = link(loc);

            if (is_parameter_token_needing_expansion(t)) {
                expand_parameter_token(t);
                continue;
            }

            decode_token(t, &cur_cmd, &cur_chr, &cur_cs);
            break;
        }

        if (loc > limit) {
            state = NEW_LINE;
            if (!move_to_next_input_line()) continue;
        }

        int c = buffer[loc++];
        int cat = cat_code(c);

        reswitch:
        switch (state) {
            case NEW_LINE:
                if (cat == SPACER) continue;
                if (cat == CAR_RET) { cur_cmd = PAR_END; cur_chr = 0; return; }
                state = MID_LINE;
                break;

            case SKIP_BLANKS:
                if (cat == SPACER || cat == CAR_RET) continue;
                state = MID_LINE;
                break;

            case MID_LINE:
                if (cat == SPACER) state = SKIP_BLANKS;
                if (cat == CAR_RET) { loc = limit + 1; cur_cmd = SPACER; cur_chr = ' '; return; }
                break;
        }

        if (cat == ESCAPE) {
            scan_control_sequence_name(&cur_cs);
            lookup_control_sequence(cur_cs, &cur_cmd, &cur_chr);
            state = command_state_after_control_sequence(cur_cmd);
            break;
        }

        if (cat == ACTIVE_CHAR) {
            cur_cs = active_cs(c);
            lookup_control_sequence(cur_cs, &cur_cmd, &cur_chr);
            break;
        }

        if (cat == SUP_MARK && next_chars_form_expanded_code()) {
            c = expanded_character_code();
            cat = cat_code(c);
            goto reswitch;
        }

        cur_cmd = cat;
        cur_chr = c;
        break;
    }

    if (alignment_entry_just_ended()) insert_v_template_and_restart();
}

    Pointer r = link(ref_count);      // parameter text, then body

    int acc = 0;

void scan_dimen(bool mu, bool inf, bool shortcut) {

    int fraction = 0;

    arith_error = false;

        get_next_nonblank_nonsign_token(&negative);

\subsection{3.42 Paragraph line breaking}\label{paragraph-line-breaking}

\begin{lstlisting}[language=C]
void line_break(int final_widow_penalty) {
    pack_begin_line = mode_line;

    prepare_paragraph_for_line_breaking();   // add infinite penalty and parfillskip

    int thresholds[3];
    int pass_count = 0;
    if (pretolerance >= 0) thresholds[pass_count++] = pretolerance;
    thresholds[pass_count++] = tolerance;
    if (emergency_stretch > 0) thresholds[pass_count++] = tolerance;

    bool second_pass = (pretolerance < 0);
    bool final_pass = false;

    for (int pass = 0; pass < pass_count; pass++) {
        threshold = min(thresholds[pass], INF_BAD);
        second_pass = (pass > 0 || pretolerance < 0);
        final_pass = (pass == pass_count - 1);

        if (final_pass && emergency_stretch > 0) background[2] += emergency_stretch;
        if (second_pass) initialize_hyphenation_for_paragraph();

        create_initial_active_breakpoint();
        cur_p = link(temp_head);
        prev_p = cur_p;
        auto_breaking = true;

        while (cur_p != NULL && active_list_not_empty()) {
            if (is_char_node(cur_p)) {
                advance_over_character_run_and_update_widths();
            } else {
                switch (type(cur_p)) {
                    case HLIST_NODE:
                    case VLIST_NODE:
                    case RULE_NODE:
                        act_width += width(cur_p);
                        break;
                    case GLUE_NODE:
                        if (legal_glue_break(prev_p, auto_breaking)) try_break(0, UNHYPHENATED);
                        add_glue_widths_to_active_width(cur_p);
                        if (second_pass && auto_breaking) try_to_hyphenate_following_word();
                        break;
                    case KERN_NODE:
                        if (subtype(cur_p) == EXPLICIT) try_kern_break_then_add_width();
                        else act_width += width(cur_p);
                        break;
                    case DISC_NODE:
                        try_discretionary_break();
                        break;
                    case MATH_NODE:
                        auto_breaking = (subtype(cur_p) == AFTER);
                        try_kern_break_then_add_width();
                        break;
                    case PENALTY_NODE:
                        try_break(penalty(cur_p), UNHYPHENATED);
                        break;
                    default:
                        break;
                }
            }
            prev_p = cur_p;
            cur_p = link(cur_p);
        }

        if (cur_p == NULL && final_break_succeeds()) break;
        cleanup_break_nodes();
    }

    post_line_break(final_widow_penalty);
    cleanup_break_nodes();
    pack_begin_line = 0;
}

\subsection{\texorpdfstring{3.43 Breakpoint evaluation:
\texttt{try\_break}}{3.43 Breakpoint evaluation: try\_break}}\label{breakpoint-evaluation-try_break}

\begin{lstlisting}[language=C]
void try_break(int penalty, HyphenationFlag hyph) {
    if (abs(penalty) >= INF_PENALTY) {
        if (penalty > 0) return;
        penalty = EJECT_PENALTY;
    }

    reset_best_place_by_fitness_class();

    Pointer prev_r = active;
    Pointer r = link(active);
    copy_active_width_to_cur_active_width();

    while (r != last_active) {
        if (type(r) == DELTA_NODE) {
            update_cur_active_width_from_delta(r);
            prev_r = r;
            r = link(r);
            continue;
        }

        int line = line_number(r);
        Scaled line_width = determine_line_width(line);
        Scaled shortfall = line_width - cur_active_width[NATURAL_WIDTH];

        int b;
        if (shortfall > 0) b = badness(shortfall, available_stretch(cur_active_width));
        else b = badness(-shortfall, available_shrink(cur_active_width));

        if (b > INF_BAD || penalty == EJECT_PENALTY) {
            deactivate_if_impossible(prev_r, r);
        }

        if (b <= threshold) {
            FitnessClass fit = classify_fitness(shortfall, b);
            int d = compute_demerits(b, penalty, hyph, fit, fitness(r));
            if (d < minimal_demerits[fit]) {
                minimal_demerits[fit] = d;
                best_place[fit] = break_node(r);
                best_place_line[fit] = line;
            }
        }

        prev_r = r;
        r = link(r);
    }

    for_each_fitness_class_with_feasible_break() {
        Pointer passive_node = create_passive_break_node(cur_p, best_place[fit]);
        Pointer active_node = create_active_break_node(passive_node, fit, minimal_demerits[fit]);
        insert_active_node_with_delta_widths(active_node);
    }
}

\subsection{3.48 Vertical page-break cost
computation}\label{vertical-page-break-cost-computation}

\begin{lstlisting}[language=C]
int vert_break(Pointer p, Scaled target_height, Scaled max_depth) {
    Scaled page_so_far[7] = {0};
    Pointer best_place = NULL;
    int least_cost = AWFUL_BAD;
    Pointer prev_p = NULL;

    for (; p != NULL; p = link(p)) {
        int pi = INF_PENALTY;

        switch (type(p)) {
            case HLIST_NODE:
            case VLIST_NODE:
            case RULE_NODE:
                page_so_far[HEIGHT] += page_so_far[DEPTH] + height(p);
                page_so_far[DEPTH] = depth(p);
                break;
            case GLUE_NODE:
                if (precedes_breakable_box(prev_p)) pi = 0;
                add_glue_to_page_so_far(p, page_so_far);
                break;
            case KERN_NODE:
                if (link(p) != NULL && type(link(p)) == GLUE_NODE) pi = 0;
                page_so_far[HEIGHT] += page_so_far[DEPTH] + width(p);
                page_so_far[DEPTH] = 0;
                break;
            case PENALTY_NODE:
                pi = penalty(p);
                break;
            case INS_NODE:
                account_for_insert_in_vertical_break(p);
                break;
        }

        if (page_so_far[DEPTH] > max_depth) {
            page_so_far[HEIGHT] += page_so_far[DEPTH] - max_depth;
            page_so_far[DEPTH] = max_depth;
        }

        if (pi < INF_PENALTY) {
            int b = compute_page_badness(page_so_far, target_height);
            int c = page_break_cost(b, pi);
            if (c <= least_cost) {
                best_place = p;
                least_cost = c;
                best_height_plus_depth = page_so_far[HEIGHT] + page_so_far[DEPTH];
            }
            if (c == AWFUL_BAD || pi <= EJECT_PENALTY) break;
        }

        prev_p = p;
    }

    return best_place;
}

    get_x_token();
    if (cur_cmd == LETTER || cur_cmd == OTHER_CHAR || cur_cmd == CHAR_GIVEN) {
        set_math_char(math_code(cur_chr), p);
    } else if (cur_cmd == MATH_CHAR_NUM) {
        scan_fifteen_bit_int();
        set_math_char(cur_val, p);
    } else {
        back_input_current_token();
        scan_left_brace();
        push_math(MATH_GROUP);
        cur_mlist = p;
    }
}

void sub_sup(void) {
    Pointer p = tail;
    if (!scriptable_noad(p)) {
        p = new_noad();
        tail_append(p);
    }
    Pointer field = (cur_cmd == SUP_MARK) ? supscr(p) : subscr(p);
    if (!field_is_empty(field)) report_double_script_error();
    scan_math(field);
}

void after_math(void) {
    finish_current_math_list();
    if (display_math()) build_display_formula_and_append_to_vlist();
    else convert_inline_math_to_hlist_and_append();
}
\end{lstlisting}

void prefixed_command(void) {
    PrefixFlags prefixes = scan_optional_prefixes();  // global, long, outer
    get_x_token();

    switch (cur_cmd) {
        case DEF:
            define_macro(prefixes);
            break;
        case LET:
            define_equivalent_from_next_token(prefixes);
            break;
        case SHORTHAND_DEF:
            define_shorthand_register(prefixes);
            break;
        case REGISTER:
            do_register_command(prefixes);
            break;
        case SET_BOX:
            scan_box_assignment(prefixes);
            break;
        case DEF_FONT:
            new_font(prefixes);
            break;
        case SET_INTERACTION:
            new_interaction();
            break;
        case ASSIGN_INT:
        case ASSIGN_DIMEN:
        case ASSIGN_GLUE:
        case ASSIGN_MU_GLUE:
        case ASSIGN_TOKS:
            scan_and_store_assignment(prefixes, cur_cmd, cur_chr);
            break;
        default:
            report_missing_assignment_target();
    }

    after_assignment_token_if_any();
}

\subsection{3.58 Register arithmetic}\label{register-arithmetic}

\begin{lstlisting}[language=C]
void do_register_command(Command a) {
    ValueLevel level = decode_register_level(a);
    Pointer target = scan_register_location(level);
    Operator op = cur_cmd;  // advance, multiply, divide, assign

    if (op == ASSIGN) {
        scan_value_of_level(level);
        store_value(target, cur_val, level);
    } else if (op == ADVANCE) {
        scan_optional_by();
        scan_value_of_level(level);
        add_value_to_register(target, cur_val, level);
    } else if (op == MULTIPLY) {
        scan_optional_by();
        scan_int();
        multiply_register(target, cur_val, level);
    } else if (op == DIVIDE) {
        scan_optional_by();
        scan_int();
        divide_register(target, cur_val, level);
    }
}

void do_extension(void) {
    switch (cur_chr) {
        case OPEN_NODE:    new_write_whatsit(OPEN_NODE); scan_openout_operands(); break;
        case WRITE_NODE:   new_write_whatsit(WRITE_NODE); scan_write_tokens(); break;
        case CLOSE_NODE:   new_write_whatsit(CLOSE_NODE); scan_stream_number(); break;
        case SPECIAL_NODE: new_write_whatsit(SPECIAL_NODE); scan_special_tokens(); break;
        case IMMEDIATE_CODE:
            get_x_token();
            if (cur_cmd == EXTENSION) execute_extension_immediately();
            else back_input_current_token();
            break;
        case SET_LANGUAGE_CODE:
            fix_language();
            break;
    }
}

void out_what(Pointer p) {
    switch (subtype(p)) {
        case OPEN_NODE:    open_write_stream(p); break;
        case WRITE_NODE:   write_out(p); break;
        case CLOSE_NODE:   close_write_stream(p); break;
        case SPECIAL_NODE: special_out(p); break;
        case LANGUAGE_NODE: /* no DVI output */ break;

}

    expand_write_tokens_in_safe_scanner_context(p);
    write_result_to_selected_stream_or_terminal();
}

}

    else cur_lang = min(cur_val, 255);
    append_language_node_if_hmode_language_changed();

\section{Most important algorithmic
invariants}\label{most-important-algorithmic-invariants}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  \textbf{Everything is a token before it is executed.} Input lines
  become tokens; expandable tokens reduce to other tokens; only
  unexpandable commands reach the executive dispatcher.
\item
  \textbf{All local state is stack-disciplined.} Groups save old
  equivalents and restore them through \passthrough{\lstinline!unsave!};
  semantic modes are tracked through the nest.
\item
  \textbf{Node lists are the universal intermediate representation.}
  Paragraphs, pages, math formulas, insertions, and alignments are all
  represented as linked nodes in TeX's memory array.
\item
  \textbf{Line breaking is dynamic programming over legal breakpoints.}
  Active nodes represent feasible prior breaks; passive nodes remember
  chosen break chains.
\item
  \textbf{Page breaking is greedy with cost memory.} The page builder
  tracks the best page break seen so far until a forced or sufficiently
  bad breakpoint triggers output.
\item
  \textbf{Math typesetting is a tree/list conversion.} Noads encode
  semantic math atoms; \passthrough{\lstinline!mlist\_to\_hlist!} turns
  them into boxes, kerns, rules, delimiters, and glue.
\item
  \textbf{Hyphenation is trie-driven.} Patterns are compiled into a
  trie, then applied to collected word characters to mark discretionary
  positions.
\item
  \textbf{DVI output is a traversal with stateful movement compression.}
  Horizontal and vertical movements are compressed by remembering
  previous movements at compatible stack levels.
\end{enumerate}

\section{Recommended next
decomposition}\label{recommended-next-decomposition}

For implementation study, split the algorithms into these independent
review packets:

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  Scanner/expander: \passthrough{\lstinline!get\_next!},
  \passthrough{\lstinline!expand!},
  \passthrough{\lstinline!macro\_call!}, conditionals.
\item
  Value scanners: integers, dimensions, glue, rules, filenames, fonts.
\item
  Memory and node model: allocation, copying, flushing, semantic nest.
\item
  Paragraph engine: \passthrough{\lstinline!line\_break!},
  \passthrough{\lstinline!try\_break!}, hyphenation, post-line
  reconstruction.
\item
  Page engine: \passthrough{\lstinline!vert\_break!},
  \passthrough{\lstinline!build\_page!}, insertions, output routine
  handling.
\item
  Math engine: noad construction,
  \passthrough{\lstinline!mlist\_to\_hlist!}, fractions, radicals,
  delimiters.
\item
  DVI backend: packing, shipping, movement compression, whatsit output.
\item
  Format lifecycle: INITEX initialization, dump, undump, production
  startup.
\end{enumerate}

\subsection{2.1 Graph, Vertex, Arc, and Utility Fields}\label{graph-vertex-arc-and-utility-fields}

\begin{lstlisting}[language=C]
type Util = union {
    Vertex* V;
    Arc*    A;
    Graph*  G;
    char*   S;
    long    I;
};

type Arc = struct {
    Vertex* tip;       // destination vertex
    Arc*    next;      // next outgoing arc from same source
    long    len;       // arc length or weight
    Util    a, b;      // algorithm-specific utility fields
};

type Vertex = struct {
    Arc*  arcs;        // outgoing adjacency list
    char* name;        // symbolic name
    Util  u, v, w, x, y, z;
};

type Graph = struct {
    Vertex* vertices;
    long    n;         // number of vertices
    long    m;         // number of arcs
    char    id[];
    Area    data;
    Area    aux_data;
    Util    uu, vv, ww, xx, yy, zz;
    char    util_types[15];
};
\end{lstlisting}

\textbf{Design point:} Every algorithm reuses the utility fields with local macros such as \passthrough{\lstinline!rank!}, \passthrough{\lstinline!parent!}, \passthrough{\lstinline!dist!}, \passthrough{\lstinline!backlink!}, \passthrough{\lstinline!marked!}, \passthrough{\lstinline!master!}, or \passthrough{\lstinline!size!}. This avoids allocating per-algorithm wrapper structures.

\subsubsection{Algorithm: Allocate a Block in an Area}\label{algorithm-allocate-a-block-in-an-area}

\begin{lstlisting}[language=C]
char* gb_alloc(long n, Area s) {
    if (n <= 0 || n > MAX_ALLOC) {
        gb_trouble_code |= ILLEGAL_REQUEST;
        return NULL;
    }

    n = round_up_to_pointer_multiple(n);
    long block_size = round_up_to_256(n + 2 * sizeof(char*));
    char* loc = calloc(block_size, 1);

    if (loc == NULL) {
        gb_trouble_code |= ALLOCATION_FAILED;
        return NULL;
    }

    AreaNode* header = (AreaNode*)(loc + n);
    header->first = loc;
    header->next  = s.head;
    s.head = header;

    return loc;
}
\end{lstlisting}

\subsubsection{Algorithm: Free an Area}\label{algorithm-free-an-area}

\begin{lstlisting}[language=C]
void gb_free(Area s) {
    while (s.head != NULL) {
        AreaNode* next = s.head->next;
        free(s.head->first);
        s.head = next;
    }
}
\end{lstlisting}

\subsubsection{Algorithm: Fill the Input Buffer and Strip Trailing Blanks}\label{algorithm-fill-the-input-buffer-and-strip-trailing-blanks}

\begin{lstlisting}[language=C]
void fill_buf(void) {
    if (fgets(buffer, sizeof(buffer), cur_file) == NULL) {
        io_errors |= FILE_ENDED_PREMATURELY;
        buffer[0] = '\0';
        more_data = false;
        return;
    }

    char* p = first_null(buffer) - 1;

    if (p < buffer || *p != '\n') {
        io_errors |= MISSING_NEWLINE;
        p++;
    }

    while (p >= buffer && *p == ' ') {
        p--;
    }

    *++p = '\n';
    *++p = '\0';
    cur_pos = buffer;
}
\end{lstlisting}

\subsubsection{Algorithm: Compute a Data-File Checksum}\label{algorithm-compute-a-data-file-checksum}

\begin{lstlisting}[language=C]
long new_checksum(char* s, long old_checksum) {
    long a = old_checksum;

    for (char* p = s; *p != '\0'; p++) {
        a = (2 * a + imap_ord(*p)) % CHECKSUM_PRIME;
    }

    return a;
}
\end{lstlisting}

\subsection{\texorpdfstring{3.3 Random Number Generation (\texttt{gb\_flip.w})}{3.3 Random Number Generation (gb\_flip.w)}}\label{random-number-generation-gb_flip.w}

\subsubsection{Algorithm: Subtractive Random Cycle}\label{algorithm-subtractive-random-cycle}

\begin{lstlisting}[language=C]
long gb_flip_cycle(void) {
    for (i = 1, j = 32; j <= 55; i++, j++) {
        A[i] = (A[i] - A[j]) mod 2^31;
    }

    for (j = 1; i <= 55; i++, j++) {
        A[i] = (A[i] - A[j]) mod 2^31;
    }

    gb_fptr = &A[54];
    return A[55];
}
\end{lstlisting}

\subsubsection{Algorithm: Initialize Random State}\label{algorithm-initialize-random-state}

\begin{lstlisting}[language=C]
void gb_init_rand(long seed) {
    seed = seed mod 2^31;

    long prev = seed;
    long next = 1;
    A[55] = prev;

    for (i = 21; i != 0; i = (i + 21) % 55) {
        A[i] = next;
        next = (prev - next) mod 2^31;
        seed = cyclic_shift_right_31_bit(seed, 1);
        next = (next - seed) mod 2^31;
        prev = A[i];
    }

    // Warm up transient state.
    for (k = 0; k < 5; k++) gb_flip_cycle();
}
\end{lstlisting}

\subsubsection{\texorpdfstring{Algorithm: Unbiased Uniform Integer in \texttt{{[}0,\ m)}}{Algorithm: Unbiased Uniform Integer in {[}0, m)}}\label{algorithm-unbiased-uniform-integer-in-0-m}

\begin{lstlisting}[language=C]
long gb_unif_rand(long m) {
    unsigned long t = 2^31 - (2^31 % m);
    long r;

    do {
        r = gb_next_rand();
    } while ((unsigned long)r >= t);

    return r % m;
}
\end{lstlisting}

\subsection{\texorpdfstring{3.4 Linked-List Radix Sort (\texttt{gb\_sort.w})}{3.4 Linked-List Radix Sort (gb\_sort.w)}}\label{linked-list-radix-sort-gb_sort.w}

\subsubsection{Algorithm: Randomized Stable-Enough Radix Distribution}\label{algorithm-randomized-stable-enough-radix-distribution}

\begin{lstlisting}[language=C]
void gb_linksort(Node* list) {
    clear_lists(alt_sorted, 256);

    // Two random passes: equal-key items are pseudo-randomized.
    for (p = list; p != NULL; p = q) {
        q = p->link;
        k = high8(gb_next_rand());
        push(&alt_sorted[k], p);
    }

    clear_lists(gb_sorted, 256);
    redistribute_random(alt_sorted, gb_sorted);

    // Four byte passes over the nonnegative key.
    redistribute_by_byte(gb_sorted, alt_sorted, 0);   // low byte
    redistribute_by_byte(alt_sorted, gb_sorted, 8);   // next byte
    redistribute_by_byte(gb_sorted, alt_sorted, 16);  // next byte
    redistribute_by_byte(alt_sorted, gb_sorted, 24);  // high byte
}
\end{lstlisting}

\subsubsection{Algorithm: Redistribute by Key Byte}\label{algorithm-redistribute-by-key-byte}

\begin{lstlisting}[language=C]
void redistribute_by_byte(Node* source[256], Node* dest[256], int shift) {
    clear_lists(dest, 256);

    for (bucket in traversal_order_for_this_pass) {
        for (p = source[bucket]; p != NULL; p = q) {
            q = p->link;
            k = (p->key >> shift) & 0xff;
            push(&dest[k], p);
        }
    }
}
\end{lstlisting}

Graph* restore_graph(char* filename) {
    open_raw_data_file(filename);
    if (io_errors) fail(EARLY_DATA_FAULT);

    parse_header_line(&util_types, &vertex_count, &arc_count);

    Graph* g = gb_new_graph(0);
    replace_default_storage(g);
    g->vertices = allocate_vertex_block(vertex_count, g->data);
    Arc* arcs   = allocate_arc_block(arc_count, g->data);

    parse_graph_record_fields(g, util_types);
    expect_section("* Vertices");
    for (i = 0; i < vertex_count; i++) {
        parse_vertex_record(&g->vertices[i], util_types);
    }

    expect_section("* Arcs");
    for (i = 0; i < arc_count; i++) {
        parse_arc_record(&arcs[i], util_types);
    }

    verify_checksum();
    close_raw_data_file();
    return g;

fail:
    close_raw_data_file();
    gb_recycle(g);
    return NULL;
}

bool fill_field(Util* field, char type_code) {

long save_graph(Graph* g, char* filename) {

\subsection{\texorpdfstring{3.6 Dijkstra Shortest Paths (\texttt{gb\_dijk.w})}{3.6 Dijkstra Shortest Paths (gb\_dijk.w)}}\label{dijkstra-shortest-paths-gb_dijk.w}

\subsubsection{Algorithm: Dijkstra with Optional Potential Function}\label{algorithm-dijkstra-with-optional-potential-function}

\begin{lstlisting}[language=C]
long dijkstra(Vertex* source, Vertex* target, Graph* g, long (*h)(Vertex*)) {
    if (h == NULL) h = zero_heuristic;

    for (v in g->vertices) {
        v->backlink = NULL;       // unseen
    }

    source->backlink = source;
    source->dist = 0;
    source->hh_val = h(source);
    init_queue(0);

    Vertex* t = source;

    while (t != target) {
        long base = t->dist - t->hh_val;

        for (Arc* a = t->arcs; a != NULL; a = a->next) {
            Vertex* v = a->tip;
            long candidate;

            if (v->backlink != NULL) {
                candidate = base + a->len + v->hh_val;
                if (candidate < v->dist) {
                    v->backlink = t;
                    requeue(v, candidate);
                }
            } else {
                v->hh_val = h(v);
                v->backlink = t;
                enqueue(v, base + a->len + v->hh_val);
            }
        }

        t = del_min();
        if (t == NULL) return -1;
    }

    return target->dist - target->hh_val + source->hh_val;
}
\end{lstlisting}

\begin{lstlisting}[language=C]
void init_dlist(long d) {
    head.left = &head;
    head.right = &head;
    head.dist = d - 1;
}

void enlist(Vertex* v, long d) {
    v->dist = d;

    // Insert so that following left links from head gives decreasing keys;
    // the minimum is therefore near the right/front deletion end.
    Vertex* p = find_position_from_tail(d);
    insert_after(p, v);
}

void reenlist(Vertex* v, long smaller_d) {
    remove_from_dlist(v);
    enlist(v, smaller_d);
}

    if (head.right == &head) return NULL;
    Vertex* v = head.right;
    remove_from_dlist(v);

\begin{lstlisting}[language=C]
void print_dijkstra_result(Vertex* target) {
    if (target->backlink == NULL) {
        print_unreachable(target);
        return;
    }

    // Reverse backlink chain from target to source.
    Vertex* reversed = NULL;
    Vertex* p = target;
    do {
        Vertex* q = p->backlink;
        p->backlink = reversed;
        reversed = p;
        p = q;
    } while (reversed != p);

    Vertex* source = p;
    for (Vertex* v = source; v != NULL; v = v->backlink) {
        print_true_distance_and_name(v, source);
    }

    // Restore original backlink chain.
    restore_backlinks(source, target);
}

\begin{lstlisting}[language=C]
long gate_eval(Graph* g, char* in_vec, char* out_vec) {
    if (g == NULL) return -2;

    Vertex* v = g->vertices;

    if (in_vec != NULL) {
        while (*in_vec != '\0' && v < g->vertices + g->n) {
            v->val = *in_vec - '0';
            v++;
            in_vec++;
        }
    }

    for (; v < g->vertices + g->n; v++) {
        switch (v->typ) {
            case 'I':
                continue;

            case 'L':
                v->val = v->alt->val;
                break;

            case '&':
                t = 1;
                for (a = v->arcs; a != NULL; a = a->next) t &= a->tip->val;
                v->val = t;
                break;

            case '|':
                t = 0;
                for (a = v->arcs; a != NULL; a = a->next) t |= a->tip->val;
                v->val = t;
                break;

            case '^':
                t = 0;
                for (a = v->arcs; a != NULL; a = a->next) t ^= a->tip->val;
                v->val = t;
                break;

            case '~':
                v->val = 1 - v->arcs->tip->val;
                break;

            default:
                return -1;
        }
    }

    if (out_vec != NULL) {
        for (Arc* a = g->outs; a != NULL; a = a->next) {
            *out_vec++ = '0' + value_of_output_tip(a->tip);
        }
        *out_vec = '\0';
    }

    return 0;
}
\end{lstlisting}

Graph* random_graph(n, m, multi, self, directed,
                    dist_from, dist_to, min_len, max_len, seed) {

    gb_init_rand(seed);

    Graph* g = gb_new_graph(n);

    AliasTable* to_table   = dist_to   ? build_walker_alias(dist_to, n)   : NULL;

    while (m > 0) {
        Vertex* u = from_table ? sample_alias(from_table) : random_vertex(g);
        Vertex* v = to_table   ? sample_alias(to_table)   : random_vertex(g);

        } else if (directed) {
            gb_new_arc(u, v, len);
        } else {
            gb_new_edge(u, v, len);
        }

        m--;
    }

    return g;
}

void merge_components_if_different(Vertex* u, Vertex* v) {
    Vertex* mu = u->master;
    Vertex* mv = v->master;

    if (mu == mv) return;

    // Change master of smaller component.
    if (mu->size < mv->size) {
        mv->size += mu->size;
        if (mu->size == 1) isol--;
        for (t = mu->link; t != mu; t = t->link) t->master = mv;
        mu->master = mv;
    } else {
        if (mu->size == 1) isol--;
        mu->size += mv->size;
        if (mv->size == 1) isol--;
        for (t = mv->link; t != mv; t = t->link) t->master = mu;
        mv->master = mu;
    }

    // Splice circular lists.
    t = mu->link;
    mu->link = mv->link;
    mv->link = t;

    comp--;

            v->dist = -1;
            v->parent = NULL;

        root->dist = 0;

            }

\subsubsection{Algorithm: Ternary Multiply/Divide Subroutine Executed by RISC}\label{algorithm-ternary-multiplydivide-subroutine-executed-by-risc}

\begin{lstlisting}[language=C]
// Computes x * floor(y / z) using shift/subtract logic.
long tri(long x, long y, long z) {
    result = 0;
    shift_count = 0;
    overflow_flag = false;

    y -= z;
    while (y >= 0) {
        z <<= 1;
        x <<= 1;
        if (overflow(x)) overflow_flag = true;
        shift_count++;
        y -= z;
    }

    while (shift_count >= 0) {
        z >>= 1;
        x >>= 1;
        y += z;

        if (y >= 0) {
            result += x;
            if (overflow(result)) overflow_flag = true;
            y -= z;
        }

        shift_count--;
    }

    if (overflow_flag) force_overflow_status();
    return result;
}
\end{lstlisting}

\section{6. Cross-Cutting Algorithmic Patterns}\label{cross-cutting-algorithmic-patterns}

\subsection{6.1 Explicit Recursion Elimination}\label{explicit-recursion-elimination}

\passthrough{\lstinline!roget\_components.w!} and \passthrough{\lstinline!book\_components.w!} both avoid relying on the C runtime stack. Their DFS algorithms store recursion state inside vertex utility fields:

\begin{lstlisting}[language=C]
while (current != NULL) {
    if (current has untagged outgoing arc) {
        inspect_or_descend();
    } else {
        mature_current_vertex();
        current = current->parent;
    }
}
\end{lstlisting}

This is a deliberate portability strategy: the source avoids deep recursive calls on systems with small or inefficient stacks.

\subsection{6.2 Utility-Field Rebinding}\label{utility-field-rebinding}

The same six \passthrough{\lstinline!Vertex!} utility fields and two \passthrough{\lstinline!Arc!} utility fields are rebound repeatedly:

\begin{lstlisting}[language=C]
#define dist      z.I
#define backlink  y.V
#define rank      z.I
#define parent    y.V
#define link      w.V
#define min       v.V
#define master    y.V
#define size      x.I
\end{lstlisting}

This pattern keeps graph records compact and lets algorithms annotate graphs without external hash maps.

\subsection{6.3 Canonical Randomness}\label{canonical-randomness}

The GraphBase random-number layer supports repeatable experiments:

\begin{lstlisting}[language=C]
gb_init_rand(seed);
experiment_graph = generator(parameters, seed);
run_algorithm(experiment_graph);
\end{lstlisting}

The same seed and parameters are intended to produce the same graph across machines.

\subsection{6.4 Source-Level Performance Instrumentation}\label{source-level-performance-instrumentation}

Several demonstrations count \passthrough{\lstinline!mems!}, a machine-independent approximation to memory references:

\begin{lstlisting}[language=C]
#define o    mems++
#define oo   mems += 2
#define ooo  mems += 3
\end{lstlisting}

The algorithm itself is manually annotated, allowing comparative measurements independent of compiler and CPU timing noise.

int julian_day(int month, int day, int year) {
    if (year == 0) error("no year zero");
    if (year < 0) year++;
    int y, m;
    if (month > 2) { y = year;     m = month + 1;  }
    else           { y = year - 1; m = month + 13; }

    int jd = floor(365.25 * y) + floor(30.6001 * m) + day + JULIAN_OFFSET;
    if (date_is_gregorian(month, day, year)) {
        int c = floor(y / 100.0);
        jd += 2 - c + floor(c / 4.0);
    }
    return jd;
}
\end{lstlisting}

Date calendar_date(int jd) {
    int a = jd;
    if (jd >= GREGORIAN_START_JD) {
        int alpha = floor((jd - GREGORIAN_CORRECTION_BASE - 0.25) / 36524.25);
        a = jd + 1 + alpha - floor(alpha / 4.0);
    }
    int b = a + 1524;
    int c = floor((b - 122.1) / 365.25);
    int d = floor(365.25 * c);
    int e = floor((b - d) / 30.6001);
    int day = b - d - floor(30.6001 * e);
    int month = e - 1;
    if (month > 12) month -= 12;
    int year = c - 4715;
    if (month > 2) year--;
    if (year <= 0) year--;
    return Date(year, month, day);
}

\subsubsection{Algorithm: Gauss-Jordan elimination with full pivoting}\label{algorithm-gauss-jordan-elimination-with-full-pivoting}

\begin{lstlisting}[language=C]
void gauss_jordan(Matrix A, Matrix B) {
    int n = A.rows;
    int m = B.cols;
    int pivot_used[n] = {0};
    int pivot_row[n], pivot_col[n];

    for (int step = 0; step < n; step++) {
        Pivot p = largest_unused_pivot(A, pivot_used);
        if (abs(A[p.row][p.col]) == 0) error("singular matrix");
        pivot_used[p.col]++;

        if (p.row != p.col) {
            swap_rows(A, p.row, p.col);
            swap_rows(B, p.row, p.col);
        }
        pivot_row[step] = p.row;
        pivot_col[step] = p.col;

        double inv = 1.0 / A[p.col][p.col];
        scale_row(A, p.col, inv);
        scale_row(B, p.col, inv);

        for (int r = 0; r < n; r++) if (r != p.col) {
            double factor = A[r][p.col];
            A[r][p.col] = 0;
            subtract_scaled_row(A, r, p.col, factor);
            subtract_scaled_row(B, r, p.col, factor);
        }
    }
    for (int step = n - 1; step >= 0; step--)
        if (pivot_row[step] != pivot_col[step])
            swap_columns(A, pivot_row[step], pivot_col[step]);
}
\end{lstlisting}

\subsubsection{Algorithm: Gaussian elimination with backsubstitution}\label{algorithm-gaussian-elimination-with-backsubstitution}

\begin{lstlisting}[language=C]
Vector solve_by_elimination(Matrix A, Vector b) {
    int n = A.rows;
    for (int k = 0; k < n; k++) {
        int p = row_with_largest_abs_in_column(A, k, k, n - 1);
        swap_rows(A, k, p);
        swap_entries(b, k, p);
        if (A[k][k] == 0) error("singular matrix");

        for (int i = k + 1; i < n; i++) {
            double factor = A[i][k] / A[k][k];
            A[i][k] = 0;
            for (int j = k + 1; j < n; j++) A[i][j] -= factor * A[k][j];
            b[i] -= factor * b[k];
        }
    }
    return back_substitute(A, b);
}
\end{lstlisting}

\subsubsection{Algorithm: LU decomposition with partial pivoting}\label{algorithm-lu-decomposition-with-partial-pivoting}

\begin{lstlisting}[language=C]
LUFactor lu_decompose(Matrix A) {
    int n = A.rows;
    int perm[n];
    double row_scale[n];
    int parity = 1;

    for (int i = 0; i < n; i++) row_scale[i] = 1.0 / max_abs_row(A, i);

    for (int k = 0; k < n; k++) {
        int p = argmax_i(row_scale[i] * abs(A[i][k]), i = k..n-1);
        if (p != k) { swap_rows(A, p, k); swap(row_scale[p], row_scale[k]); parity = -parity; }
        perm[k] = p;
        if (A[k][k] == 0) A[k][k] = TINY;

        for (int i = k + 1; i < n; i++) {
            A[i][k] /= A[k][k];                 // lower multiplier
            for (int j = k + 1; j < n; j++)
                A[i][j] -= A[i][k] * A[k][j];   // Schur update
        }
    }
    return LUFactor(A, perm, parity);
}
\end{lstlisting}

\subsubsection{Algorithm: Solve using LU factors}\label{algorithm-solve-using-lu-factors}

\begin{lstlisting}[language=C]
Vector lu_solve(LUFactor F, Vector b) {
    int n = b.n;
    Vector x = b;
    for (int i = 0; i < n; i++) swap(x[i], x[F.perm[i]]);

    for (int i = 0; i < n; i++) {               // forward substitution, L has unit diagonal
        for (int j = 0; j < i; j++) x[i] -= F.LU[i][j] * x[j];
    }
    for (int i = n - 1; i >= 0; i--) {          // back substitution through U
        for (int j = i + 1; j < n; j++) x[i] -= F.LU[i][j] * x[j];
        x[i] /= F.LU[i][i];
    }
    return x;
}
\end{lstlisting}

\subsubsection{Algorithm: Tridiagonal linear solver}\label{algorithm-tridiagonal-linear-solver}

\begin{lstlisting}[language=C]
Vector solve_tridiagonal(Vector lower, Vector diag, Vector upper, Vector rhs) {
    int n = diag.n;
    Vector cprime(n), x(n);
    double pivot = diag[0];
    cprime[0] = upper[0] / pivot;
    x[0] = rhs[0] / pivot;

    for (int i = 1; i < n; i++) {
        pivot = diag[i] - lower[i] * cprime[i - 1];
        if (pivot == 0) error("singular tridiagonal system");
        cprime[i] = (i < n - 1) ? upper[i] / pivot : 0;
        x[i] = (rhs[i] - lower[i] * x[i - 1]) / pivot;
    }
    for (int i = n - 2; i >= 0; i--) x[i] -= cprime[i] * x[i + 1];
    return x;
}
\end{lstlisting}

\subsubsection{Algorithm: Singular value decomposition}\label{algorithm-singular-value-decomposition}

\begin{lstlisting}[language=C]
SVD svd(Matrix A) {
    Matrix U = A;
    Vector W;
    Matrix V = identity(A.cols);

    for (int col = 0; col < A.cols; col++)
        householder_bidiagonalize_column_and_row(U, W, V, col);

    accumulate_right_transformations(V);
    accumulate_left_transformations(U);

    for (int k = A.cols - 1; k >= 0; k--) {
        for (int iter = 0; iter < MAX_ITER; iter++) {
            int split = find_negligible_offdiagonal(W, U, k);
            if (split == k) break;
            apply_implicit_qr_shift_to_bidiagonal(U, W, V, split, k);
        }
        force_singular_value_nonnegative(W[k], V, k);
    }
    sort_singular_values_descending(U, W, V);
    return SVD(U, W, V);
}
\end{lstlisting}

\subsubsection{Algorithm: Solve least squares with SVD thresholding}\label{algorithm-solve-least-squares-with-svd-thresholding}

\begin{lstlisting}[language=C]
Vector svd_solve(SVD S, Vector b, double threshold) {
    Vector tmp(S.W.n);
    for (int j = 0; j < S.W.n; j++) {
        double projection = dot(column(S.U, j), b);
        tmp[j] = (S.W[j] > threshold) ? projection / S.W[j] : 0;
    }
    return multiply(S.V, tmp);
}
\end{lstlisting}

\subsubsection{Algorithm: Sparse conjugate gradient}\label{algorithm-sparse-conjugate-gradient}

\begin{lstlisting}[language=C]
Vector conjugate_gradient(SparseMatrix A, Vector b, Vector x) {
    Vector r = b - A * x;
    Vector z = precondition(r);
    Vector p = z;
    double rz_old = dot(r, z);

    for (int iter = 0; iter < MAX_ITER; iter++) {
        Vector Ap = A * p;
        double alpha = rz_old / dot(p, Ap);
        x += alpha * p;
        r -= alpha * Ap;
        if (norm(r) < TOL) break;
        z = precondition(r);
        double rz_new = dot(r, z);
        p = z + (rz_new / rz_old) * p;
        rz_old = rz_new;
    }
    return x;
}
\end{lstlisting}

\subsubsection{Algorithm: Cholesky decomposition and solve}\label{algorithm-cholesky-decomposition-and-solve}

\begin{lstlisting}[language=C]
Cholesky cholesky(Matrix A) {
    int n = A.rows;
    Matrix L(n, n, 0);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j <= i; j++) {
            double s = A[i][j];
            for (int k = 0; k < j; k++) s -= L[i][k] * L[j][k];
            if (i == j) {
                if (s <= 0) error("not positive definite");
                L[i][j] = sqrt(s);
            } else {
                L[i][j] = s / L[j][j];
            }
        }
    }
    return Cholesky(L);
}

Vector cholesky_solve(Cholesky C, Vector b) {
    Vector y = forward_substitute(C.L, b);
    return back_substitute(transpose(C.L), y);
}
\end{lstlisting}

\subsubsection{Algorithm: QR decomposition by Householder reflectors}\label{algorithm-qr-decomposition-by-householder-reflectors}

\begin{lstlisting}[language=C]
QR qr_decompose(Matrix A) {
    int m = A.rows, n = A.cols;
    Vector tau(n);
    for (int k = 0; k < n; k++) {
        Vector v = householder_vector(A[k..m-1][k]);
        tau[k] = reflector_scale(v);
        apply_reflector_left(A[k..m-1][k..n-1], v, tau[k]);
        store_reflector_below_diagonal(A, k, v);
    }
    return QR(A, tau);
}

Vector qr_solve(QR F, Vector b) {
    apply_Q_transpose_to_vector(F, b);

\subsubsection{Algorithm: Binary locate in an ordered table}\label{algorithm-binary-locate-in-an-ordered-table}

\begin{lstlisting}[language=C]
int locate(Vector x, double target) {
    int lo = 0, hi = x.n - 1;
    bool ascending = x[hi] >= x[lo];
    while (hi - lo > 1) {
        int mid = (hi + lo) / 2;
        if ((target >= x[mid]) == ascending) lo = mid;
        else hi = mid;
    }
    return lo;
}

\subsubsection{Algorithm: Hunt from previous index}\label{algorithm-hunt-from-previous-index}

\begin{lstlisting}[language=C]
int hunt(Vector x, double target, int guess) {
    int lo, hi, inc = 1;
    if (target >= x[guess]) {
        lo = guess;
        do { hi = lo + inc; inc *= 2; lo = hi; }
        while (hi < x.n - 1 && target >= x[hi]);
        hi = min(hi, x.n - 1);
    } else {
        hi = guess;
        do { lo = hi - inc; inc *= 2; hi = lo; }
        while (lo > 0 && target < x[lo]);
        lo = max(lo, 0);
    }
    return binary_refine(x, target, lo, hi);
}

\subsubsection{Algorithm: Neville polynomial interpolation}\label{algorithm-neville-polynomial-interpolation}

\begin{lstlisting}[language=C]
InterpResult polynomial_interpolate(Vector xa, Vector ya, double x) {
    int n = xa.n;
    Vector c = ya, d = ya;
    int nearest = index_min_abs(xa - x);
    double y = ya[nearest--];

    for (int m = 1; m < n; m++) {
        for (int i = 0; i < n - m; i++) {
            double left = xa[i] - x;
            double right = xa[i + m] - x;
            double denom = left - right;
            if (denom == 0) error("duplicate interpolation node");
            double w = (c[i + 1] - d[i]) / denom;
            d[i] = right * w;
            c[i] = left * w;
        }
        double dy = (2 * (nearest + 1) < n - m) ? c[nearest + 1] : d[nearest--];
        y += dy;
    }
    return InterpResult(y, last_error_estimate);
}
\end{lstlisting}

\subsubsection{Algorithm: Cubic spline second derivatives}\label{algorithm-cubic-spline-second-derivatives}

\begin{lstlisting}[language=C]
Vector spline_second_derivatives(Vector x, Vector y, Boundary bc) {
    int n = x.n;
    Vector y2(n), u(n - 1);
    initialize_left_boundary(y2, u, x, y, bc.left);
    for (int i = 1; i < n - 1; i++) {
        double sig = (x[i] - x[i - 1]) / (x[i + 1] - x[i - 1]);
        double p = sig * y2[i - 1] + 2.0;
        y2[i] = (sig - 1.0) / p;
        u[i] = (slope(x,y,i+1,i) - slope(x,y,i,i-1));
        u[i] = (6.0 * u[i] / (x[i + 1] - x[i - 1]) - sig * u[i - 1]) / p;
    }
    apply_right_boundary(y2, u, x, y, bc.right);
    for (int k = n - 2; k >= 0; k--) y2[k] = y2[k] * y2[k + 1] + u[k];
    return y2;
}
\end{lstlisting}

\subsubsection{Algorithm: Cubic spline interpolation}\label{algorithm-cubic-spline-interpolation}

double spline_eval(Vector x, Vector y, Vector y2, double xq) {
    int klo = locate(x, xq);
    int khi = klo + 1;
    double h = x[khi] - x[klo];
    double a = (x[khi] - xq) / h;
    double b = (xq - x[klo]) / h;
    return a*y[klo] + b*y[khi]
         + ((a*a*a - a)*y2[klo] + (b*b*b - b)*y2[khi]) * h*h / 6.0;

    int nearest = index_min_abs(xa - x);
    double y = ya[nearest--];

            c[i] = t * denom;

\subsection{6. Integration}\label{integration}

\subsubsection{Algorithm: Composite trapezoid refinement}\label{algorithm-composite-trapezoid-refinement}

\begin{lstlisting}[language=C]
double trapezoid_next(Function f, double a, double b, int level, double previous) {
    if (level == 1) return 0.5 * (b - a) * (f(a) + f(b));
    int new_points = 1 << (level - 2);
    double h = (b - a) / new_points;
    double x = a + 0.5 * h;
    double sum = 0;
    for (int i = 0; i < new_points; i++, x += h) sum += f(x);
    return 0.5 * (previous + (b - a) * sum / new_points);
}
\end{lstlisting}

\subsubsection{Algorithm: Composite midpoint refinement}\label{algorithm-composite-midpoint-refinement}

\begin{lstlisting}[language=C]
double midpoint_next(Function f, double a, double b, int level, double previous) {
    if (level == 1) return (b - a) * f(0.5 * (a + b));
    int intervals = pow(3, level - 2);
    double h = (b - a) / (3 * intervals);
    double sum = 0;
    double x = a + 0.5 * h;
    for (int i = 0; i < intervals; i++) {
        sum += f(x);
        x += 2 * h;
        sum += f(x);
        x += h;
    }
    return (previous + (b - a) * sum / intervals) / 3.0;
}
\end{lstlisting}

            if (estimated_converged(extrap, previous_extrap)) return extrap;

    }

}

\subsubsection{Algorithm: Adaptive Simpson integration}\label{algorithm-adaptive-simpson-integration}

double adaptive_simpson(Function f, double a, double b, double tol) {
    double c = (a + b) / 2;
    double S = simpson_estimate(f, a, b);
    return recurse(f, a, c, b, S, tol, MAX_DEPTH);
}

double recurse(Function f, double a, double c, double b, double S, double tol, int depth) {
    double left = simpson_estimate(f, a, c);
    double right = simpson_estimate(f, c, b);
    if (depth == 0 || abs(left + right - S) < 15 * tol)
        return left + right + (left + right - S) / 15.0;
    return recurse(f, a, (a+c)/2, c, left, tol/2, depth-1)
         + recurse(f, c, (c+b)/2, b, right, tol/2, depth-1);
}

\begin{lstlisting}[language=C]
MCResult monte_carlo(Function f, Box box, int samples) {
    double sum = 0, sumsq = 0;
    for (int i = 0; i < samples; i++) {
        Point x = uniform_point(box);
        double y = f(x);
        sum += y;
        sumsq += y * y;
    }
    double mean = sum / samples;
    double variance = (sumsq / samples - mean * mean) / samples;
    return MCResult(volume(box) * mean, volume(box) * sqrt(variance));
}
\end{lstlisting}

    double f = tiny_nonzero(coeff.b0());
    double C = f, D = 0;
    for (int n = 1; n <= MAX_ITER; n++) {
        D = tiny_nonzero(coeff.b(n) + coeff.a(n) * D);

Derivative ridders_derivative(Function f, double x) {
    Matrix D(MAX_LEVEL, MAX_LEVEL);
    double h = INITIAL_H;
    for (int i = 0; i < MAX_LEVEL; i++) {
        D[i][0] = (f(x + h) - f(x - h)) / (2*h);
        double factor = STEP_RATIO * STEP_RATIO;
        for (int j = 1; j <= i; j++) {
            D[i][j] = (factor * D[i][j-1] - D[i-1][j-1]) / (factor - 1);
            factor *= STEP_RATIO * STEP_RATIO;
        }
        if (converged_by_error_table(D, i)) return best_entry(D);
        h /= STEP_RATIO;
    }
    return best_entry(D);
}

Vector chebyshev_fit(Function f, double a, double b, int n) {
    Vector c(n);
    for (int k = 0; k < n; k++) {
        double sum = 0;
        for (int j = 0; j < n; j++) {
            double y = cos(PI * (j + 0.5) / n);
            double x = 0.5 * ((b - a) * y + (b + a));
            sum += f(x) * cos(PI * k * (j + 0.5) / n);
        }
        c[k] = 2.0 * sum / n;
    }
    return c;
}

    Vector rhs(n);
    for (int row = 0; row < n; row++) {
        for (int col = 0; col < n; col++) A[row][col] = series[m + row - col];
        rhs[row] = -series[m + row + 1];

    Vector q = concat({1}, qtail);
    Vector p(m + 1);

    double factor = sqrt(-2 * log(s) / s);
    saved = v * factor;
    has_saved = true;
    return u * factor;
}

double bisection(Function f, double lo, double hi) {
    double flo = f(lo), fhi = f(hi);
    assert(flo * fhi <= 0);
    for (int iter = 0; iter < MAX_ITER; iter++) {
        double mid = 0.5 * (lo + hi);
        double fm = f(mid);
        if (abs(fm) < FTOL || abs(hi - lo) < XTOL) return mid;
        if (flo * fm <= 0) { hi = mid; fhi = fm; }
        else              { lo = mid; flo = fm; }

\begin{lstlisting}[language=C]
double ridders(Function f, double x1, double x2) {
    double f1 = f(x1), f2 = f(x2);
    for (int iter = 0; iter < MAX_ITER; iter++) {
        double xm = 0.5 * (x1 + x2);
        double fm = f(xm);
        double s = sqrt(fm*fm - f1*f2);
        if (s == 0) return xm;
        double xnew = xm + (xm - x1) * sign(f1 - f2) * fm / s;
        double fnew = f(xnew);
        if (abs(fnew) < FTOL) return xnew;
        update_bracket_using(x1, f1, xm, fm, x2, f2, xnew, fnew);
    }
    error("Ridders failed");
}

\subsubsection{Algorithm: Brent root finder}\label{algorithm-brent-root-finder}

\begin{lstlisting}[language=C]
double brent_root(Function f, double a, double b) {
    double c = b, fa = f(a), fb = f(b), fc = fb;
    double d = 0, e = 0;
    for (int iter = 0; iter < MAX_ITER; iter++) {
        if (fb * fc > 0) { c = a; fc = fa; e = d = b - a; }
        if (abs(fc) < abs(fb)) rotate(a,b,c, fa,fb,fc);
        double tol = 2*EPS*abs(b) + 0.5*XTOL;
        double m = 0.5 * (c - b);
        if (abs(m) <= tol || fb == 0) return b;
        if (abs(e) >= tol && abs(fa) > abs(fb)) d = inverse_quadratic_or_secant_step(a,b,c,fa,fb,fc,m);
        else { d = m; e = m; }
        a = b; fa = fb;
        b += (abs(d) > tol) ? d : sign(tol, m);
        fb = f(b);
    }
    error("Brent failed");
}

Complex laguerre_root(VectorComplex a, Complex x) {

Vector nelder_mead(Function f, Simplex S) {
    for (int iter = 0; iter < MAX_ITER; iter++) {
        order_vertices_by_value(S, f);
        if (simplex_size(S) < XTOL && value_spread(S,f) < FTOL) return best_vertex(S);
        Point centroid = centroid_excluding_worst(S);
        Point reflected = centroid + ALPHA * (centroid - worst(S));
        if (f(reflected) < f(best(S))) {
            Point expanded = centroid + GAMMA * (reflected - centroid);
            replace_worst(S, f(expanded) < f(reflected) ? expanded : reflected);
        } else if (f(reflected) < f(second_worst(S))) {
            replace_worst(S, reflected);
        } else {
            Point contracted = centroid + RHO * (worst(S) - centroid);
            if (f(contracted) < f(worst(S))) replace_worst(S, contracted);
            else shrink_toward_best(S, SIGMA);
        }
    }
    return best_vertex(S);
}

        Vector x_start = x;
        double f_start = f(x);
        int best_dir = -1;
        double biggest_drop = 0;
        for (int i = 0; i < directions.cols; i++) {
            double before = f(x);
            x = line_minimize(f, x, directions.col(i));
            double drop = before - f(x);

        Vector new_direction = x - x_start;

\subsubsection{Algorithm: Conjugate gradient minimization}\label{algorithm-conjugate-gradient-minimization}

\begin{lstlisting}[language=C]
Vector conjugate_gradient_min(Function f, Gradient grad, Vector x) {
    Vector g = -grad(x);
    Vector h = g;
    for (int iter = 0; iter < MAX_ITER; iter++) {
        x = line_minimize(f, x, h);
        Vector g_new = -grad(x);
        if (norm(g_new) < GTOL) return x;
        double gamma = dot(g_new - g, g_new) / max(dot(g, g), TINY);
        h = g_new + gamma * h;
        g = g_new;
    }
    return x;
}

\subsubsection{Algorithm: BFGS quasi-Newton minimization}\label{algorithm-bfgs-quasi-newton-minimization}

\begin{lstlisting}[language=C]
Vector bfgs(Function f, Gradient grad, Vector x) {
    Matrix H = identity(x.n);
    Vector g = grad(x);
    for (int iter = 0; iter < MAX_ITER; iter++) {
        Vector p = -H * g;
        Vector x_new = line_search_wolfe(f, grad, x, p);
        Vector s = x_new - x;
        Vector g_new = grad(x_new);
        Vector y = g_new - g;
        double rho = 1.0 / dot(y, s);
        H = (I - rho*s*y^T) * H * (I - rho*y*s^T) + rho*s*s^T;
        x = x_new; g = g_new;
        if (norm(g) < GTOL) return x;
    }
    return x;
}

\subsection{14. FFT and Spectral Applications}\label{fft-and-spectral-applications}

\subsubsection{Algorithm: In-place radix-2 complex FFT}\label{algorithm-in-place-radix-2-complex-fft}

\begin{lstlisting}[language=C]
void fft(Complex data[], int n, int sign) {
    bit_reverse_permute(data, n);
    for (int len = 2; len <= n; len *= 2) {
        Complex wlen = exp(sign * 2*pi*i / len);
        for (int start = 0; start < n; start += len) {
            Complex w = 1;
            for (int j = 0; j < len/2; j++) {
                Complex u = data[start+j];
                Complex v = w * data[start+j+len/2];
                data[start+j] = u + v;
                data[start+j+len/2] = u - v;
                w *= wlen;

\subsubsection{Algorithm: Linear prediction coefficients by Levinson-Durbin}\label{algorithm-linear-prediction-coefficients-by-levinson-durbin}

\begin{lstlisting}[language=C]
Vector levinson_durbin(Vector autocorr, int order) {
    Vector a(order + 1); a[0] = 1;
    double error = autocorr[0];
    for (int m = 1; m <= order; m++) {
        double acc = autocorr[m];
        for (int j = 1; j < m; j++) acc += a[j] * autocorr[m - j];
        double kappa = -acc / error;
        Vector old = a;
        a[m] = kappa;
        for (int j = 1; j < m; j++) a[j] = old[j] + kappa * old[m - j];
        error *= (1 - kappa*kappa);
    }
    return a;
}

\subsubsection{Algorithm: Lomb-Scargle uneven-sample periodogram}\label{algorithm-lomb-scargle-uneven-sample-periodogram}

\begin{lstlisting}[language=C]
Spectrum lomb_scargle(Vector t, Vector y, Vector omega) {
    subtract_mean(y);
    for each frequency w in omega {
        double tau = atan2(sum(sin(2*w*t)), sum(cos(2*w*t))) / (2*w);
        double csum = 0, ssum = 0, cc = 0, ss = 0;
        for (int i = 0; i < t.n; i++) {
            double c = cos(w * (t[i] - tau));
            double s = sin(w * (t[i] - tau));
            csum += y[i]*c; ssum += y[i]*s;
            cc += c*c; ss += s*s;
        }
        P[w] = 0.5 * (csum*csum/cc + ssum*ssum/ss) / variance(y);
    }
    return P;
}

Vector savitzky_golay(int left, int right, int derivative, int degree) {
    int n = left + right + 1;
    Matrix A(degree+1, degree+1);
    for (int i = 0; i <= degree; i++) for (int j = 0; j <= degree; j++)

    for (int iter = 0; iter < MAX_ITER; iter++) {

\subsubsection{Algorithm: Lax-Wendroff hyperbolic update}\label{algorithm-lax-wendroff-hyperbolic-update}

\begin{lstlisting}[language=C]
Grid lax_wendroff(Grid u, double dt, double dx, Flux F) {
    for each cell i:
        half[i+1/2] = 0.5*(u[i+1] + u[i]) - 0.5*(dt/dx)*(F(u[i+1]) - F(u[i]));
    for each cell i:
        u_new[i] = u[i] - (dt/dx)*(F(half[i+1/2]) - F(half[i-1/2]));
    apply_boundary_conditions(u_new);
    return u_new;
}

\subsubsection{Algorithm: Crank-Nicolson diffusion step}\label{algorithm-crank-nicolson-diffusion-step}

\begin{lstlisting}[language=C]
Vector crank_nicolson(Vector u, double r) {
    Tridiagonal A, B;
    for each interior i:
        A.lower[i] = -r/2; A.diag[i] = 1+r; A.upper[i] = -r/2;
        rhs[i] = (r/2)*u[i-1] + (1-r)*u[i] + (r/2)*u[i+1];
    apply_boundary_terms(A, rhs);
    return solve_tridiagonal(A.lower, A.diag, A.upper, rhs);
}
\end{lstlisting}

\subsubsection{Algorithm: Successive over-relaxation}\label{algorithm-successive-over-relaxation}

\begin{lstlisting}[language=C]
Grid SOR(Grid u, Function rhs, double omega) {
    for (int sweep = 0; sweep < MAX_SWEEPS; sweep++) {
        double max_change = 0;
        for each interior point p in red_black_or_lexicographic_order:
            double residual = stencil_laplacian(u,p) - rhs(p);
            double delta = -omega * residual / stencil_diagonal();
            u[p] += delta;
            max_change = max(max_change, abs(delta));
        if (max_change < TOL) break;
    }
    return u;
}

void nearest(KDNode node, Point q, Best *best) {
    if (node == NULL || distance_to_box(q, node.box) >= best->dist) return;
    double d = distance(q, node.point);
    if (d < best->dist) update(best, node.point, d);
    KDNode first = q[node.axis] < node.point[node.axis] ? node.left : node.right;
    KDNode second = first == node.left ? node.right : node.left;
    nearest(first, q, best);
    nearest(second, q, best);
}

        if (edge_straddles_horizontal_ray(a,b,q)) {

void arithmetic_encode(SymbolStream s, Model M) {
    Range low = 0, high = TOP;
    for each symbol c in s {
        Range range = high - low + 1;
        high = low + range * M.cum_high(c) / M.total - 1;
        low  = low + range * M.cum_low(c)  / M.total;
        while (top_bits_equal(low, high)) output_top_bit_and_shift(low, high);
        while (underflow_condition(low, high)) handle_underflow_bit(low, high);
        M.update(c);
    }
    flush_final_bits(low);
}

BigInt add(BigInt a, BigInt b) {
    BigInt c;
    int carry = 0;
    for (int i = 0; i < max(a.len,b.len); i++) {
        int s = digit(a,i) + digit(b,i) + carry;
        c.digit[i] = s % BASE;
        carry = s / BASE;
    }
    if (carry) append_digit(c, carry);
    return normalize(c);
}

BigInt multiply(BigInt a, BigInt b) {
    BigInt c = zero_with_length(a.len + b.len);
    for (int i = 0; i < a.len; i++) {
        long carry = 0;
        for (int j = 0; j < b.len || carry; j++) {
            long cur = c[i+j] + carry + (long)a[i] * digit(b,j);
            c[i+j] = cur % BASE;
            carry = cur / BASE;
        }
    }
    return normalize(c);
}

\section{1. Core String Utilities}\label{core-string-utilities}

\subsection{1.1 Naive exact matching}\label{naive-exact-matching}

\begin{lstlisting}[language=C]
vector<int> naive_match(string P, string T) {
    int n = P.length, m = T.length;
    vector<int> occ;

    for (int s = 0; s + n <= m; s++) {
        int i = 0;
        while (i < n && P[i] == T[s + i]) i++;
        if (i == n) occ.push_back(s);
    }
    return occ;
}
\end{lstlisting}

\textbf{Use.} Baseline exact matching.\\
\textbf{Cost.} Worst case \passthrough{\lstinline!O(nm)!} comparisons.

\subsection{1.2 Longest common prefix by direct comparison}\label{longest-common-prefix-by-direct-comparison}

\begin{lstlisting}[language=C]
int lcp_direct(string A, int i, string B, int j) {
    int k = 0;
    while (i + k < A.length && j + k < B.length && A[i + k] == B[j + k])
        k++;
    return k;
}
\end{lstlisting}

\textbf{Use.} Primitive in naive suffix-tree construction, direct matching, and verification steps.

\section{2. Fundamental Preprocessing: Z Values}\label{fundamental-preprocessing-z-values}

For string \passthrough{\lstinline!S!}, \passthrough{\lstinline!Z[i]!} is the length of the longest prefix of \passthrough{\lstinline!S!} that matches the substring starting at \passthrough{\lstinline!i!}.

\subsection{2.1 Linear-time Z algorithm}\label{linear-time-z-algorithm}

\begin{lstlisting}[language=C]
vector<int> z_values(string S) {
    int n = S.length;
    vector<int> Z(n, 0);

    int L = 0;          // left boundary of current rightmost Z-box
    int R = 0;          // right boundary, exclusive

    for (int i = 1; i < n; i++) {
        if (i >= R) {
            int k = 0;
            while (i + k < n && S[k] == S[i + k]) k++;
            Z[i] = k;
            if (k > 0) { L = i; R = i + k; }
        } else {
            int k = i - L;
            int beta = R - i;

            if (Z[k] < beta) {
                Z[i] = Z[k];
            } else {
                int q = beta;
                while (i + q < n && S[q] == S[i + q]) q++;
                Z[i] = q;
                L = i; R = i + q;
            }
        }
    }
    return Z;
}
\end{lstlisting}

\textbf{Invariant.} \passthrough{\lstinline![L, R)!} is the rightmost discovered substring matching a prefix.\\
\textbf{Cost.} \passthrough{\lstinline!O(|S|)!}.

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection{2.2 Exact matching using Z values}\label{exact-matching-using-z-values}

\begin{lstlisting}[language=C]
vector<int> z_exact_match(string P, string T) {
    char sentinel = choose_char_not_in(P, T);
    string S = P + sentinel + T;
    vector<int> Z = z_values(S);

    int n = P.length;
    vector<int> occ;

    for (int i = n + 1; i < S.length; i++) {
        if (Z[i] == n) {
            occ.push_back(i - (n + 1));
        }
    }
    return occ;
}
\end{lstlisting}

\textbf{Use.} Simplest linear-time exact matcher.\\
\textbf{Cost.} \passthrough{\lstinline!O(n + m)!}.

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection{2.3 Suffix-prefix matching via Z values}\label{suffix-prefix-matching-via-z-values}

Given strings \passthrough{\lstinline!A!} and \passthrough{\lstinline!B!}, find the longest suffix of \passthrough{\lstinline!A!} that is a prefix of \passthrough{\lstinline!B!}.

\begin{lstlisting}[language=C]
int longest_suffix_prefix(string A, string B) {
    char sentinel = choose_char_not_in(A, B);
    string S = B + sentinel + A;
    vector<int> Z = z_values(S);

    int best = 0;
    int offset = B.length + 1;

    for (int i = offset; i < S.length; i++) {
        int len = Z[i];
        if (i + len == S.length) {
            best = max(best, len);
        }
    }
    return best;
}
\end{lstlisting}

bool is_circular_rotation(string A, string B) {

\subsection{3.3 N array for Boyer-Moore}\label{n-array-for-boyer-moore}

\passthrough{\lstinline!N[j]!} is the length of the longest suffix of \passthrough{\lstinline!P[0..j]!} that is also a suffix of \passthrough{\lstinline!P!}.

\begin{lstlisting}[language=C]
vector<int> n_values(string P) {
    string R = reverse(P);
    vector<int> Zr = z_values(R);
    int n = P.length;
    vector<int> N(n, 0);

    for (int j = 0; j < n; j++) {
        N[j] = Zr[n - 1 - j];
    }
    return N;
}
\end{lstlisting}

\subsection{\texorpdfstring{3.4 Strong good-suffix table \texttt{Lprime}}{3.4 Strong good-suffix table Lprime}}\label{strong-good-suffix-table-lprime}

\passthrough{\lstinline!Lprime[i]!} is the rightmost endpoint of a copy of suffix \passthrough{\lstinline!P[i..n)!} that is preceded by a different character.

\begin{lstlisting}[language=C]
vector<int> bm_Lprime(string P) {
    int n = P.length;
    vector<int> N = n_values(P);
    vector<int> Lp(n + 1, -1);

    for (int j = 0; j < n - 1; j++) {
        int len = N[j];
        if (len > 0) {
            int i = n - len;
            Lp[i] = j;       // endpoint j of matching copy
        }
    }
    return Lp;
}
\end{lstlisting}

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection{\texorpdfstring{3.5 Small \texttt{lprime} table}{3.5 Small lprime table}}\label{small-lprime-table}

\passthrough{\lstinline!lprime[i]!} is the length of the largest suffix of \passthrough{\lstinline!P[i..n)!} that is also a prefix of \passthrough{\lstinline!P!}.

\begin{lstlisting}[language=C]
vector<int> bm_lprime(string P) {
    int n = P.length;
    vector<int> N = n_values(P);
    vector<int> lp(n + 1, 0);

    int longest = 0;
    for (int j = 0; j < n; j++) {
        if (N[j] == j + 1) {
            longest = j + 1;
        }
        int i = n - 1 - j;
        lp[i] = longest;
    }
    return lp;
}
\end{lstlisting}

\subsection{3.6 Boyer-Moore search}\label{boyer-moore-search}

\begin{lstlisting}[language=C]
vector<int> boyer_moore(string P, string T) {
    int n = P.length, m = T.length;
    vector<int> occ;
    auto R = bad_char_rightmost(P);
    vector<int> Lp = bm_Lprime(P);
    vector<int> lp = bm_lprime(P);

    int k = n - 1;               // text index under P[n-1]

    while (k < m) {
        int i = n - 1;
        int h = k;

        while (i >= 0 && P[i] == T[h]) {
            i--; h--;
        }

        if (i < 0) {
            occ.push_back(h + 1);
            k += max(1, n - lp[1]);
        } else {
            int bc = i - (R.count(T[h]) ? R[T[h]] : -1);
            bc = max(1, bc);

            int gs = 1;
            int suffix_start = i + 1;
            if (suffix_start < n) {
                if (Lp[suffix_start] >= 0)
                    gs = n - 1 - Lp[suffix_start];
                else
                    gs = n - lp[suffix_start];
            }

            k += max(bc, gs);
        }
    }
    return occ;
}
\end{lstlisting}

\textbf{Notes.} This is the standard structure: right-to-left comparison, bad-character shift, strong good-suffix shift.\\
\textbf{Cost.} Linear worst case with the correct full implementation; often sublinear in practice.

\subsection{3.7 Galil rule for overlapping occurrences}\label{galil-rule-for-overlapping-occurrences}

\begin{lstlisting}[language=C]
vector<int> boyer_moore_galil(string P, string T) {
    int n = P.length, m = T.length;
    vector<int> occ;
    auto R = bad_char_rightmost(P);
    vector<int> Lp = bm_Lprime(P);
    vector<int> lp = bm_lprime(P);

    int k = n - 1;
    int protected_left = -1;   // left boundary known matched
    int protected_right = -1;  // right boundary known matched

    while (k < m) {
        int i = n - 1;
        int h = k;

        while (i >= 0) {
            if (protected_left <= h && h <= protected_right) {
                int skip = h - protected_left + 1;
                i -= skip;
                h -= skip;
                continue;
            }
            if (P[i] != T[h]) break;
            i--; h--;
        }

        if (i < 0) {
            occ.push_back(h + 1);
            int shift = max(1, n - lp[1]);
            protected_left = k - n + 1 + shift;
            protected_right = k;
            k += shift;
        } else {
            int bc = i - (R.count(T[h]) ? R[T[h]] : -1);
            int suffix_start = i + 1;
            int gs = 1;
            if (suffix_start < n) {
                if (Lp[suffix_start] >= 0) gs = n - 1 - Lp[suffix_start];
                else gs = n - lp[suffix_start];
            }
            k += max(1, max(bc, gs));
            protected_left = protected_right = -1;
        }
    }
    return occ;
}
\end{lstlisting}

\subsection{3.8 Apostolico-Giancarlo Boyer-Moore variant}\label{apostolico-giancarlo-boyer-moore-variant}

\begin{lstlisting}[language=C]
vector<int> apostolico_giancarlo(string P, string T) {
    int n = P.length, m = T.length;
    vector<int> occ;
    auto R = bad_char_rightmost(P);
    vector<int> N = n_values(P);
    vector<int> M(m, -1);        // M[j] records matched suffix length ending at T[j]

    int k = n - 1;

    while (k < m) {
        int i = n - 1;
        int h = k;

        while (i >= 0) {
            if (M[h] < 0 || M[h] == 0 || N[i] == 0) {
                if (P[i] != T[h]) break;
                i--; h--;
                continue;
            }

            if (M[h] < N[i]) {
                i -= M[h];
                h -= M[h];
                continue;
            }

            if (M[h] > N[i]) {
                if (N[i] == i + 1) {
                    i = -1;       // entire pattern matched
                } else {
                    i -= N[i] + 1; // infer mismatch after skipped match
                    h -= N[i] + 1;
                }
                break;
            }

            // M[h] == N[i]
            i -= M[h];
            h -= M[h];
        }

        int matched = k - h;
        if (i < 0) {
            occ.push_back(k - n + 1);
            M[k] = n;
            k += max(1, n - bm_lprime(P)[1]);
        } else {
            M[k] = matched;
            int bc = max(1, i - (R.count(T[h]) ? R[T[h]] : -1));
            k += bc; // can be combined with good-suffix shift
        }
    }

    return occ;
}
\end{lstlisting}

\textbf{Purpose.} Simulate Boyer-Moore shifts while avoiding many repeated explicit matches.

\section{4. Knuth-Morris-Pratt and Real-Time Matching}\label{knuth-morris-pratt-and-real-time-matching}

\subsection{\texorpdfstring{4.1 Compute \texttt{sp} from Z values}{4.1 Compute sp from Z values}}\label{compute-sp-from-z-values}

\passthrough{\lstinline!sp[i]!} is the longest proper suffix of \passthrough{\lstinline!P[0..i]!} that is also a prefix of \passthrough{\lstinline!P!}.

\begin{lstlisting}[language=C]
vector<int> sp_from_z(string P) {
    int n = P.length;
    vector<int> Z = z_values(P);
    vector<int> sp(n, 0);

    for (int j = 1; j < n; j++) {
        int i = j + Z[j] - 1;
        if (i < n && Z[j] > 0)
            sp[i] = max(sp[i], Z[j]);
    }

    // propagate weaker sp values if desired
    for (int i = n - 2; i >= 0; i--) {
        sp[i] = max(sp[i], sp[i + 1] - 1);
    }
    return sp;
}
\end{lstlisting}

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection{4.2 Classical KMP prefix-function construction}\label{classical-kmp-prefix-function-construction}

\begin{lstlisting}[language=C]
vector<int> prefix_function(string P) {
    int n = P.length;
    vector<int> pi(n, 0);

    for (int i = 1; i < n; i++) {
        int k = pi[i - 1];

        while (k > 0 && P[k] != P[i])
            k = pi[k - 1];

        if (P[k] == P[i]) k++;
        pi[i] = k;
    }

    return pi;
}
\end{lstlisting}

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection{4.3 KMP search}\label{kmp-search}

\begin{lstlisting}[language=C]
vector<int> kmp_match(string P, string T) {
    int n = P.length, m = T.length;
    vector<int> pi = prefix_function(P);
    vector<int> occ;

    int q = 0;                  // matched length
    for (int i = 0; i < m; i++) {
        while (q > 0 && P[q] != T[i])
            q = pi[q - 1];

        if (P[q] == T[i]) q++;

        if (q == n) {
            occ.push_back(i - n + 1);
            q = pi[n - 1];
        }
    }
    return occ;
}
\end{lstlisting}

\textbf{Cost.} \passthrough{\lstinline!O(n + m)!}.

\subsection{4.4 Strong failure function for optimized KMP shifts}\label{strong-failure-function-for-optimized-kmp-shifts}

vector<int> strong_prefix_function(string P) {
    int n = P.length;
    vector<int> pi = prefix_function(P);
    vector<int> spi(n, 0);

    for (int i = 0; i < n; i++) {
        int v = pi[i];
        if (i + 1 < n && v > 0 && P[v] == P[i + 1])
            spi[i] = spi[v - 1];
        else
            spi[i] = v;

}

\subsection{4.5 Real-time string matcher transition table}\label{real-time-string-matcher-transition-table}

\begin{lstlisting}[language=C]
vector<map<char,int>> realtime_automaton(string P, set<char> Sigma) {
    int n = P.length;
    vector<map<char,int>> next(n + 1);

    for (int q = 0; q <= n; q++) {
        for (char x : Sigma) {
            string candidate = P.substr(0, q) + x;
            int best = min(n, q + 1);

            while (best > 0) {
                if (suffix(candidate, best) == P.substr(0, best))
                    break;
                best--;
            }
            next[q][x] = best;
        }
    }
    return next;
}

vector<int> realtime_match(string P, string T, set<char> Sigma) {
    auto next = realtime_automaton(P, Sigma);
    int n = P.length;
    int q = 0;
    vector<int> occ;

    for (int i = 0; i < T.length; i++) {
        q = next[q][T[i]];
        if (q == n) occ.push_back(i - n + 1);
    }
    return occ;
}
\end{lstlisting}

\subsection{5.2 Aho-Corasick failure links}\label{aho-corasick-failure-links}

\begin{lstlisting}[language=C]
void build_failure_links(TrieNode* root, set<char> Sigma) {
    queue<TrieNode*> q;

    root->fail = root;
    for (char c : Sigma) {
        if (root->child.count(c)) {
            TrieNode* u = root->child[c];
            u->fail = root;
            q.push(u);
        } else {
            root->child[c] = root;       // optional total transition
        }
    }

    while (!q.empty()) {
        TrieNode* v = q.front(); q.pop();

        for (auto [c, u] : v->child) {
            if (u == root) continue;

            TrieNode* f = v->fail;
            while (!f->child.count(c)) f = f->fail;
            u->fail = f->child[c];

            for (int id : u->fail->output)
                u->output.push_back(id);

            q.push(u);
        }
    }
}
\end{lstlisting}

\subsection{5.3 Aho-Corasick search}\label{aho-corasick-search}

\begin{lstlisting}[language=C]
vector<pair<int,int>> aho_corasick(vector<string> P, string T, set<char> Sigma) {
    TrieNode* root = build_keyword_tree(P);
    build_failure_links(root, Sigma);

    vector<pair<int,int>> hits;  // (pattern_id, start_index)
    TrieNode* state = root;

    for (int i = 0; i < T.length; i++) {
        char c = T[i];

        while (state != root && !state->child.count(c))
            state = state->fail;

        if (state->child.count(c)) state = state->child[c];
        else state = root;

        for (int id : state->output)
            hits.push_back({id, i - (int)P[id].length + 1});
    }
    return hits;
}
\end{lstlisting}

\textbf{Cost.} \passthrough{\lstinline!O(total\_pattern\_length + |T| + output\_count)!} for fixed alphabets.

    map<char, uint64_t> mask;

\subsection{6.3 Karp-Rabin fingerprint matching}\label{karp-rabin-fingerprint-matching}

\begin{lstlisting}[language=C]
vector<int> karp_rabin(string P, string T, uint64_t base, uint64_t mod) {
    int n = P.length, m = T.length;
    vector<int> occ;
    if (n > m) return occ;

    uint64_t power = 1;
    for (int i = 1; i < n; i++) power = (power * base) % mod;

    uint64_t hp = 0, ht = 0;
    for (int i = 0; i < n; i++) {
        hp = (hp * base + code(P[i])) % mod;
        ht = (ht * base + code(T[i])) % mod;
    }

    for (int s = 0; s + n <= m; s++) {
        if (hp == ht && T.substr(s, n) == P)
            occ.push_back(s);

        if (s + n < m) {
            ht = (ht + mod - (code(T[s]) * power) % mod) % mod;
            ht = (ht * base + code(T[s + n])) % mod;
        }
    }
    return occ;
}
\end{lstlisting}

\subsection{7.1 Naive suffix tree construction}\label{naive-suffix-tree-construction}

\begin{lstlisting}[language=C]
SuffixTree naive_suffix_tree(string S) {
    S += terminal_not_in(S);
    SuffixTree tree;
    tree.root = new Node();

    for (int i = 0; i < S.length; i++) {
        Node* v = tree.root;
        int j = i;

        while (j < S.length) {
            Edge* e = find_edge_starting_with(v, S[j]);

            if (e == NULL) {
                add_leaf_edge(v, j, S.length);
                break;
            }

            int k = 0;
            while (e->start + k < e->end &&
                   j + k < S.length &&
                   S[e->start + k] == S[j + k]) {
                k++;
            }

            if (e->start + k == e->end) {
                v = e->child;
                j += k;
            } else {
                Node* split = split_edge(e, k);
                add_leaf_edge(split, j + k, S.length);
                break;
            }
        }
    }
    return tree;
}
\end{lstlisting}

\textbf{Cost.} \passthrough{\lstinline!O(n\^2)!} worst case if implemented directly.

\subsection{7.2 Ukkonen-style online suffix tree construction}\label{ukkonen-style-online-suffix-tree-construction}

\begin{lstlisting}[language=C]
SuffixTree ukkonen(string S) {
    S += terminal_not_in(S);

    SuffixTree st;
    st.root = new Node();
    Node* active_node = st.root;
    int active_edge = -1;
    int active_len = 0;
    int remainder = 0;
    Node* last_internal = NULL;
    int leaf_end = -1;

    for (int pos = 0; pos < S.length; pos++) {
        leaf_end = pos;
        remainder++;
        last_internal = NULL;

        while (remainder > 0) {
            if (active_len == 0) active_edge = pos;

            Edge* e = find_edge_starting_with(active_node, S[active_edge]);

            if (e == NULL) {
                add_leaf_edge(active_node, pos, INF_LEAF_END);

                if (last_internal != NULL) {
                    last_internal->suffix_link = active_node;
                    last_internal = NULL;
                }
            } else {
                if (walk_down_if_needed(active_node, e, active_edge, active_len))
                    continue;

                char next_char = S[e->start + active_len];
                if (next_char == S[pos]) {
                    active_len++;

                    if (last_internal != NULL) {
                        last_internal->suffix_link = active_node;
                        last_internal = NULL;
                    }
                    break;
                }

                Node* split = split_edge(e, active_len);
                add_leaf_edge(split, pos, INF_LEAF_END);

                if (last_internal != NULL)
                    last_internal->suffix_link = split;
                last_internal = split;
            }

            remainder--;

            if (active_node == st.root && active_len > 0) {
                active_len--;
                active_edge = pos - remainder + 1;
            } else if (active_node != st.root) {
                active_node = active_node->suffix_link;
            }
        }
    }

    finalize_leaf_ends(st, S.length);
    return st;
}
\end{lstlisting}

\textbf{Key ideas.} Online extension, active point, suffix links, implicit-to-explicit phase conversion.\\
\textbf{Cost.} \passthrough{\lstinline!O(n)!} with constant-time edge lookup for fixed alphabets.

\section{9. Lowest Common Ancestor and Longest Common Extension}\label{lowest-common-ancestor-and-longest-common-extension}

\subsection{9.1 Euler tour + RMQ LCA preprocessing}\label{euler-tour-rmq-lca-preprocessing}

\begin{lstlisting}[language=C]
struct LCAIndex {
    vector<Node*> euler;
    vector<int> depth;
    vector<int> first;
    SparseTable<int> rmq;
};

LCAIndex preprocess_lca(Node* root) {
    LCAIndex I;

    function<void(Node*, int)> dfs = [&](Node* v, int d) {
        I.first[v->id] = I.euler.size();
        I.euler.push_back(v);
        I.depth.push_back(d);

        for (Node* u : v->children) {
            dfs(u, d + 1);
            I.euler.push_back(v);
            I.depth.push_back(d);
        }
    };

    dfs(root, 0);
    I.rmq = build_rmq_on_min_depth(I.depth);
    return I;
}

Node* lca(LCAIndex I, Node* a, Node* b) {
    int i = I.first[a->id];
    int j = I.first[b->id];
    if (i > j) swap(i, j);
    int k = I.rmq.argmin(i, j);
    return I.euler[k];
}
\end{lstlisting}

\textbf{Cost.} \passthrough{\lstinline!O(n log n)!} preprocessing with sparse table, \passthrough{\lstinline!O(1)!} query. Gusfield also presents a specialized linear-time preprocessing route to constant-time LCA.

\begin{lstlisting}[language=C]
vector<int> k_mismatch_match(string P, string T, int k) {
    string S = P + unique_terminal() + T;
    SuffixTree st = ukkonen(S);
    LCAIndex I = preprocess_lca(st.root);
    vector<int> occ;

    int n = P.length;
    int offset = n + 1;

    for (int s = 0; s + n <= T.length; s++) {
        int mismatches = 0;
        int p = 0;

        while (p < n && mismatches <= k) {
            int l = lce(st, I, p, offset + s + p);
            p += l;
            if (p < n) {
                mismatches++;
                p++;
            }

\subsection{10.1 Edit distance}\label{edit-distance}

\begin{lstlisting}[language=C]
int edit_distance(string A, string B) {
    int n = A.length, m = B.length;
    vector<vector<int>> D(n + 1, vector<int>(m + 1));

    for (int i = 0; i <= n; i++) D[i][0] = i;
    for (int j = 0; j <= m; j++) D[0][j] = j;

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            int sub = D[i - 1][j - 1] + (A[i - 1] == B[j - 1] ? 0 : 1);
            int del = D[i - 1][j] + 1;
            int ins = D[i][j - 1] + 1;
            D[i][j] = min(sub, min(del, ins));
        }
    }
    return D[n][m];
}

\subsection{10.2 Weighted edit distance}\label{weighted-edit-distance}

\begin{lstlisting}[language=C]
int weighted_edit_distance(string A, string B,
                           int (*sub_cost)(char,char),
                           int (*del_cost)(char),
                           int (*ins_cost)(char)) {
    int n = A.length, m = B.length;
    Matrix<int> D(n + 1, m + 1, INF);
    D[0][0] = 0;

    for (int i = 1; i <= n; i++)
        D[i][0] = D[i - 1][0] + del_cost(A[i - 1]);

    for (int j = 1; j <= m; j++)
        D[0][j] = D[0][j - 1] + ins_cost(B[j - 1]);

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            D[i][j] = min3(
                D[i - 1][j - 1] + sub_cost(A[i - 1], B[j - 1]),
                D[i - 1][j] + del_cost(A[i - 1]),
                D[i][j - 1] + ins_cost(B[j - 1])
            );
        }
    }
    return D[n][m];
}

Alignment local_alignment(string A, string B,
                          int (*score)(char,char),
                          int gap) {

    int best = 0, bi = 0, bj = 0;

                H[i - 1][j] - gap,

            if (H[i][j] > best) {
                best = H[i][j]; bi = i; bj = j;
            }
        }
    }

    return traceback_until_zero(A, B, H, bi, bj);
}

int affine_gap_alignment(string A, string B,
                         int (*sub)(char,char),
                         int open, int extend) {
    int n = A.length, m = B.length;
    Matrix<int> M(n + 1, m + 1, -INF);
    Matrix<int> Ix(n + 1, m + 1, -INF);   // gap in B
    Matrix<int> Iy(n + 1, m + 1, -INF);   // gap in A

    M[0][0] = 0;

    for (int i = 1; i <= n; i++)
        Ix[i][0] = -open - (i - 1) * extend;

    for (int j = 1; j <= m; j++)
        Iy[0][j] = -open - (j - 1) * extend;

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            Ix[i][j] = max(M[i - 1][j] - open, Ix[i - 1][j] - extend);
            Iy[i][j] = max(M[i][j - 1] - open, Iy[i][j - 1] - extend);

            M[i][j] = max3(M[i - 1][j - 1],
                           Ix[i - 1][j - 1],
                           Iy[i - 1][j - 1]) + sub(A[i - 1], B[j - 1]);
        }
    }

    return max3(M[n][m], Ix[n][m], Iy[n][m]);
}

Alignment hirschberg(string A, string B) {
    if (A.length == 0) return align_all_gaps_in_A(B);
    if (B.length == 0) return align_all_gaps_in_B(A);
    if (A.length == 1 || B.length == 1) return small_alignment(A, B);

    Alignment right = hirschberg(A.substr(mid), B.substr(split));
    return concatenate(left, right);
}

\subsection{10.7 Banded edit distance for bounded differences}\label{banded-edit-distance-for-bounded-differences}

\begin{lstlisting}[language=C]
int banded_edit_distance(string A, string B, int k) {
    int n = A.length, m = B.length;
    Matrix<int> D(n + 1, 2 * k + 1, INF);

    auto col = [&](int i, int j) { return j - i + k; };

    D[0][col(0,0)] = 0;

    for (int i = 0; i <= n; i++) {
        int lo = max(0, i - k);
        int hi = min(m, i + k);

        for (int j = lo; j <= hi; j++) {
            int c = col(i, j);
            if (i > 0 && in_band(i - 1, j, k))
                D[i][c] = min(D[i][c], D[i - 1][col(i - 1, j)] + 1);
            if (j > 0 && in_band(i, j - 1, k))
                D[i][c] = min(D[i][c], D[i][col(i, j - 1)] + 1);
            if (i > 0 && j > 0 && in_band(i - 1, j - 1, k))
                D[i][c] = min(D[i][c],
                              D[i - 1][col(i - 1, j - 1)] +
                              (A[i - 1] == B[j - 1] ? 0 : 1));
        }
    }

    return in_band(n, m, k) ? D[n][col(n,m)] : INF;
}
\end{lstlisting}

bool edit_distance_at_most_k(string A, string B, int k) {
    int n = A.length, m = B.length;
    if (abs(n - m) > k) return false;

    vector<int> prev(2 * k + 1, INF), cur(2 * k + 1, INF);

    prev[k] = 0;
    for (int i = 0; i <= n; i++) {
        fill(cur.begin(), cur.end(), INF);
        int lo = max(0, i - k);
        int hi = min(m, i + k);

        for (int j = lo; j <= hi; j++) {
            update_banded_cell(A, B, i, j, k, prev, cur);
        }

        if (min(cur) > k) return false;
        swap(prev, cur);
    }

    return prev[m - n + k] <= k;
}

string greedy_assembly(vector<string> reads) {

\subsection{15.5 Shortest superstring exact DP}\label{shortest-superstring-exact-dp}

\begin{lstlisting}[language=C]
string shortest_superstring_dp(vector<string> R) {
    remove_substrings(R);
    int n = R.size();

    Matrix<int> ov(n, n);
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            if (i != j) ov[i][j] = longest_suffix_prefix(R[i], R[j]);

    Matrix<int> dp(1 << n, n, INF);
    Matrix<int> parent(1 << n, n, -1);

    for (int i = 0; i < n; i++)
        dp[1 << i][i] = R[i].length;

    for (int mask = 1; mask < (1 << n); mask++) {
        for (int j = 0; j < n; j++) if (mask & (1 << j)) {
            for (int k = 0; k < n; k++) if (!(mask & (1 << k))) {
                int next = mask | (1 << k);
                int cost = dp[mask][j] + R[k].length - ov[j][k];
                if (cost < dp[next][k]) {
                    dp[next][k] = cost;
                    parent[next][k] = j;
                }
            }
        }
    }

    return reconstruct_superstring(R, ov, dp, parent);
}
\end{lstlisting}

Tree upgma(Matrix<double> D) {
    vector<Cluster> C = singleton_clusters(D.size);

    while (C.size() > 1) {
        pair<int,int> ab = closest_pair(D, C);
        Cluster u = merge(C[ab.first], C[ab.second]);

        for (Cluster x : C) {
            if (x == C[ab.first] || x == C[ab.second]) continue;
            D[u][x] = (C[ab.first].size * D[C[ab.first]][x] +
                       C[ab.second].size * D[C[ab.second]][x]) /
                      (C[ab.first].size + C[ab.second].size);
            D[x][u] = D[u][x];
        }

        replace_pair_with_cluster(C, ab, u);
    }

    return C[0].tree;
}

\subsection{16.3 Small parsimony by Sankoff DP}\label{small-parsimony-by-sankoff-dp}

\begin{lstlisting}[language=C]
int small_parsimony(Tree T, vector<string> leaf_labels, Alphabet Sigma) {
    int m = leaf_labels[0].length;
    int total = 0;

    for (int pos = 0; pos < m; pos++) {
        postorder(T.root, [&](Node* v) {
            for (char a : Sigma) {
                if (v->is_leaf) {
                    v->score[a] = (leaf_labels[v->id][pos] == a) ? 0 : INF;
                } else {
                    int sum = 0;
                    for (Node* child : v->children) {
                        int best = INF;
                        for (char b : Sigma)
                            best = min(best, child->score[b] + mutation_cost(a, b));
                        sum += best;
                    }
                    v->score[a] = sum;
                }
            }
        });

        total += min_a(T.root->score[a]);
    }

    return total;
}
\end{lstlisting}

\subsection{16.4 Fitch parsimony for unordered characters}\label{fitch-parsimony-for-unordered-characters}

\begin{lstlisting}[language=C]
int fitch_score(Tree T, vector<string> leaf_labels, Alphabet Sigma) {
    int score = 0;
    int m = leaf_labels[0].length;

    for (int pos = 0; pos < m; pos++) {
        postorder(T.root, [&](Node* v) {
            if (v->is_leaf) {
                v->set = { leaf_labels[v->id][pos] };
            } else {
                set<char> inter = intersection(v->left->set, v->right->set);
                if (!inter.empty()) {
                    v->set = inter;
                } else {
                    v->set = union_set(v->left->set, v->right->set);
                    score++;
                }
            }
        });
    }

    return score;
}

\subsection{16.5 Perfect phylogeny test schema}\label{perfect-phylogeny-test-schema}

\begin{lstlisting}[language=C]
bool perfect_phylogeny_binary(Matrix<int> M) {
    // Columns are characters; rows are taxa.
    for (int a = 0; a < M.cols; a++) {
        for (int b = a + 1; b < M.cols; b++) {
            bool seen01 = false, seen10 = false, seen11 = false;

            for (int r = 0; r < M.rows; r++) {
                if (M[r][a] == 0 && M[r][b] == 1) seen01 = true;
                if (M[r][a] == 1 && M[r][b] == 0) seen10 = true;
                if (M[r][a] == 1 && M[r][b] == 1) seen11 = true;
            }

            if (seen01 && seen10 && seen11)
                return false;
        }
    }
    return true;
}

\textbf{Principle.} Binary perfect phylogeny requires no conflicting pair of characters.

int frameshift_alignment(string dna, string protein) {
    int n = dna.length, m = protein.length;
    Matrix<int> D(n + 1, m + 1, INF);
    D[0][0] = 0;

    for (int i = 0; i <= n; i++) {
        for (int j = 0; j <= m; j++) {
            if (D[i][j] == INF) continue;

            // Normal codon consumption.
            if (i + 3 <= n && j + 1 <= m) {
                char aa = translate(dna.substr(i, 3));
                D[i + 3][j + 1] = min(D[i + 3][j + 1],
                                      D[i][j] + aa_cost(aa, protein[j]));
            }

            // Frameshift insert/delete in DNA reading frame.
            if (i + 1 <= n)
                D[i + 1][j] = min(D[i + 1][j], D[i][j] + frameshift_penalty);

            if (j + 1 <= m)
                D[i][j + 1] = min(D[i][j + 1], D[i][j] + protein_gap_penalty);
        }
    }

    return min_over_i(D[i][m]);
}

Path gene_prediction(string dna, Model model) {
    int n = dna.length;
    vector<State> states = model.states;
    Matrix<double> V(n + 1, states.size(), -INF);
    Matrix<State> back(n + 1, states.size());

    V[0][model.start] = 0;

    for (int i = 0; i < n; i++) {
        for (State s : states) {
            if (V[i][s] == -INF) continue;

            for (Transition e : model.outgoing[s]) {
                int len = e.emitted_length;
                if (i + len <= n) {
                    double score = V[i][s] + e.weight +
                                   emission_score(e.to, dna.substr(i, len));
                    if (score > V[i + len][e.to]) {
                        V[i + len][e.to] = score;
                        back[i + len][e.to] = s;
                    }
                }
            }
        }
    }

    return traceback_best_state_path(V, back);
}
\end{lstlisting}

\subsection{18.1 Reversal distance search schema}\label{reversal-distance-search-schema}

\begin{lstlisting}[language=C]
int reversal_distance(vector<int> A, vector<int> B) {
    queue<vector<int>> q;
    map<vector<int>, int> dist;

    q.push(A);
    dist[A] = 0;

    while (!q.empty()) {
        vector<int> x = q.front(); q.pop();
        if (x == B) return dist[x];

        for (int i = 0; i < x.size(); i++) {
            for (int j = i + 1; j < x.size(); j++) {
                vector<int> y = reverse_segment(x, i, j);
                if (!dist.count(y)) {
                    dist[y] = dist[x] + 1;
                    q.push(y);
                }
            }
        }
    }
    return INF;
}
\end{lstlisting}

\textbf{Cost.} Exponential; useful as the exact conceptual baseline.

int edit_distance_two_rows(string A, string B) {
    vector<int> prev(B.length + 1), cur(B.length + 1);

    for (int j = 0; j <= B.length; j++) prev[j] = j;

    for (int i = 1; i <= A.length; i++) {
        cur[0] = i;
        for (int j = 1; j <= B.length; j++) {
            cur[j] = min3(prev[j - 1] + (A[i - 1] != B[j - 1]),
                          prev[j] + 1,
                          cur[j - 1] + 1);
        }
        swap(prev, cur);
    }

    return prev[B.length];
}

\section{22. Caveats}\label{caveats}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  The uploaded book is not a source-code manual. It says it avoids detailed code and emphasizes ideas, proofs, and derivations. This document therefore reconstructs algorithms into C-like pseudocode from the book's algorithmic exposition.
\item
  Several advanced methods in the book have multiple known implementations. Where the book presents an algorithmic idea rather than low-level implementation code, this document records the implementation-level schema.
\item
  Some algorithms, especially Ukkonen suffix-tree construction, constant-time LCA, Four-Russians alignment, parametric alignment, and genome rearrangement methods, require careful engineering beyond the compact pseudocode shown here.
\item
  For production-grade string indexes, suffix arrays/FM-indexes may be preferable to full suffix trees because of memory locality and lower constants, even though suffix trees are central to the book's exposition.
\end{enumerate}

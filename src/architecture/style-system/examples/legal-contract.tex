  A short reference contract demonstrating the legal family module:
  parties block, recitals, definitions, articles with nested clauses,
  defined-term typography, and a two-party signature panel.%
}

\begin{document}
\makedoctocpage

\Parties{Acme, LLC, a Delaware limited liability company (``\textbf{Provider}'')}%
        {Beta Co., a Georgia corporation (``\textbf{Customer}'')}

\begin{recitals}
  \Recital{Provider operates a software-as-a-service platform (the
           \DefinedTerm{Service}) for managed document processing.}
  \Recital{Customer wishes to access and use the \DefinedTerm{Service} on
           the terms set forth in this Agreement.}
  \Recital{The parties wish to set out their respective rights and
           obligations in writing.}
\end{recitals}

\begin{definitions}
  \DefnTerm{Service} the SaaS platform identified in Schedule A.
  \DefnTerm{Effective Date} the date on which both parties have signed
  this Agreement.
  \DefnTerm{Term} the period defined in Article~\ref{art:term}.
  \DefnTerm{Confidential Information} non-public information designated
  as confidential or that a reasonable person would understand to be
  confidential.
\end{definitions}

\Article{Service\label{art:service}}
\Clause{Provision. The \DefinedTerm{Provider} shall make the
        \DefinedTerm{Service} available to the \DefinedTerm{Customer}
        subject to the terms and conditions of this Agreement.}
\Clause{Service levels. The \DefinedTerm{Provider} shall use commercially
        reasonable efforts to maintain the \DefinedTerm{Service}
        availability of at least 99.5\% measured monthly, excluding
        scheduled maintenance windows.}
\Clause{Support. The \DefinedTerm{Provider} shall provide standard
        business-hours support; premium support is available under a
        separate Order Form.}

\Article{Term and termination\label{art:term}}
\Clause{Initial term. This Agreement begins on the
        \DefinedTerm{Effective Date} and continues for one (1) year
        (the ``Initial Term'').}
\Clause{Renewal. Following the Initial Term, this Agreement shall renew
        automatically for successive one-year terms unless either party
        gives written notice of non-renewal at least thirty (30) days
        prior to the end of the then-current term.}
\Clause{Termination for cause. Either party may terminate this Agreement
        for material breach upon thirty (30) days' written notice if the
        breach is not cured within that period.}

\Article{Confidentiality}
\Clause{Each party shall protect the \DefinedTerm{Confidential Information}
        of the other with at least the same degree of care it uses to
        protect its own confidential information of like nature, but in
        no event less than reasonable care.}
\Clause{The obligations in this Article survive termination of this
        Agreement for a period of three (3) years.}

\begin{clausebox}[Limitation of liability]
  Except for breaches of confidentiality, infringement of intellectual
  property rights, and a party's gross negligence or wilful misconduct,
  in no event shall either party's aggregate liability under this
  Agreement exceed the fees paid or payable by the
  \DefinedTerm{Customer} during the twelve (12) months preceding the
  event giving rise to the claim.
\end{clausebox}

\makesignaturepanel{Provider -- Acme, LLC}{Customer -- Beta Co.}

\end{document}

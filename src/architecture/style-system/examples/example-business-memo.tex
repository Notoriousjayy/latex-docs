\section*{Appendix A: Procedure for migrating an existing document}

\begin{procedure}{Migrate a document to a family module}
  \PStep{Identify the appropriate concrete module by document function
         (legal / financial / business-admin / hr / personal-official, or
         a technical subtype).}%
        {The chosen module's display label matches the document's
         purpose without strain.}
  \PStep{In the document preamble, replace
         \texttt{\textbackslash usepackage\{latex-docs-style\}} with
         \texttt{\textbackslash usepackage\{latex-docs-<chosen>\}}.}%
        {The compiled PDF is byte-similar to the previous output;
         differences are limited to the new \texttt{DocClass} field
         appearing in the title meta line and right-side header.}
  \PStep{Where the document's body re-implements an environment or macro
         the new module already provides, delete the local definitions
         and use the module's version.}%
        {Document compiles with zero new warnings; visual diff is limited
         to deduplicated structures.}
  \PStep{Open a PR; CI verifies the migration on every commit.}%
        {The smoke check is green; the rendered PDF is reviewed by the
         document's owner.}
\end{procedure}

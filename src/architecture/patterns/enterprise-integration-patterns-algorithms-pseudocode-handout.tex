\textbf{Source analyzed:} Gregor Hohpe and Bobby Woolf, \emph{Enterprise
Integration Patterns: Designing, Building, and Deploying Messaging
Solutions}.

\textbf{Extraction scope.} The book is a pattern language rather than an
algorithms textbook. Its executable material appears as Java/JMS,
C\#/.NET/MSMQ, Web Services, and integration-tool examples. This handout
treats each recurring message-flow mechanism and each code-backed
pattern as an algorithmic unit. The pseudocode is intentionally
\textbf{C-like and implementation-neutral}: it captures control flow,
state, invariants, and failure handling without cloning the book's
copyrighted example source.

\textbf{Core abstraction.} Most algorithms in the book can be
represented as repeated transformations over the same primitive:

\begin{lstlisting}[language=C]
typedef struct Message {
    HeaderMap headers;
    Body body;
    Time createdAt;
    Time expiresAt;
    bool persistent;
} Message;

typedef struct Channel {
    char *name;
    Queue<Message> durableQueue;
    SubscriberList subscribers;
    ChannelKind kind;      // POINT_TO_POINT, PUBLISH_SUBSCRIBE, CONTROL, INVALID, DEAD_LETTER
} Channel;

typedef struct Endpoint {
    Channel *input;
    Channel **outputs;
    Handler handler;
} Endpoint;
\end{lstlisting}

The general message-processing loop is:

\begin{lstlisting}[language=C]
void run_endpoint(Endpoint *ep) {
    while (system_is_running()) {
        Message msg = receive(ep->input);
        ProcessingResult result = ep->handler(msg);
        acknowledge_or_recover(ep->input, msg, result);
    }
}
\end{lstlisting}

\section{2. Integration Style

\subsection{2.1 File Transfer}\label{file-transfer}

\begin{lstlisting}[language=C]
void file_transfer_producer(App *source, Directory *dropbox, Schedule schedule) {
    while (true) {
        wait_until(schedule.next_time());
        DataSet data = source->export_changed_data();
        File tmp = write_temp_file(data);
        File final = rename_atomically(tmp, make_unique_name(schedule.now()));
        move_to_directory(final, dropbox);
    }
}

void file_transfer_consumer(App *target, Directory *dropbox) {
    while (true) {
        File f = wait_for_complete_file(dropbox);
        lock_file(f);
        DataSet data = parse_file(f);
        target->import_data(transform_if_needed(data));
        mark_or_delete_file(f);
        unlock_file(f);
    }
}
\end{lstlisting}

\textbf{Key invariant:} no consumer reads a file while the producer is
still writing it.

\subsection{2.2 Shared Database}\label{shared-database}

\begin{lstlisting}[language=C]
void shared_database_update(Transaction *tx, Table *table, Row row) {
    begin(tx);
    Row current = select_for_update(table, row.primaryKey);
    validate_semantics(current, row);
    upsert(table, row);
    commit(tx);
}
\end{lstlisting}

\textbf{Key invariant:} all participating applications share a database
schema contract; transaction isolation protects concurrent updates.

\subsection{2.3 Remote Procedure
Invocation}\label{remote-procedure-invocation}

\begin{lstlisting}[language=C]
Response invoke_remote(ServiceRef svc, Request req) {
    SerializedRequest bytes = marshal(req);
    NetworkReply reply = send_synchronous(svc.address, bytes);
    if (reply.failed) {
        throw RemoteCallFailed;
    }
    return unmarshal(reply.bytes);
}
\end{lstlisting}

\textbf{Key hazard:} a remote call looks syntactically like a local call
but has different latency, failure, and coupling properties.

\subsection{2.4 Messaging}\label{messaging}

\begin{lstlisting}[language=C]
void send_message(Channel *ch, Body body, HeaderMap headers) {
    Message msg = make_message(body, headers);
    put(ch, msg);                 // store locally or durably
    return;                       // sender continues immediately
}

void messaging_system_deliver(Channel *ch) {
    while (true) {
        Message msg = next_undelivered(ch);
        if (network_available(msg.destination)) {
            forward(msg);
            mark_delivered(msg);
        } else {
            backoff_and_retry(msg);
        }
    }
}
\end{lstlisting}

\subsection{3.1 Message Transmission: Create, Send, Deliver, Receive,
Process}\label{message-transmission-create-send-deliver-receive-process}

\begin{lstlisting}[language=C]
void transmit(Sender *sender, Channel *ch, Receiver *receiver, Body payload) {
    Message msg;
    msg.headers = default_headers(sender, ch, receiver);
    msg.body = payload;
    msg.createdAt = now();

    send_to_channel(ch, msg);          // sender-side store
    deliver_to_receiver_node(ch, msg); // store-and-forward hop(s)
    Message received = receive(ch);    // receiver-side dequeue/read
    receiver->process(received.body);
}
\end{lstlisting}

\subsection{3.2 Message Channel}\label{message-channel}

\begin{lstlisting}[language=C]
void send_to_channel(Channel *ch, Message msg) {
    validate_channel_exists(ch);
    enqueue(ch->durableQueue, msg);
}

Message receive_from_channel(Channel *ch, Selector selector) {
    while (true) {
        Message msg = peek(ch->durableQueue);
        if (selector_matches(selector, msg)) {
            return dequeue(ch->durableQueue);
        }
        wait_for_more_messages(ch);
    }
}
\end{lstlisting}

\subsection{3.3 Message Envelope}\label{message-envelope}

\begin{lstlisting}[language=C]
Message make_message(Body body, HeaderMap headers) {
    Message msg;
    msg.headers = headers;
    msg.body = marshal_body(body);
    msg.createdAt = now();
    msg.expiresAt = headers.get_time_or_infinite("ExpiresAt");
    msg.persistent = headers.get_bool("Persistent", false);
    return msg;
}
\end{lstlisting}

\subsection{3.4 Pipes and Filters}\label{pipes-and-filters}

\begin{lstlisting}[language=C]
void filter_loop(Channel *input, Channel *output, FilterFn fn) {
    while (true) {
        Message in = receive(input);
        Message out = fn(in);
        send(output, out);
        ack(input, in);
    }
}
\end{lstlisting}

\subsection{3.5 Event-Driven Filter

\begin{lstlisting}[language=C]
void on_message(Event e) {
    Message in = end_receive(e);
    Message out = process_message(in);
    send(output_channel, out);
    begin_receive(input_channel, on_message);
}

void start_event_driven_filter(Channel *input) {

}
\end{lstlisting}

\subsection{3.6 Transactional Filter
Loop}\label{transactional-filter-loop}

\begin{lstlisting}[language=C]
void transactional_filter(Channel *input, Channel *output, FilterFn fn) {
    while (true) {
        begin_transaction();
        Message in = receive(input);
        Message out = fn(in);
        send(output, out);
        ack(input, in);
        commit_transaction();
    on_error:
        rollback_transaction(); // input message becomes visible again or moves to recovery
    }
}
\end{lstlisting}

void validate_or_reject(Channel *target, Message msg, Validator v) {

\subsection{4.5 Dead Letter Channel}\label{dead-letter-channel}

\begin{lstlisting}[language=C]
void deliver_with_dead_letter(Channel *target, Message msg, int maxAttempts) {
    for (int attempt = 1; attempt <= maxAttempts; attempt++) {
        if (try_deliver(target, msg)) {
            return;
        }
        sleep(backoff(attempt));
    }
    msg.headers["DeadLetterReason"] = "Delivery attempts exhausted";
    send(dead_letter_channel(), msg);
}
\end{lstlisting}

\subsection{4.6 Guaranteed Delivery}\label{guaranteed-delivery}

void guaranteed_send(Channel *ch, Message msg) {

void guaranteed_forwarder(Channel *ch) {
    while (true) {
        Message msg = next_persisted_unacknowledged(ch);

Message request_reply(Requester *r, Channel *requestCh, Channel *replyCh, Body requestBody) {
    Message req = make_message(requestBody, empty_headers());
    req.headers["ReplyTo"] = replyCh->name;
    req.headers["MessageId"] = new_uuid();
    req.headers["CorrelationId"] = req.headers["MessageId"];

    send(requestCh, req);

    while (true) {
        Message reply = receive(replyCh);

            return reply;

        stash_unmatched_reply(reply);

\subsection{5.6 Correlation Identifier}\label{correlation-identifier}

\begin{lstlisting}[language=C]
void remember_request(CorrelationStore *store, Message req, Continuation cont) {
    store->byCorrelationId[req.headers["CorrelationId"]] = cont;
}

void dispatch_reply(CorrelationStore *store, Message reply) {
    char *id = reply.headers["CorrelationId"];
    Continuation cont = store->byCorrelationId[id];
    if (cont == NULL) {
        send(invalid_message_channel(), reply);
        return;
    }
    cont(reply);
    remove(store->byCorrelationId, id);
}
\end{lstlisting}

\subsection{5.7 Message Sequence}\label{message-sequence}

\begin{lstlisting}[language=C]
void send_sequence(Channel *ch, Body *parts, int n) {
    char *seqId = new_uuid();
    for (int i = 0; i < n; i++) {
        Message msg = make_message(parts[i], empty_headers());
        msg.headers["SequenceId"] = seqId;
        msg.headers["SequenceNumber"] = i + 1;
        msg.headers["SequenceSize"] = n;
        msg.headers["EndOfSequence"] = (i == n - 1);
        send(ch, msg);
    }
}
\end{lstlisting}

bool is_expired(Message msg) {

void receive_skip_expired(Channel *ch) {

\section{6. Message Routing
Algorithms}\label{message-routing-algorithms}

\subsection{6.1 Simple Router}\label{simple-router}

\begin{lstlisting}[language=C]
void simple_router(Channel *in, Channel *out1, Channel *out2) {
    bool toggle = false;
    while (true) {
        Message msg = receive(in);
        toggle = !toggle;
        if (toggle) send(out1, msg);
        else        send(out2, msg);
        ack(in, msg);
    }
}

\subsection{6.4 Dynamic Router}\label{dynamic-router}

\begin{lstlisting}[language=C]
typedef struct DynamicRule {
    Predicate predicate;
    Channel *destination;
    int priority;
} DynamicRule;

void handle_control_message(RuleBase *rules, Message control) {
    if (control.body.action == SUBSCRIBE) {
        DynamicRule rule = parse_rule(control.body);
        if (!conflicts_with_existing_rule(rules, rule)) {
            add_rule(rules, rule);
        } else {
            resolve_rule_conflict(rules, rule);
        }
    }
    if (control.body.action == UNSUBSCRIBE) {
        remove_rule_for_destination(rules, control.body.destination);
    }
}

void dynamic_router(Channel *in, Channel *control, RuleBase *rules) {
    while (true) {
        if (has_message(control)) {
            handle_control_message(rules, receive(control));
        }
        if (has_message(in)) {
            Message msg = receive(in);
            List<DynamicRule> matches = matching_rules(rules, msg);
            route_by_conflict_policy(msg, matches);
            ack(in, msg);
        }
    }
}

Conflict policies:

\begin{lstlisting}[language=C]
void route_by_conflict_policy(Message msg, List<DynamicRule> matches) {
    if (matches.count == 0) {
        send(invalid_message_channel(), msg);
    } else if (policy == FIRST_MATCH) {
        send(matches[0].destination, msg);
    } else if (policy == HIGHEST_PRIORITY) {
        send(max_priority(matches).destination, msg);
    } else if (policy == ALL_MATCHES) {
        for_each(matches, r) send(r.destination, clone_message(msg));
    }
}

\subsection{6.7 Aggregator}\label{aggregator}

The aggregator is one of the most algorithmic patterns in the book. It
has three explicit decision points: correlation, completeness, and
aggregation.

\begin{lstlisting}[language=C]
typedef struct AggregateState {
    char *correlationId;
    Map<int, Message> messages;
    Time firstSeen;
    bool closed;
} AggregateState;

void aggregator(Channel *in, Channel *out, AggregationSpec spec) {
    Map<char*, AggregateState> groups;

    while (true) {
        Message msg = receive(in);
        char *key = spec.correlation_key(msg);
        AggregateState *state = get_or_create_group(&groups, key);
        state->messages[msg.headers["SequenceNumber"]] = msg;

        if (spec.is_complete(state) || spec.timed_out(state)) {
            Message result = spec.aggregate(state);
            send(out, result);
            state->closed = true;
            remove_group(&groups, key);
        }
        ack(in, msg);
    }
}
\end{lstlisting}

Common completeness strategies:

\begin{lstlisting}[language=C]
bool complete_by_sequence_size(AggregateState *s) {
    int expected = any_message(s)->headers["SequenceSize"];
    return s->messages.count == expected;
}

bool complete_by_timeout(AggregateState *s) {
    return now() - s->firstSeen > MAX_WAIT;
}

bool complete_by_quorum(AggregateState *s) {
    return s->messages.count >= REQUIRED_REPLY_COUNT;
}
\end{lstlisting}

Common aggregation strategies:

\begin{lstlisting}[language=C]
Message aggregate_best_bid(AggregateState *s) {
    Message best = first_message(s);
    for_each_message(s, msg) {
        if (msg.body.price < best.body.price) best = msg;
    }
    return best;
}

Message aggregate_concatenate(AggregateState *s) {
    Body result = empty_body();
    for (int i = 1; i <= s->messages.count; i++) {
        append(result, s->messages[i].body);
    }
    return make_message(result, headers_from_group(s));
}
\end{lstlisting}

\subsection{6.8 Resequencer}\label{resequencer}

\begin{lstlisting}[language=C]
typedef struct SequenceBuffer {
    int nextExpected;
    Map<int, Message> buffered;
    Time lastProgress;
} SequenceBuffer;

void resequencer(Channel *in, Channel *out) {
    Map<char*, SequenceBuffer> buffers;

    while (true) {
        Message msg = receive(in);
        char *seqId = msg.headers["SequenceId"];
        int n = msg.headers["SequenceNumber"];
        SequenceBuffer *buf = get_or_create_buffer(&buffers, seqId);
        buf->buffered[n] = msg;

        while (exists(buf->buffered, buf->nextExpected)) {
            Message ready = remove(buf->buffered, buf->nextExpected);
            send(out, ready);
            buf->nextExpected++;
            buf->lastProgress = now();
        }

        if (buffer_timed_out(buf)) {
            handle_gap(buf); // release partial, dead-letter gap, or request resend
        }
        ack(in, msg);
    }
}
\end{lstlisting}

\subsection{6.9 Composed Message
Processor}\label{composed-message-processor}

\begin{lstlisting}[language=C]
void composed_message_processor(Channel *in, Channel *out, Pipeline p) {
    // Conceptual composition: Splitter -> Router/Processors -> Aggregator
    Message whole = receive(in);
    BodyList parts = p.split(whole.body);
    char *seqId = new_uuid();

    for (int i = 0; i < parts.count; i++) {
        Message part = make_part(whole, parts[i], seqId, i + 1, parts.count);
        Channel *route = p.route(part);
        send(route, part);
    }

    AggregateState state = wait_for_replies(seqId, parts.count, p.timeout);
    Message result = p.aggregate(&state);
    send(out, result);
    ack(in, whole);
}
\end{lstlisting}

void loan_broker(Channel *loanRequests, BankDirectory *banks, Channel *bankReplies, Channel *loanReplies) {
    while (true) {
        Message customerReq = receive(loanRequests);
        char *corr = new_uuid();
        List<Bank> candidates = select_candidate_banks(banks, customerReq.body);

        for_each(candidates, bank) {
            Message bankReq = map_customer_request_to_bank_request(customerReq, bank);
            bankReq.headers["CorrelationId"] = corr;
            bankReq.headers["ReplyTo"] = bankReplies->name;
            send(bank.requestChannel, bankReq);
        }

        AggregateState offers = collect_until(bankReplies, corr,
                                             complete_by_count_or_timeout(candidates.count, BANK_TIMEOUT));
        Message best = choose_lowest_rate_offer(offers);
        best.headers["CorrelationId"] = customerReq.headers["CorrelationId"];
        send(loanReplies, best);
        ack(loanRequests, customerReq);
    }
}
\end{lstlisting}

\begin{lstlisting}[language=C]
void validate_order(Channel *newOrder, Channel *creditReq, Channel *inventoryReq,
                    Channel *validationReplies, Channel *validOrder, Channel *invalidOrder) {
    Message order = receive(newOrder);
    char *corr = new_uuid();

    Message credit = make_credit_request(order, corr, validationReplies);
    Message inventory = make_inventory_request(order, corr, validationReplies);
    send(creditReq, credit);
    send(inventoryReq, inventory);

    AggregateState replies = collect_until(validationReplies, corr, complete_by_count(2));
    if (credit_ok(replies) && inventory_ok(replies)) {
        send(validOrder, order);
    } else {
        send(invalidOrder, annotate_rejection(order, replies));
    }
    ack(newOrder, order);
}
\end{lstlisting}

\begin{lstlisting}[language=C]
void check_inventory_for_order(Channel *orders, Channel *itemStatus, Channel *orderStatus) {
    Message order = receive(orders);
    char *seqId = new_uuid();

    for (int i = 0; i < order.body.items.count; i++) {
        Message itemMsg = make_order_item_message(order, order.body.items[i], seqId, i + 1, order.body.items.count);
        Channel *inventory = item_inventory_channel(itemMsg.body.itemNumber);
        send(inventory, itemMsg);
    }

    AggregateState status = collect_until(itemStatus, seqId, complete_by_sequence_size);
    Message wholeStatus = aggregate_concatenate(&status);
    send(orderStatus, wholeStatus);
    ack(orders, order);
}

\section{13. Operational Invariants}\label{operational-invariants}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  \textbf{At-most-one consumption} for point-to-point channels: once a
  consumer commits the receive, no other consumer observes that message.
\item
  \textbf{At-least-once delivery} for guaranteed delivery: duplicates
  may appear after retry unless the receiver is idempotent.
\item
  \textbf{Correlation integrity} for request-reply, aggregator,
  scatter-gather, and process manager: every reply or submessage must
  carry a stable key linking it to the original work item.
\item
  \textbf{Completeness policy required} for aggregators: the system must
  know when to release a combined result by count, sequence end marker,
  timeout, quorum, or application-specific predicate.
\item
  \textbf{Ordering is not free} in asynchronous systems: if a route can
  reorder messages, use a resequencer or design the receiver to tolerate
  reordering.
\item
  \textbf{Transformation boundary matters}: translating at application
  edges preserves a canonical bus; translating in many interior
  components increases coupling.
\item
  \textbf{Transaction boundary matters}: a nontransactional
  receive-process-send loop can lose messages after receive and before
  send.
\item
  \textbf{Management traffic should be separated}: control bus, test
  messages, wire taps, and history should not corrupt business message
  semantics.
\end{enumerate}

\section{15. Implementation Notes for C-Like
Systems}\label{implementation-notes-for-c-like-systems}

A C or C++ implementation should separate the following interfaces:

\begin{lstlisting}[language=C]
typedef struct IMessageStore IMessageStore;
typedef struct IChannel IChannel;
typedef struct ISerializer ISerializer;
typedef struct IRouter IRouter;
typedef struct ITransformer ITransformer;
typedef struct ICorrelationStore ICorrelationStore;
\end{lstlisting}

Recommended responsibilities:

\begin{lstlisting}[language=C]
struct IChannelVTable {
    bool (*send)(IChannel*, Message*);
    bool (*receive)(IChannel*, Selector*, Message* out);
    bool (*ack)(IChannel*, MessageId);
    bool (*nack)(IChannel*, MessageId, NackReason);
};

struct ITransformerVTable {
    bool (*transform)(ITransformer*, Message* in, Message* out);
};

struct IRouterVTable {
    int (*route)(IRouter*, Message* in, ChannelRef* outRefs, int maxRefs);
};
\end{lstlisting}

This gives the same architectural property emphasized by the pattern
language: applications depend on stable messaging abstractions, while
adapters, translators, routers, and stores can vary independently.

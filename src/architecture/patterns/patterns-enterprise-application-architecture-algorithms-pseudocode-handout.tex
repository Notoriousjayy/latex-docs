\section{1. Domain Logic Patterns}\label{domain-logic-patterns}

\subsection{1.1 Transaction Script}\label{transaction-script}

A transaction script is one procedure per business action. It obtains
data, applies validation/calculation rules, writes results, and returns
response data.

\begin{lstlisting}[language=C]
void calculate_revenue_recognitions(Id contract_id) {
    Transaction *tx = begin_transaction();
    try {
        Row contract = contract_gateway_find(contract_id);
        Row product  = product_gateway_find(row_id(contract, "product_id"));
        Money revenue = row_money(contract, "revenue");
        Date signed_on = row_date(contract, "signed_on");

        if (row_equals(product, "type", "word_processor")) {
            recognition_gateway_insert(contract_id, revenue, signed_on);
        }
        else if (row_equals(product, "type", "spreadsheet")) {
            Money part = money_divide(revenue, 3);
            recognition_gateway_insert(contract_id, part, signed_on);
            recognition_gateway_insert(contract_id, part, date_add_days(signed_on, 60));
            recognition_gateway_insert(contract_id,
                money_subtract(revenue, money_times(part, 2)),
                date_add_days(signed_on, 90));
        }
        else if (row_equals(product, "type", "database")) {
            Money part = money_divide(revenue, 3);
            recognition_gateway_insert(contract_id, part, signed_on);
            recognition_gateway_insert(contract_id, part, date_add_days(signed_on, 30));
            recognition_gateway_insert(contract_id,
                money_subtract(revenue, money_times(part, 2)),
                date_add_days(signed_on, 60));
        }
        else {
            throw_error("unknown product type");
        }
        commit_transaction(tx);
    }
    catch_any {
        rollback_transaction(tx);
        throw;
    }
}
\end{lstlisting}

\begin{lstlisting}[language=C]
Money recognized_revenue(Id contract_id, Date as_of) {
    ResultSet *rows = recognition_gateway_find_for_contract(contract_id);
    Money total = money_zero();

    while (resultset_next(rows)) {
        if (date_less_equal(row_date(current_row(rows), "date"), as_of)) {
            total = money_add(total, row_money(current_row(rows), "amount"));
        }
    }
    return total;
}
\end{lstlisting}

\subsection{1.2 Domain Model}\label{domain-model}

A domain model distributes behavior across objects instead of
concentrating it in a single transaction procedure.

\begin{lstlisting}[language=C]
typedef struct RecognitionStrategy RecognitionStrategy;
struct RecognitionStrategy {
    void (*calculate)(RecognitionStrategy *self, struct Contract *contract);
};

typedef struct Product {
    Id id;
    char *name;
    RecognitionStrategy *recognition_strategy;
} Product;

typedef struct Contract {
    DomainObject base;
    Product *product;
    Money revenue;
    Date signed_on;
    List *recognitions;
} Contract;

void contract_calculate_recognitions(Contract *c) {
    c->product->recognition_strategy->calculate(c->product->recognition_strategy, c);
}
\end{lstlisting}

\begin{lstlisting}[language=C]
void complete_recognition_calculate(RecognitionStrategy *s, Contract *c) {
    contract_add_recognition(c, c->revenue, c->signed_on);
}
\end{lstlisting}

\begin{lstlisting}[language=C]
typedef struct ThreeWayStrategy {
    RecognitionStrategy base;
    int first_delay_days;
    int second_delay_days;
} ThreeWayStrategy;

void three_way_recognition_calculate(RecognitionStrategy *base, Contract *c) {
    ThreeWayStrategy *s = (ThreeWayStrategy *)base;
    Money part = money_divide(c->revenue, 3);
    Money last = money_subtract(c->revenue, money_times(part, 2));

    contract_add_recognition(c, part, c->signed_on);
    contract_add_recognition(c, part, date_add_days(c->signed_on, s->first_delay_days));
    contract_add_recognition(c, last, date_add_days(c->signed_on, s->second_delay_days));
}
\end{lstlisting}

\subsection{1.3 Table Module}\label{table-module}

A table module places behavior in one object per logical table or record
set, not one object per row.

\begin{lstlisting}[language=C]
typedef struct ContractModule { ResultSet *contracts; } ContractModule;
typedef struct ProductModule { ResultSet *products; } ProductModule;
typedef struct RecognitionModule { ResultSet *recognitions; } RecognitionModule;

void contract_module_calculate_recognitions(ContractModule *cm,
                                            ProductModule *pm,
                                            RecognitionModule *rm,
                                            Id contract_id) {
    Row contract = resultset_find_by_id(cm->contracts, contract_id);
    Row product = resultset_find_by_id(pm->products, row_id(contract, "product_id"));
    Money revenue = row_money(contract, "revenue");
    Date signed_on = row_date(contract, "signed_on");

    if (row_equals(product, "type", "word_processor")) {
        recognition_module_insert(rm, contract_id, revenue, signed_on);
    }
    else {
        int d1 = row_equals(product, "type", "spreadsheet") ? 60 : 30;
        int d2 = row_equals(product, "type", "spreadsheet") ? 90 : 60;
        Money part = money_divide(revenue, 3);
        recognition_module_insert(rm, contract_id, part, signed_on);
        recognition_module_insert(rm, contract_id, part, date_add_days(signed_on, d1));
        recognition_module_insert(rm, contract_id,
            money_subtract(revenue, money_times(part, 2)), date_add_days(signed_on, d2));
    }
}
\end{lstlisting}

\subsection{1.4 Service Layer}\label{service-layer}

A service layer defines an application API and centralizes
transaction/security handling.

\begin{lstlisting}[language=C]
Response service_execute(Request *req,
                         const char *permission,
                         void (*operation)(Request *req, Response *res)) {
    authorize(permission);
    Transaction *tx = begin_transaction();
    unit_of_work_start();
    Response res = response_new();

    try {
        operation(req, &res);
        unit_of_work_commit(current_uow());
        commit_transaction(tx);
        return res;
    }
    catch_any {
        rollback_transaction(tx);
        response_set_error(&res, current_error());
        return res;
    }
}
\end{lstlisting}

\section{2. Data Source Architectural

\subsection{2.1 Table Data Gateway}\label{table-data-gateway}

A table data gateway has one object/class per table. Methods wrap SQL
and return record sets or scalar results.

ResultSet *person_gateway_find_all(void) {

ResultSet *person_gateway_find_by_last_name(char *last_name) {

                    "from person where lastname = :last", p);

Id person_gateway_insert(char *last, char *first, int dependents) {

\subsection{2.3 Active Record}\label{active-record}

Active Record combines domain behavior and persistence behavior in the
same object.

\begin{lstlisting}[language=C]
typedef struct PersonAR {
    DomainObject base;
    char *last_name;
    char *first_name;
    int number_of_dependents;
} PersonAR;

void person_ar_save(PersonAR *p) {
    if (p->base.state == NEW) {
        person_ar_insert(p);
        p->base.state = CLEAN;
    }
    else if (p->base.state == DIRTY) {
        person_ar_update(p);
        p->base.state = CLEAN;
    }
}

PersonAR *person_ar_find(Id id) {
    PersonAR *cached = identity_map_get(current_identity_map(), "Person", id);
    if (cached != NULL) return cached;

    Map *params = map_new();
    map_put_id(params, "id", id);
    ResultSet *rs = db_query("select * from people where id = :id", params);
    if (!resultset_next(rs)) return NULL;

    PersonAR *p = person_ar_load(resultset_current(rs));
    identity_map_put(current_identity_map(), "Person", p->base.id, p);
    return p;
}
\end{lstlisting}

\subsection{2.4 Data Mapper}\label{data-mapper}

A data mapper keeps domain objects independent of database tables and
SQL.

\begin{lstlisting}[language=C]
typedef struct Mapper Mapper;
struct Mapper {
    char *class_name;
    char *table_name;
    DomainObject *(*load)(Mapper *m, Row row);
    void (*insert)(Mapper *m, DomainObject *obj);
    void (*update)(Mapper *m, DomainObject *obj);
    void (*remove)(Mapper *m, DomainObject *obj);
};

DomainObject *mapper_find(Mapper *m, Id id) {
    DomainObject *cached = identity_map_get(current_identity_map(), m->class_name, id);
    if (cached != NULL) return cached;

    Map *p = map_new();
    map_put_id(p, "id", id);
    ResultSet *rs = db_query(format("select * from %s where id = :id", m->table_name), p);
    if (!resultset_next(rs)) return NULL;

    DomainObject *obj = m->load(m, resultset_current(rs));
    identity_map_put(current_identity_map(), m->class_name, obj->id, obj);
    unit_of_work_register_clean(current_uow(), obj);
    return obj;
}
\end{lstlisting}

\begin{lstlisting}[language=C]
void person_mapper_update(Mapper *m, DomainObject *obj) {

    map_put_string(params, "first", p->first_name);
    db_execute("update people set lastname = :last, firstname = :first where id = :id", params);
}
\end{lstlisting}

\section{3. Object-Relational Behavioral
Patterns}\label{object-relational-behavioral-patterns}

\subsection{3.1 Unit of Work}\label{unit-of-work}

Unit of Work tracks all changed objects in a business transaction and
writes them in one commit.

\begin{lstlisting}[language=C]
typedef struct UnitOfWork {
    List *new_objects;
    List *dirty_objects;
    List *removed_objects;
    List *read_records;
    IdentityMap *identity_map;
} UnitOfWork;

void uow_register_new(UnitOfWork *uow, DomainObject *obj) {
    assert_true(obj != NULL, "object required");
    assert_true(obj->id != 0, "id required");
    assert_true(!list_contains(uow->dirty_objects, obj), "new object cannot be dirty");
    assert_true(!list_contains(uow->removed_objects, obj), "new object cannot be removed");
    assert_true(!list_contains(uow->new_objects, obj), "already new");
    list_add(uow->new_objects, obj);
    identity_map_put(uow->identity_map, object_class(obj), obj->id, obj);
}

void uow_register_dirty(UnitOfWork *uow, DomainObject *obj) {
    assert_true(obj != NULL, "object required");
    assert_true(obj->id != 0, "id required");
    assert_true(!list_contains(uow->removed_objects, obj), "removed object cannot be dirty");
    if (!list_contains(uow->dirty_objects, obj) && !list_contains(uow->new_objects, obj)) {
        list_add(uow->dirty_objects, obj);
    }
}

void uow_register_removed(UnitOfWork *uow, DomainObject *obj) {
    assert_true(obj != NULL, "object required");
    assert_true(obj->id != 0, "id required");
    if (list_contains(uow->new_objects, obj)) {
        list_remove(uow->new_objects, obj);
        identity_map_remove(uow->identity_map, object_class(obj), obj->id);
        return;
    }
    list_remove(uow->dirty_objects, obj);
    if (!list_contains(uow->removed_objects, obj)) list_add(uow->removed_objects, obj);
    identity_map_remove(uow->identity_map, object_class(obj), obj->id);
}

void uow_register_clean(UnitOfWork *uow, DomainObject *obj) {
    assert_true(obj != NULL, "object required");
    assert_true(obj->id != 0, "id required");
    identity_map_put(uow->identity_map, object_class(obj), obj->id, obj);
}
\end{lstlisting}

\begin{lstlisting}[language=C]
void uow_commit(UnitOfWork *uow) {
    Transaction *tx = begin_transaction();
    try {
        for_each(obj, uow->new_objects) {
            Mapper *m = mapper_registry_get(object_class(obj));
            m->insert(m, obj);
        }
        for_each(obj, uow->dirty_objects) {
            Mapper *m = mapper_registry_get(object_class(obj));
            m->update(m, obj);
        }
        for_each(obj, uow->removed_objects) {
            Mapper *m = mapper_registry_get(object_class(obj));
            m->remove(m, obj);
        }
        commit_transaction(tx);
        uow_clear(uow);
    }
    catch_any {
        rollback_transaction(tx);
        throw;
    }
}
\end{lstlisting}

\subsection{3.2 Identity Map}\label{identity-map}

Identity Map guarantees that each database row is represented by only
one in-memory object in a business transaction.

\begin{lstlisting}[language=C]
typedef struct IdentityMap { Map *objects; } IdentityMap;

char *identity_key(char *class_name, Id id) {
    return format("%s:%ld", class_name, id);
}

void *identity_map_get(IdentityMap *m, char *class_name, Id id) {
    return map_get(m->objects, identity_key(class_name, id));
}

void identity_map_put(IdentityMap *m, char *class_name, Id id, void *obj) {
    char *key = identity_key(class_name, id);
    void *existing = map_get(m->objects, key);
    if (existing != NULL && existing != obj) throw_error("duplicate identity");
    map_put(m->objects, key, obj);
}
\end{lstlisting}

\subsection{3.3 Lazy Load}\label{lazy-load}

Lazy Load delays loading an association or expensive field until the
field is needed.

\subsection{5.1 Metadata Mapping}\label{metadata-mapping}

\begin{lstlisting}[language=C]
typedef struct FieldMapping {
    char *field_name;
    char *column_name;
    char *target_class;
    char *relationship_kind;   // scalar, foreign_key, embedded, collection
    void (*set)(void *obj, void *value);
    void *(*get)(void *obj);
} FieldMapping;

typedef struct ClassMapping {
    char *class_name;
    char *table_name;
    FieldMapping *fields;
    int field_count;
    void *(*allocate)(void);
} ClassMapping;

void *metadata_load(ClassMapping *cm, Row row) {
    Id id = row_id(row, "id");
    void *cached = identity_map_get(current_identity_map(), cm->class_name, id);
    if (cached != NULL) return cached;

    void *obj = cm->allocate();
    set_domain_id(obj, id);
    identity_map_put(current_identity_map(), cm->class_name, id, obj);

    for (int i = 0; i < cm->field_count; i++) {
        FieldMapping *fm = &cm->fields[i];
        if (strcmp(fm->relationship_kind, "scalar") == 0) {
            fm->set(obj, row_value(row, fm->column_name));
        }
        else if (strcmp(fm->relationship_kind, "foreign_key") == 0) {
            Id target_id = row_id(row, fm->column_name);
            fm->set(obj, mapper_find_by_class(fm->target_class, target_id));
        }
        else if (strcmp(fm->relationship_kind, "embedded") == 0) {
            fm->set(obj, embedded_value_from_row(fm, row));
        }
    }
    return obj;
}
\end{lstlisting}

\subsection{5.2 Query Object}\label{query-object}

\begin{lstlisting}[language=C]
typedef enum { Q_EQ, Q_GT, Q_AND, Q_OR } QueryKind;

typedef struct QueryExpr {
    QueryKind kind;
    char *field;
    void *value;
    struct QueryExpr *left;
    struct QueryExpr *right;
} QueryExpr;

char *query_to_sql(ClassMapping *cm, QueryExpr *q, Map *params) {
    if (q->kind == Q_EQ) {
        FieldMapping *fm = class_mapping_field(cm, q->field);
        char *param = unique_param(q->field);
        map_put(params, param, q->value);
        return format("%s = :%s", fm->column_name, param);
    }
    if (q->kind == Q_GT) {
        FieldMapping *fm = class_mapping_field(cm, q->field);
        char *param = unique_param(q->field);
        map_put(params, param, q->value);
        return format("%s > :%s", fm->column_name, param);
    }
    if (q->kind == Q_AND) return format("(%s and %s)", query_to_sql(cm, q->left, params), query_to_sql(cm, q->right, params));
    if (q->kind == Q_OR)  return format("(%s or %s)", query_to_sql(cm, q->left, params), query_to_sql(cm, q->right, params));
    throw_error("unknown query expression");
}

List *query_execute(ClassMapping *cm, QueryExpr *where) {
    Map *params = map_new();
    char *sql_where = query_to_sql(cm, where, params);
    ResultSet *rs = db_query(format("select * from %s where %s", cm->table_name, sql_where), params);
    List *out = list_new();
    while (resultset_next(rs)) list_add(out, metadata_load(cm, resultset_current(rs)));
    return out;
}
\end{lstlisting}

\subsection{8.1 Optimistic Offline Lock}\label{optimistic-offline-lock}

Optimistic locking detects conflict during commit by comparing the
version read earlier with the current database version.

\begin{lstlisting}[language=C]
void mapper_update_with_optimistic_lock(DomainObject *obj, char *table, Map *values) {
    map_put_id(values, "id", obj->id);
    map_put_int(values, "old_version", obj->version);
    map_put_int(values, "new_version", obj->version + 1);

    int rows = db_execute(build_update_sql_with_version(table, values), values);
    if (rows != 1) throw_error("optimistic lock failure");
    obj->version++;
}
\end{lstlisting}

\begin{lstlisting}[language=C]
typedef struct ReadRecord { char *class_name; Id id; int version; } ReadRecord;

void uow_validate_read_set(UnitOfWork *uow) {
    for_each(r, uow->read_records) {
        int current = db_select_version(class_table(r->class_name), r->id);
        if (current != r->version) throw_error("inconsistent read detected");
    }
}

void uow_commit_optimistic(UnitOfWork *uow) {
    Transaction *tx = begin_transaction();
    try {
        uow_validate_read_set(uow);
        uow_insert_new(uow);
        uow_update_dirty(uow);
        uow_delete_removed(uow);
        commit_transaction(tx);
    }
    catch_any {
        rollback_transaction(tx);
        throw;
    }
}
\end{lstlisting}

\subsection{8.2 Pessimistic Offline
Lock}\label{pessimistic-offline-lock}

Pessimistic locking prevents conflict by acquiring an application lock
before edit.

\begin{lstlisting}[language=C]
int lock_try_acquire(char *type, Id id, Id owner_id) {
    Transaction *tx = begin_transaction();
    try {
        OfflineLock *existing = lock_table_find(type, id);
        if (existing == NULL) {
            lock_table_insert(type, id, owner_id, now());
            commit_transaction(tx);
            return 1;
        }
        if (existing->owner_id == owner_id) {
            commit_transaction(tx);
            return 1;
        }
        if (lock_expired(existing)) {
            lock_table_replace(type, id, owner_id, now());
            commit_transaction(tx);
            return 1;
        }
        rollback_transaction(tx);
        return 0;
    }
    catch_any {
        rollback_transaction(tx);
        throw;
    }
}

void lock_release(char *type, Id id, Id owner_id) {
    OfflineLock *existing = lock_table_find(type, id);
    if (existing != NULL && existing->owner_id == owner_id) lock_table_delete(type, id);
}
\end{lstlisting}

List *money_allocate(Money total, int *ratios, int n) {
    int ratio_sum = sum_ints(ratios, n);
    long cents = money_to_cents(total);
    long remainder = cents;
    List *parts = list_new();

    for (int i = 0; i < n; i++) {
        long share = cents * ratios[i] / ratio_sum;
        remainder -= share;
        list_add(parts, money_from_cents(share, total.currency));
    }
    for (int i = 0; remainder > 0; i = (i + 1) % n, remainder--) {
        money_increment_cent(list_get(parts, i));
    }
    return parts;
}

\subsection{11.1
Connection-per-transaction}\label{connection-per-transaction}

\begin{lstlisting}[language=C]
void execute_in_transaction(void (*work)(void)) {
    Connection *c = connection_pool_acquire();
    registry_put_scoped(THREAD_SCOPE, "db.connection", c);
    Transaction *tx = db_begin(c);
    try {
        work();
        db_commit(tx);
    }
    catch_any {
        db_rollback(tx);
        throw;
    }
    finally {
        registry_remove(THREAD_SCOPE, "db.connection");
        connection_pool_release(c);
    }
}
\end{lstlisting}

\subsection{11.2 Referential write
ordering}\label{referential-write-ordering}

\begin{lstlisting}[language=C]
List *topological_sort_writes(List *write_ops) {
    Graph *g = graph_new();
    for_each(op, write_ops) {
        graph_add_node(g, op);
        for_each(dep_table, op->depends_on_tables) {
            for_each(dep_op, writes_for_table(write_ops, dep_table)) {
                graph_add_edge(g, dep_op, op);
            }
        }
    }
    return graph_topological_sort(g);
}

void execute_ordered_writes(List *write_ops) {
    List *ordered = topological_sort_writes(write_ops);
    for_each(op, ordered) op->execute(op);
}
\end{lstlisting}

\subsection{11.3 Full enterprise request
skeleton}\label{full-enterprise-request-skeleton}

\begin{lstlisting}[language=C]
Response enterprise_request(Request *req) {
    authenticate(req);
    authorize_request(req);
    unit_of_work_start();

    Transaction *tx = begin_transaction();
    try {
        Command *cmd = route_to_command(req);
        ModelResult model = cmd->execute(req);
        unit_of_work_commit(current_uow());
        commit_transaction(tx);

        View *view = select_view(model);
        return view_render(view, model);
    }
    catch_any {
        rollback_transaction(tx);
        return error_response(current_error());
    }
    finally {
        unit_of_work_clear_current();
    }
}

\section{12. Pattern Selection Heuristics as
Algorithms}\label{pattern-selection-heuristics-as-algorithms}

\begin{lstlisting}[language=C]
Pattern choose_domain_logic_pattern(int domain_complexity,
                                    int recordset_tooling_strength,
                                    int team_domain_model_skill) {
    if (domain_complexity <= 3) {
        return recordset_tooling_strength >= 7 ? TABLE_MODULE : TRANSACTION_SCRIPT;
    }
    if (domain_complexity <= 6) {
        if (recordset_tooling_strength >= 7) return TABLE_MODULE;
        if (team_domain_model_skill >= 6) return DOMAIN_MODEL;
        return TRANSACTION_SCRIPT;
    }
    return DOMAIN_MODEL;
}

Pattern choose_data_source_pattern(Pattern domain_pattern,
                                   int object_table_match,
                                   int mapping_complexity) {
    if (domain_pattern == TRANSACTION_SCRIPT) return TABLE_DATA_GATEWAY;
    if (domain_pattern == TABLE_MODULE) return TABLE_DATA_GATEWAY;
    if (domain_pattern == DOMAIN_MODEL && object_table_match && mapping_complexity <= 3) return ACTIVE_RECORD;
    if (domain_pattern == DOMAIN_MODEL) return DATA_MAPPER;
    return TABLE_DATA_GATEWAY;
}

Pattern choose_offline_lock_pattern(int conflict_frequency,
                                    int conflict_cost,
                                    int liveness_priority) {
    if (conflict_frequency <= LOW && liveness_priority >= MEDIUM) return OPTIMISTIC_OFFLINE_LOCK;
    if (conflict_cost >= HIGH) return PESSIMISTIC_OFFLINE_LOCK;
    return OPTIMISTIC_OFFLINE_LOCK;
}

Pattern choose_session_state_pattern(int state_size,
                                     int state_sensitivity,
                                     int cluster_migration_required) {
    if (state_size <= SMALL && state_sensitivity == LOW) return CLIENT_SESSION_STATE;
    if (cluster_migration_required) return DATABASE_SESSION_STATE;
    return SERVER_SESSION_STATE;
}
\end{lstlisting}

\section{13. Summary}\label{summary}

The implementation algorithms in \emph{Patterns of Enterprise
Application Architecture} are less about numerical computation and more
about preserving boundaries in mutable, persistent, concurrent
enterprise systems. The core reusable algorithms are:

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  Route requests through controllers into domain/application services.
\item
  Keep domain logic out of views and SQL plumbing.
\item
  Encapsulate table access with gateways or isolate object persistence
  with mappers.
\item
  Preserve in-memory identity with Identity Map.
\item
  Track writes with Unit of Work.
\item
  Use Lazy Load to balance object graph completeness against database
  round trips.
\item
  Map associations, inheritance, and value objects explicitly.
\item
  Use metadata and query objects to remove repetitive mapping code.
\item
  Use DTOs and Remote Facades to reduce remote-call chatiness.
\item
  Use optimistic or pessimistic offline locks when business transactions
  span system transactions.
\item
  Store session state deliberately: client, server, or database.
\item
  Use base patterns---Gateway, Mapper, Registry, Value Object, Special
  Case, Plugin, Service Stub, and Record Set---to localize variability
  and integration complexity.
\end{enumerate}

\section{Design Patterns --- Systematic Algorithm Extraction in C-like Pseudocode}\label{design-patterns-systematic-algorithm-extraction-in-c-like-pseudocode}

\textbf{Source analyzed:} Erich Gamma, Richard Helm, Ralph Johnson, and John Vlissides, \emph{Design Patterns: Elements of Reusable Object-Oriented Software}.

\textbf{Extraction scope.} The book is a pattern catalog, not an algorithms textbook. Its executable material appears mainly as C++ and Smalltalk examples plus the Lexi case study and foundation collection classes. This document extracts the algorithmic behavior from those examples and rewrites it as independent, C-like pseudocode. It does not transcribe the original source code; it captures the control flow, object collaboration, state transitions, traversal logic, construction protocols, and extension points behind the examples.

\textbf{Core idea.} Most algorithms in the book are object-collaboration algorithms: one object delegates, creates, wraps, traverses, notifies, routes, or interprets through an interface. The pseudocode therefore uses C structs and function pointers to approximate object-oriented dispatch.

\section{Part I --- Lexi Case Study Algorithms}\label{part-i-lexi-case-study-algorithms}

The Lexi case study is the book's most integrated example. It introduces recursive document structure, replaceable formatting algorithms, UI embellishment, portable widgets/windows, undoable commands, and traversal/visitor mechanisms.

\subsection{2. Recursive Glyph Composition}\label{recursive-glyph-composition}

A document is represented as a tree of \passthrough{\lstinline!Glyph!} objects. Leaves are characters, images, and primitive shapes; composites are rows, columns, pages, and other structural containers.

\begin{lstlisting}[language=C]
typedef struct GlyphVTable GlyphVTable;

typedef struct Glyph {
    GlyphVTable *ops;
    struct Glyph *parent;
} Glyph;

struct GlyphVTable {
    void (*draw)(Glyph *self, Window *w);
    void (*bounds)(Glyph *self, Rect *out);
    bool (*intersects)(Glyph *self, Point p);
    void (*insert)(Glyph *self, Glyph *child, int index);
    void (*remove)(Glyph *self, Glyph *child);
    Glyph *(*child)(Glyph *self, int index);
    int (*child_count)(Glyph *self);
    Iterator *(*create_iterator)(Glyph *self);
    void (*accept)(Glyph *self, void *visitor);
};
\end{lstlisting}

    CompositeGlyph *g = (CompositeGlyph*) self;

    }

\end{lstlisting}

\subsubsection{2.4 Hit detection over a glyph tree}\label{hit-detection-over-a-glyph-tree}

\begin{lstlisting}[language=C]
Glyph *glyph_at_point(Glyph *root, Point p) {
    if (!root->ops->intersects(root, p)) {
        return NULL_PTR;
    }

    for (int i = root->ops->child_count(root) - 1; i >= 0; --i) {
        Glyph *child = root->ops->child(root, i);
        Glyph *hit = glyph_at_point(child, p);
        if (hit != NULL_PTR) {
            return hit;
        }
    }

    return root;
}
\end{lstlisting}

\textbf{Pattern role:} Composite. The algorithm treats leaves and composites through the same \passthrough{\lstinline!Glyph!} interface.

\subsection{3. Formatting by Strategy: Composition and Compositor}\label{formatting-by-strategy-composition-and-compositor}

A \passthrough{\lstinline!Composition!} owns visible glyphs and delegates line breaking to a \passthrough{\lstinline!Compositor!} strategy. Different compositors implement different quality/performance trade-offs.

    CompositorVTable *ops;
    Composition *composition;

struct Composition {
    CompositeGlyph base;
    Compositor *compositor;
};
\end{lstlisting}

\subsubsection{3.1 Composition repair algorithm}\label{composition-repair-algorithm}

\begin{lstlisting}[language=C]
void composition_repair(Composition *c) {
    remove_existing_layout_glyphs(c);
    c->compositor->ops->set_composition(c->compositor, c);
    c->compositor->ops->compose(c->compositor);
    mark_clean(c);
}
\end{lstlisting}

\subsubsection{3.2 Simple greedy compositor}\label{simple-greedy-compositor}

\begin{lstlisting}[language=C]
void simple_compositor_compose(Compositor *self) {
    Composition *c = self->composition;
    RowGlyph *current_row = row_create();
    int width = 0;

    for (Iterator *it = glyph_children(c); !iterator_done(it); iterator_next(it)) {
        Glyph *g = iterator_current(it);
        int glyph_width = glyph_measure_width(g);

        if (width + glyph_width > c->target_line_width && row_child_count(current_row) > 0) {
            composition_append_row(c, current_row);
            current_row = row_create();
            width = 0;
        }

        row_append(current_row, g);
        width += glyph_width;
    }

    if (row_child_count(current_row) > 0) {
        composition_append_row(c, current_row);
    }
}
\end{lstlisting}

\subsubsection{3.3 TeX-like dynamic-programming compositor skeleton}\label{tex-like-dynamic-programming-compositor-skeleton}

The book references a TeX-style algorithm as one possible strategy. The pattern-level algorithm is to encapsulate this algorithm behind the same compositor interface.

\begin{lstlisting}[language=C]
void tex_like_compositor_compose(Compositor *self) {
    Composition *c = self->composition;
    BreakGraph graph = build_possible_line_breaks(c->visible_glyphs);

    for (int i = 0; i < graph.break_count; ++i) {
        graph.cost[i] = INFINITY;
        graph.previous[i] = -1;
    }
    graph.cost[0] = 0;

    for (int i = 0; i < graph.break_count; ++i) {
        for (int j = i + 1; j < graph.break_count; ++j) {
            LineMetrics m = measure_segment(graph, i, j);
            if (!line_feasible(m, c->target_line_width)) {
                continue;
            }

            int badness = line_badness(m, c->target_line_width);
            int candidate = graph.cost[i] + badness + penalty_at_break(graph, j);
            if (candidate < graph.cost[j]) {
                graph.cost[j] = candidate;
                graph.previous[j] = i;
            }
        }
    }

    materialize_rows_from_breakpoints(c, graph.previous);
}
\end{lstlisting}

\textbf{Pattern role:} Strategy. \passthrough{\lstinline!Composition!} is stable while line-breaking algorithms vary independently.

void path_window_imp_device_rect(WindowImp *self, double x0, double y0, double x1, double y1) {
    PathWindowImp *p = (PathWindowImp*) self;

    Point corners[4];
    corners[0] = point(min_double(x0, x1), max_double(y0, y1));
    corners[1] = point(max_double(x0, x1), max_double(y0, y1));
    corners[2] = point(max_double(x0, x1), min_double(y0, y1));
    corners[3] = point(min_double(x0, x1), min_double(y0, y1));

    path_begin(p->device);
    path_move_to(p->device, corners[3]);
    path_polyline(p->device, corners, 4);
    path_close(p->device);
    path_stroke(p->device);
}
\end{lstlisting}

\subsection{7. Command Execution and Undo History}\label{command-execution-and-undo-history}

\end{lstlisting}

\subsubsection{7.1 Menu item dispatch}\label{menu-item-dispatch}

\begin{lstlisting}[language=C]
void menu_item_clicked(MenuItem *item) {
    Command *cmd = item->command;
    cmd->ops->execute(cmd);

\end{lstlisting}

\subsubsection{7.2 Paste command}\label{paste-command}

\begin{lstlisting}[language=C]
void paste_command_execute(Command *self) {
    PasteCommand *cmd = (PasteCommand*) self;
    cmd->insertion_position = document_current_position(cmd->document);
    cmd->inserted_glyphs = clipboard_clone_contents(cmd->clipboard);

\subsubsection{7.3 Command history with arbitrary undo/redo}\label{command-history-with-arbitrary-undoredo}

\begin{lstlisting}[language=C]
typedef struct CommandHistory {
    List *commands;
    int present;          // index after the most recently executed command
} CommandHistory;

void command_history_record_if_reversible(CommandHistory *h, Command *cmd) {
    if (!cmd->ops->reversible(cmd)) {
        return;
    }

    while (list_size(h->commands) > h->present) {
        Command *discarded = list_remove_last(h->commands);
        release_command(discarded);
    }

    list_append(h->commands, cmd);
    h->present = list_size(h->commands);
}

bool command_history_undo(CommandHistory *h) {
    if (h->present == 0) {
        return false;
    }

    h->present -= 1;
    Command *cmd = list_get(h->commands, h->present);
    cmd->ops->unexecute(cmd);
    return true;
}

bool command_history_redo(CommandHistory *h) {
    if (h->present >= list_size(h->commands)) {
        return false;
    }

    Command *cmd = list_get(h->commands, h->present);
    cmd->ops->execute(cmd);
    h->present += 1;
    return true;
}
\end{lstlisting}

\textbf{Pattern role:} Command. Requests become objects that can be queued, logged, undone, and reused across UI controls.

\subsection{8. Iterator Traversal over Glyph Trees}\label{iterator-traversal-over-glyph-trees}

Iterators separate traversal state from the glyph structure.

\begin{lstlisting}[language=C]
typedef struct IteratorVTable {
    void (*first)(Iterator *self);
    void (*next)(Iterator *self);
    bool (*is_done)(Iterator *self);
    Glyph *(*current_item)(Iterator *self);
} IteratorVTable;

struct Iterator {
    IteratorVTable *ops;
};
\end{lstlisting}

\subsubsection{8.1 List iterator}\label{list-iterator}

\begin{lstlisting}[language=C]
typedef struct ListIterator {
    Iterator base;
    List *items;
    int index;
} ListIterator;

void list_iterator_first(Iterator *it) {
    ((ListIterator*) it)->index = 0;
}

void list_iterator_next(Iterator *it) {
    ((ListIterator*) it)->index += 1;
}

bool list_iterator_done(Iterator *it) {
    ListIterator *li = (ListIterator*) it;
    return li->index >= list_size(li->items);
}

Glyph *list_iterator_current(Iterator *it) {
    ListIterator *li = (ListIterator*) it;
    if (list_iterator_done(it)) {
        return NULL_PTR;
    }
    return list_get(li->items, li->index);
}
\end{lstlisting}

\subsubsection{8.2 Null iterator for leaves}\label{null-iterator-for-leaves}

\begin{lstlisting}[language=C]
void null_iterator_first(Iterator *it) { }
void null_iterator_next(Iterator *it) { }
bool null_iterator_done(Iterator *it) { return true; }
Glyph *null_iterator_current(Iterator *it) { return NULL_PTR; }
\end{lstlisting}

\subsubsection{8.3 Preorder iterator using a stack of child iterators}\label{preorder-iterator-using-a-stack-of-child-iterators}

\begin{lstlisting}[language=C]
typedef struct PreorderIterator {
    Iterator base;
    Glyph *root;
    Stack *iterators;
    bool before_root;
} PreorderIterator;

void preorder_first(Iterator *it) {
    PreorderIterator *p = (PreorderIterator*) it;
    stack_clear(p->iterators);
    p->before_root = true;
}

Glyph *preorder_current(Iterator *it) {
    PreorderIterator *p = (PreorderIterator*) it;

    if (p->before_root) {
        return p->root;
    }
    if (stack_empty(p->iterators)) {
        return NULL_PTR;
    }
    Iterator *top = stack_top(p->iterators);
    return top->ops->current_item(top);
}

void preorder_next(Iterator *it) {
    PreorderIterator *p = (PreorderIterator*) it;
    Glyph *current = preorder_current(it);

    if (current == NULL_PTR) {
        return;
    }

    if (p->before_root) {
        p->before_root = false;
        Iterator *root_it = current->ops->create_iterator(current);
        root_it->ops->first(root_it);
        if (!root_it->ops->is_done(root_it)) {
            stack_push(p->iterators, root_it);
            return;
        }
        release_iterator(root_it);
    } else {
        Iterator *child_it = current->ops->create_iterator(current);
        child_it->ops->first(child_it);
        if (!child_it->ops->is_done(child_it)) {
            stack_push(p->iterators, child_it);
            return;
        }
        release_iterator(child_it);
        stack_top(p->iterators)->ops->next(stack_top(p->iterators));
    }

    while (!stack_empty(p->iterators) && stack_top(p->iterators)->ops->is_done(stack_top(p->iterators))) {
        Iterator *finished = stack_pop(p->iterators);
        release_iterator(finished);
        if (!stack_empty(p->iterators)) {
            stack_top(p->iterators)->ops->next(stack_top(p->iterators));
        }
    }
}

bool preorder_done(Iterator *it) {
    PreorderIterator *p = (PreorderIterator*) it;
    return !p->before_root && stack_empty(p->iterators);
}
\end{lstlisting}

\textbf{Pattern role:} Iterator. Each traversal carries its own state, allowing multiple simultaneous traversals over the same structure.

\subsubsection{9.1 Generic visitor traversal}\label{generic-visitor-traversal}

\begin{lstlisting}[language=C]
void apply_visitor_preorder(Glyph *root, Visitor *visitor) {
    Iterator *it = preorder_iterator_create(root);

    for (it->ops->first(it); !it->ops->is_done(it); it->ops->next(it)) {
        Glyph *g = it->ops->current_item(it);
        g->ops->accept(g, visitor);
    }

    release_iterator(it);
}
\end{lstlisting}

\subsubsection{9.2 Spelling visitor}\label{spelling-visitor}

\begin{lstlisting}[language=C]
typedef struct SpellingVisitor {
    Visitor base;
    char current_word[MAX_WORD_SIZE];
    List *misspellings;
} SpellingVisitor;

void spelling_visit_character(Visitor *self, CharacterGlyph *ch) {
    SpellingVisitor *sp = (SpellingVisitor*) self;
    char c = character_code(ch);

    if (is_alpha(c)) {
        append_char(sp->current_word, c);
        return;
    }

    if (string_length(sp->current_word) > 0) {
        if (dictionary_is_misspelled(sp->current_word)) {
            list_append(sp->misspellings, string_copy(sp->current_word));
        }
        clear_string(sp->current_word);
    }
}

void spelling_finish(SpellingVisitor *sp) {
    if (string_length(sp->current_word) > 0 && dictionary_is_misspelled(sp->current_word)) {
        list_append(sp->misspellings, string_copy(sp->current_word));
    }
    clear_string(sp->current_word);
}
\end{lstlisting}

\subsubsection{9.3 Hyphenation visitor}\label{hyphenation-visitor}

\begin{lstlisting}[language=C]
typedef struct HyphenationVisitor {
    Visitor base;
    char current_word[MAX_WORD_SIZE];
    List *current_word_glyphs;
} HyphenationVisitor;

void hyphenation_visit_character(Visitor *self, CharacterGlyph *ch) {
    HyphenationVisitor *hv = (HyphenationVisitor*) self;
    char c = character_code(ch);

    if (is_alpha(c)) {
        append_char(hv->current_word, c);
        list_append(hv->current_word_glyphs, ch);
        return;
    }

    hyphenation_flush_word(hv);
}

void hyphenation_flush_word(HyphenationVisitor *hv) {
    List *breaks = hyphenation_points(hv->current_word);

    for (int i = list_size(breaks) - 1; i >= 0; --i) {
        int offset = list_get_int(breaks, i);
        Glyph *after = list_get(hv->current_word_glyphs, offset);
        Glyph *disc = discretionary_glyph_create();
        glyph_insert_after(after->parent, after, disc);
    }

    clear_string(hv->current_word);
    list_clear(hv->current_word_glyphs);
}
\end{lstlisting}

\textbf{Pattern role:} Visitor. Adding a new analysis means adding a visitor subclass, not changing the glyph hierarchy.

void abstraction_operation(Abstraction *a) {
    a->impl->ops->operation_impl(a->impl);
}

void refined_abstraction_operation(RefinedAbstraction *a) {
    before_hook(a);
    a->base.impl->ops->operation_impl(a->base.impl);
    after_hook(a);
}
\end{lstlisting}

A bridge is used in the Lexi window example earlier, where \passthrough{\lstinline!Window!} delegates to \passthrough{\lstinline!WindowImp!}.

\textbf{Pattern role:} Bridge. Both hierarchies can be extended without multiplying subclasses.

\subsection{19. Composite}\label{composite}

Composite objects implement the same interface as leaves and recursively forward operations to children.

\subsubsection{19.1 Equipment example: price aggregation}\label{equipment-example-price-aggregation}

\begin{lstlisting}[language=C]
typedef struct EquipmentVTable {
    Money (*net_price)(void *self);
    Money (*discount_price)(void *self);
    void (*add)(void *self, void *equipment);
    void (*remove)(void *self, void *equipment);
    Iterator *(*create_iterator)(void *self);
} EquipmentVTable;

typedef struct Equipment {
    EquipmentVTable *ops;
    char *name;
} Equipment;

Money equipment_default_net_price(Equipment *e) {
    return money_zero();
}

Money composite_equipment_net_price(Equipment *self) {
    CompositeEquipment *c = (CompositeEquipment*) self;
    Money total = money_zero();

    Iterator *it = list_iterator_create(c->children);
    for (it->ops->first(it); !it->ops->is_done(it); it->ops->next(it)) {
        Equipment *child = it->ops->current_item(it);
        total = money_add(total, child->ops->net_price(child));
    }

    release_iterator(it);
    return total;
}
\end{lstlisting}

\subsubsection{19.2 Composite traversal template}\label{composite-traversal-template}

\begin{lstlisting}[language=C]
void composite_apply(Composite *c, void (*operation)(Component *x)) {
    operation((Component*) c);

    for (Iterator *it = component_iterator(c); !iterator_done(it); iterator_next(it)) {
        Component *child = iterator_current(it);
        child->ops->apply(child, operation);
    }
}
\end{lstlisting}

\textbf{Pattern role:} Composite. Whole-part structures are processed uniformly.

\subsection{21. Facade}\label{facade}

\subsubsection{21.1 Compiler facade pipeline}\label{compiler-facade-pipeline}

\subsection{24. Chain of Responsibility}\label{chain-of-responsibility}

A request is passed along a chain until one handler accepts responsibility.

\begin{lstlisting}[language=C]
typedef struct HelpHandler {
    struct HelpHandler *successor;
    HelpTopic topic;
} HelpHandler;

    return h->topic != NO_HELP_TOPIC;
}

void handle_help(HelpHandler *h) {
    if (has_help(h)) {
        display_help_topic(h->topic);
        return;
    }

    if (h->successor != NULL_PTR) {
        handle_help(h->successor);
        return;
    }

    display_no_help_available();
}
\end{lstlisting}

\subsubsection{24.1 Widget event propagation}\label{widget-event-propagation}

\begin{lstlisting}[language=C]
void widget_help_request(Widget *w) {
    handle_help((HelpHandler*) w);
}

void button_clicked(Button *b) {
    if (button_has_local_action(b)) {
        execute_button_action(b);
    } else {
        handle_help((HelpHandler*) b);
    }
}
\end{lstlisting}

\textbf{Pattern role:} Chain of Responsibility. Sender is decoupled from the receiver that ultimately handles the request.

\subsection{26. Interpreter}\label{interpreter}

Interpreter represents grammar rules as expression objects and evaluates sentences using a context.

\subsubsection{26.1 Boolean expression interface}\label{boolean-expression-interface}

\begin{lstlisting}[language=C]
typedef struct BooleanExpVTable {
    bool (*evaluate)(void *self, Context *ctx);
    void *(*replace)(void *self, const char *name, void *replacement);
    void *(*copy)(void *self);
} BooleanExpVTable;

typedef struct BooleanExp {
    BooleanExpVTable *ops;
} BooleanExp;
\end{lstlisting}

\subsubsection{26.2 Variable expression}\label{variable-expression}

\begin{lstlisting}[language=C]
typedef struct VariableExp {
    BooleanExp base;
    char *name;
} VariableExp;

bool variable_evaluate(BooleanExp *base, Context *ctx) {
    VariableExp *v = (VariableExp*) base;
    return context_lookup(ctx, v->name);
}

BooleanExp *variable_replace(BooleanExp *base, const char *name, BooleanExp *replacement) {
    VariableExp *v = (VariableExp*) base;
    if (string_equals(v->name, name)) {
        return replacement->ops->copy(replacement);
    }
    return variable_exp_create(v->name);
}
\end{lstlisting}

\subsubsection{26.3 Constant expression}\label{constant-expression}

\begin{lstlisting}[language=C]
bool constant_evaluate(BooleanExp *base, Context *ctx) {
    ConstantExp *c = (ConstantExp*) base;
    return c->value;
}
\end{lstlisting}

\subsubsection{26.4 Composite boolean expressions}\label{composite-boolean-expressions}

\begin{lstlisting}[language=C]
bool and_evaluate(BooleanExp *base, Context *ctx) {
    AndExp *e = (AndExp*) base;
    return e->left->ops->evaluate(e->left, ctx) &&
           e->right->ops->evaluate(e->right, ctx);
}

bool or_evaluate(BooleanExp *base, Context *ctx) {
    OrExp *e = (OrExp*) base;
    return e->left->ops->evaluate(e->left, ctx) ||
           e->right->ops->evaluate(e->right, ctx);
}

bool not_evaluate(BooleanExp *base, Context *ctx) {
    NotExp *e = (NotExp*) base;
    return !e->operand->ops->evaluate(e->operand, ctx);
}
\end{lstlisting}

\subsubsection{26.5 Context assignment and evaluation}\label{context-assignment-and-evaluation}

\begin{lstlisting}[language=C]
void context_assign(Context *ctx, const char *variable, bool value) {
    map_put_bool(ctx->bindings, variable, value);
}

bool context_lookup(Context *ctx, const char *variable) {
    if (!map_contains(ctx->bindings, variable)) {
        error("unbound variable");
    }
    return map_get_bool(ctx->bindings, variable);
}

bool evaluate_sentence(BooleanExp *syntax_tree, Context *ctx) {
    return syntax_tree->ops->evaluate(syntax_tree, ctx);
}
\end{lstlisting}

\textbf{Pattern role:} Interpreter. Grammar rules map to classes; interpretation is recursive evaluation over the expression tree.

\subsection{27. Iterator}\label{iterator}

The Iterator pattern supplies sequential access without exposing aggregate representation.

\subsubsection{27.1 Generic external iterator}\label{generic-external-iterator}

\begin{lstlisting}[language=C]
typedef struct IteratorOps {
    void (*first)(void *self);
    void (*next)(void *self);
    bool (*is_done)(void *self);
    void *(*current_item)(void *self);
} IteratorOps;

typedef struct ExternalIterator {
    IteratorOps *ops;
} ExternalIterator;
\end{lstlisting}

\subsubsection{27.2 List traversal}\label{list-traversal}

\begin{lstlisting}[language=C]
void traverse_list(List *list, void (*visit)(void *item)) {
    ExternalIterator *it = list_create_iterator(list);

    for (it->ops->first(it); !it->ops->is_done(it); it->ops->next(it)) {
        visit(it->ops->current_item(it));

\begin{lstlisting}[language=C]
void list_do(List *list, void (*operation)(void *item, void *ctx), void *ctx) {

        operation(list_get(list, i), ctx);
    }
}
\end{lstlisting}

\subsubsection{27.4 Filtering iterator skeleton}\label{filtering-iterator-skeleton}

\begin{lstlisting}[language=C]
void filtering_iterator_advance_to_match(FilteringIterator *it) {
    while (!it->inner->ops->is_done(it->inner)) {
        void *item = it->inner->ops->current_item(it->inner);
        if (it->predicate(item)) {
            break;
        }
        it->inner->ops->next(it->inner);
    }
}
\end{lstlisting}

\textbf{Pattern role:} Iterator. Traversal algorithms are separated from aggregate storage algorithms.

\subsubsection{32.2 First-fit strategy}\label{first-fit-strategy}

\begin{lstlisting}[language=C]
int first_fit_compose(CompositorStrategy *s,
                      Coordinate *natural,
                      Coordinate *stretch,
                      Coordinate *shrink,
                      int n,
                      int line_width,
                      int *breaks_out) {
    int width = 0;
    int break_count = 0;

    for (int i = 0; i < n; ++i) {
        if (width + natural[i] > line_width) {
            breaks_out[break_count++] = i;
            width = 0;
        }
        width += natural[i];
    }

    breaks_out[break_count++] = n;
    return break_count;
}

\textbf{Pattern role:} Strategy. Algorithms vary independently from the object that uses them.

\subsection{44. List Core Operations}\label{list-core-operations}

\begin{lstlisting}[language=C]
typedef struct ListNode {
    void *item;
    struct ListNode *next;
    struct ListNode *prev;
} ListNode;

typedef struct List {
    ListNode *head;
    ListNode *tail;
    int size;
} List;

void list_append(List *list, void *item) {
    ListNode *node = allocate(sizeof(*node));
    node->item = item;
    node->next = NULL_PTR;
    node->prev = list->tail;

    if (list->tail != NULL_PTR) {
        list->tail->next = node;
    } else {
        list->head = node;
    }

    list->tail = node;
    list->size += 1;
}

void *list_get(List *list, int index) {
    if (index < 0 || index >= list->size) {
        error("index out of range");
    }

    ListNode *node;
    if (index < list->size / 2) {
        node = list->head;
        for (int i = 0; i < index; ++i) {
            node = node->next;
        }
    } else {
        node = list->tail;
        for (int i = list->size - 1; i > index; --i) {
            node = node->prev;
        }
    }

    return node->item;
}

void list_remove(List *list, void *item) {
    for (ListNode *node = list->head; node != NULL_PTR; node = node->next) {
        if (node->item == item) {
            if (node->prev != NULL_PTR) node->prev->next = node->next;
            else list->head = node->next;

            if (node->next != NULL_PTR) node->next->prev = node->prev;
            else list->tail = node->prev;

            release(node);
            list->size -= 1;
            return;
        }
    }
}
\end{lstlisting}

\subsection{46. Point and Rectangle Geometry}\label{point-and-rectangle-geometry}

typedef struct Point {
    double x;
    double y;
} Point;

typedef struct Rect {
    Point min;
    Point max;
} Rect;

bool rect_contains_point(Rect r, Point p) {
    return p.x >= r.min.x && p.x <= r.max.x &&
           p.y >= r.min.y && p.y <= r.max.y;
}

bool rect_intersects(Rect a, Rect b) {
    if (a.max.x < b.min.x) return false;
    if (b.max.x < a.min.x) return false;
    if (a.max.y < b.min.y) return false;
    if (b.max.y < a.min.y) return false;
    return true;
}

Rect rect_union(Rect a, Rect b) {
    Rect r;
    r.min.x = min_double(a.min.x, b.min.x);
    r.min.y = min_double(a.min.y, b.min.y);
    r.max.x = max_double(a.max.x, b.max.x);
    r.max.y = max_double(a.max.y, b.max.y);
    return r;
}

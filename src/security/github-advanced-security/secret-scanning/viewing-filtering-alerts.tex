@startuml
title Viewing and Filtering Alerts from Secret Scanning

' Activity / BPMN-style flow with swimlanes

partition "User\n(Repo/Org Owner,\nSecurity Manager, Admin)" {
  start
  :Need to view or triage\nsecret scanning alerts;
}

partition "GitHub" {
  :Repository has secret scanning\nenabled and is eligible\n(public or org-owned with\nSecret Protection);
  if ("Secret scanning enabled\nfor this repository?") then (Yes)
    :Alerts can be generated and\nshown in the Security tab;
  else (No)
    :No secret scanning alerts\navailable to view;
    stop
  endif
}

partition "User\n(Repo/Org Owner,\nSecurity Manager, Admin)" {
  :Navigate to main page\nof the repository;
  :Click **Security** tab\n(or dropdown → Security);
}

partition "GitHub" {
  :Display repository Security view;
}

partition "User\n(Repo/Org Owner,\nSecurity Manager, Admin)" {
  :In left sidebar under\n\"Vulnerability alerts\",\nclick **Secret scanning**;
}

partition "GitHub" {
  :Display Secret scanning alerts page;
  note right
    Alerts are separated into two lists:
    - Default alerts (supported & custom patterns)
    - Generic alerts (non-provider patterns
      and AI-detected generic secrets)

    Generic alerts:
    - Limited to 5000 per repository
    - Only visible in Secret scanning view
    - Show limited locations per alert
  end note
}

partition "User\n(Repo/Org Owner,\nSecurity Manager, Admin)" {
  if ("Which list to view?") then ("Default")
    :View **Default** alerts list
     (main alerts view);
  else ("Generic")
    :Toggle to **Generic** alerts list
     for non-provider and AI-detected\nsecrets;
  endif

  :Click an alert to open\ndetailed alert view;
  :Optionally assign alert to a\nuser or team via **Assignees**;
}

partition "User\n(Repo/Org Owner,\nSecurity Manager, Admin)" {
  :Apply filters to refine alerts;
}

partition "GitHub" {
  :Update alerts list based on\nselected filters or search\nqualifiers;
  note right
    Example filter qualifiers:
    - bypassed:true
    - is:open / is:closed /
      is:publicly-leaked /
      is:multi-repository
    - props:CUSTOM_PROPERTY_NAME:VALUE
    - provider:PROVIDER-NAME
    - repo:OWNER/REPOSITORY
    - resolution:false-positive /
      hidden-by-config /
      pattern-deleted /
      pattern-edited /
      revoked /
      used-in-tests /
      wont-fix
    - results:default / results:generic
    - secret-type:SECRET-NAME
    - sort:created-desc / created-asc /
      updated-desc / updated-asc
    - team:ORG/TEAM-SLUG
    - topic:TOPIC
    - validity:active / inactive / unknown
  end note
}

partition "User\n(Repo/Org Owner,\nSecurity Manager, Admin)" {
  :Review filtered alert set\nand plan remediation actions;
  stop
}

@enduml

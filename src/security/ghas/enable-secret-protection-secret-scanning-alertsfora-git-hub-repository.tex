@startuml
title Enable Secret Protection (Secret Scanning Alerts) for a GitHub Repository

partition "Repository Owner / Admin" {
  start
  :Decide to enable\nSecret Protection\n(secret scanning alerts);
  :Navigate to repository\nmain page on GitHub;
  :Click **Settings** tab\n(or dropdown → Settings);
  :In sidebar, click\n**Security → Advanced Security**;
}

partition "GitHub" {
  :Display Advanced Security\nsettings for repository;

  note right
    Secret scanning is available for:
    - Public repositories on GitHub.com
    - Organization-owned repositories
      on GitHub Team with
      Secret Protection enabled.
  end note

  if ("Repository eligible\nfor Secret Protection?") then (Yes)
    :Show **Secret Protection**\ncontrol with Enable button;
  else (No)
    :Indicate that Secret Protection\nis not available for this repository;
    stop
  endif
}

partition "Repository Owner / Admin" {
  :Click **Enable**\nnext to Secret Protection;
  :Review impact of enabling\nSecret Protection;
  :Click **Enable Secret Protection**\nto confirm;
}

partition "GitHub" {
  :Activate Secret Protection\nand start secret scanning\nalerts for users;

  note right
    A repository administrator
    can later disable secret scanning
    from the same settings page.
  end note

  stop
}

@enduml
